001package org.gbif.common.parsers;
002
003import com.google.common.base.Strings;
004import com.google.common.io.Resources;
005import org.gbif.api.vocabulary.NomenclaturalStatus;
006import org.gbif.common.parsers.core.ParseResult;
007import org.junit.Test;
008
009import java.io.IOException;
010import java.nio.charset.Charset;
011
012import static org.junit.Assert.assertTrue;
013
014/**
015 * Parses a single nomenclatural status out of a given string.
016 * TODO: Improve parser and return a status set instead, see nomStatus.txt file with comma separated values
017 */
018public class NomStatusParserTest extends ParserTestBase<NomenclaturalStatus> {
019
020    public NomStatusParserTest() {
021        super(NomStatusParser.getInstance());
022    }
023
024    /**
025     * This ensures that ALL enum values are at least parsable by the name they
026     * are created with.
027     */
028    @Test
029    public void testCompleteness() {
030        for (NomenclaturalStatus t : NomenclaturalStatus.values()) {
031            assertParseSuccess(t, t.name());
032            assertParseSuccess(t, t.name().toLowerCase());
033            assertParseSuccess(t, t.name().replace("_", "").toLowerCase());
034            if (!Strings.isNullOrEmpty(t.getLatinLabel())) {
035                assertParseSuccess(t, t.getLatinLabel());
036            }
037            if (!Strings.isNullOrEmpty(t.getAbbreviatedLabel())) {
038                assertParseSuccess(t, t.getAbbreviatedLabel());
039            }
040        }
041    }
042
043    @Test
044    public void testFailures() {
045        assertParseFailure(null);
046        assertParseFailure("");
047        assertParseFailure("Tim");
048    }
049
050
051    @Test
052    public void testParse() {
053        // run a few basic tests to check it bootstraps and appears to work
054        assertParseSuccess(NomenclaturalStatus.NUDUM, "nom nud");
055        assertParseSuccess(NomenclaturalStatus.ILLEGITIMATE, "later HOmonym");
056        assertParseSuccess(NomenclaturalStatus.NEW_COMBINATION, ParseResult.CONFIDENCE.PROBABLE, "comb nov , rejected name, homotypic syn\tcomb nov , rejected name, homotypic syn");
057        assertParseSuccess(NomenclaturalStatus.CONSERVED, "nom cons , see art 14");
058        assertParseSuccess(NomenclaturalStatus.ILLEGITIMATE, ParseResult.CONFIDENCE.PROBABLE, "nom illeg ; (1775), non mill (1768)");
059        assertParseSuccess(NomenclaturalStatus.ILLEGITIMATE, "nom illegit , arts 53 1, 53 3");
060        assertParseSuccess(NomenclaturalStatus.INVALID, ParseResult.CONFIDENCE.PROBABLE, "nom inval , art 32 1(b), see art 18 3");
061        assertParseSuccess(NomenclaturalStatus.NEW_SPECIES, "sp nov , heterotypic syn");
062    }
063
064
065    @Test
066    public void testFileCoverage() throws IOException {
067        // parses all values in our test file (generated from real occurrence data) and verifies we never get worse at parsing
068        Resources.readLines(Resources.getResource("parse/nom_status.txt"), Charset.forName("UTF8"), this);
069        BatchParseResult result = getResult();
070        System.out.println(String.format("%s out of %s lines failed to parse", result.failed, result.total));
071        assertTrue(result.failed <= 166); // out of 747
072    }
073
074}
