001package org.gbif.common.parsers.rank;
002
003import org.gbif.api.vocabulary.Rank;
004import org.gbif.common.parsers.InterpretedParserTest;
005
006import org.junit.Test;
007
008public class InterpretedRankParserTest extends InterpretedParserTest<Rank> {
009
010  public InterpretedRankParserTest() {
011    super(InterpretedRankParser.getInstance());
012  }
013
014  /**
015   * This ensures that ALL enum values are at least parsable by the name they
016   * are created with.
017   */
018  @Test
019  public void testCompleteness() {
020    for (Rank t : Rank.values()) {
021      System.out.println("Testing [" + t.name()
022                         + "].  Failures below might indicate new ranks added to the Rank enum but not to the parse file");
023      assertParseSuccess(parser, t, t.name());
024    }
025  }
026
027  @Test
028  public void testFailures() {
029    assertParseFailure(parser, null);
030    assertParseFailure(parser, "");
031    assertParseFailure(parser, "Tim");
032  }
033
034
035  @Test
036  public void testParse() {
037    // run a few basic tests to check it bootstraps and appears to work
038    assertParseSuccess(parser, Rank.SPECIES, "species");
039    assertParseSuccess(parser, Rank.SPECIES, "SPECIES");
040  }
041}
