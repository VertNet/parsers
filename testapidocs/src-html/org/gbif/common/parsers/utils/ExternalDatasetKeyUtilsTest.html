001package org.gbif.common.parsers.utils;
002
003import java.util.UUID;
004
005import org.junit.Test;
006
007import static org.junit.Assert.assertEquals;
008import static org.junit.Assert.assertNull;
009
010public class ExternalDatasetKeyUtilsTest {
011
012  private static final String EXTERNAL_DATASET_KEY =
013    "7879e569-4a13-4643-b833-d1a564675b86:urn%3Alsid%3Aknb.ecoinformatics.org%3Aknb-lter-cdr%3A8133";
014  private static final String INVALID_EXTERNAL_DATASET_KEY =
015    "7879e569:urn%3Alsid%3Aknb.ecoinformatics.org%3Aknb-lter-cdr%3A8133";
016  private static final String SHORT_EXTERNAL_DATASET_KEY = "7879e569";
017
018  @Test
019  public void testParseSourceKeyInvalid() {
020    UUID sourceKey = ExternalDatasetKeyUtils.parseSourceKey(INVALID_EXTERNAL_DATASET_KEY);
021    assertNull(sourceKey);
022  }
023
024  @Test
025  public void testParseSourceKeyShort() {
026    UUID sourceKey = ExternalDatasetKeyUtils.parseSourceKey(SHORT_EXTERNAL_DATASET_KEY);
027    assertNull(sourceKey);
028  }
029
030  @Test
031  public void testParseSourceKey() {
032    UUID sourceKey = ExternalDatasetKeyUtils.parseSourceKey(EXTERNAL_DATASET_KEY);
033    assertEquals(UUID.fromString("7879e569-4a13-4643-b833-d1a564675b86"), sourceKey);
034  }
035
036  @Test
037  public void testParseDatasetIdInvalid() {
038    String datasetId = ExternalDatasetKeyUtils.parseDatasetId(INVALID_EXTERNAL_DATASET_KEY);
039    assertNull(datasetId);
040  }
041
042  @Test
043  public void testParseDatasetId() {
044    String datasetId = ExternalDatasetKeyUtils.parseDatasetId(EXTERNAL_DATASET_KEY);
045    assertEquals("urn%3Alsid%3Aknb.ecoinformatics.org%3Aknb-lter-cdr%3A8133", datasetId);
046  }
047
048}
