020  @Test
021  public void testParseElevation() throws Exception {
022    assertResult(MeterRangeParser.parseElevation("10", "20", null), true, 15d, 5d);
023    assertResult(MeterRangeParser.parseElevation("10", "20", "1"), true, 15d, 6d);
024    assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10d, 1d);
025    assertResult(MeterRangeParser.parseElevation("10", null, null), true, 10d, null);
026    assertResult(MeterRangeParser.parseElevation(null, "10", null), true, 10d, null);
027    assertResult(MeterRangeParser.parseElevation("-800", null, null), true, -800d, null);
028    assertResult(MeterRangeParser.parseElevation("-100", "-50", null), true, -75d, 25d);
029    assertResult(MeterRangeParser.parseElevation("110", "115", null), true, 112.5d, 2.5d);
030    assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10d, 1d);
031    assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10d, 1d);
032    assertResult(MeterRangeParser.parseElevation(null, "10000", "1"), true, 10000d, 1d);
033    assertResult(MeterRangeParser.parseElevation("4061987", "4061987", null), false, null, null,
034                 OccurrenceIssue.ELEVATION_UNLIKELY);
035
036    // check units are removed
037    assertResult(MeterRangeParser.parseElevation("1000m", null, "1"), true, 1000d, 1d,
038                 OccurrenceIssue.ELEVATION_NON_NUMERIC);
039    assertResult(MeterRangeParser.parseElevation("3280ft", null, "1"), true, 999.74, 1d,
040                  OccurrenceIssue.ELEVATION_NOT_METRIC, OccurrenceIssue.ELEVATION_NON_NUMERIC);
041
042    // check out of range
043    assertResult(MeterRangeParser.parseElevation("100000000000", null, "1"), false, null, 1d,
044                 OccurrenceIssue.ELEVATION_UNLIKELY);
045
046    // nonsense
047    assertResult(MeterRangeParser.parseElevation("booya", "boom", "1"), false, null, 1d,
048                 OccurrenceIssue.ELEVATION_NON_NUMERIC);
049  }

084  private void assertResult(OccurrenceParseResult<?> pr, boolean success, Double elevation, Double precision, OccurrenceIssue ... issue) {
085    if (success) {
086      assertExpected(pr, new DoubleAccuracy(elevation, precision), ParseResult.CONFIDENCE.DEFINITE, issue);
087    } else {
088      assertFailed(pr);
089    }
090  }
091
092  private void assertExpected(OccurrenceParseResult<?> pr, Object expected, ParseResult.CONFIDENCE c, OccurrenceIssue ... issue) {
093    assertNotNull(pr);
094    assertEquals(ParseResult.STATUS.SUCCESS, pr.getStatus());
095    assertEquals(c, pr.getConfidence());
096    assertNotNull(pr.getPayload());
097    assertEquals(expected, pr.getPayload());
098    if (issue == null) {
099      assertTrue(pr.getIssues().isEmpty());
100    } else {
101      assertEquals("Issue mismatch, found issues: " + pr.getIssues(), issue.length, pr.getIssues().size());
102      for (OccurrenceIssue iss : issue) {
103        assertTrue(pr.getIssues().contains(iss));
104      }
105    }
106  }
107
108  private void assertFailed(ParseResult<?> pr) {
109    assertNotNull(pr);
110    assertEquals(ParseResult.STATUS.FAIL, pr.getStatus());
111  }
