019  @Test
020  public void testParseElevation() throws Exception {
021    assertResult(MeterRangeParser.parseElevation("10", "20", null), true, 15, 5);
022    assertResult(MeterRangeParser.parseElevation("10", "20", "1"), true, 15, 6);
023    assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10, 1);
024    assertResult(MeterRangeParser.parseElevation("10", null, null), true, 10, null);
025    assertResult(MeterRangeParser.parseElevation(null, "10", null), true, 10, null);
026    assertResult(MeterRangeParser.parseElevation("-800", null, null), true, -800, null);
027    assertResult(MeterRangeParser.parseElevation("-100", "-50", null), true, -75, 25);
028    assertResult(MeterRangeParser.parseElevation("110", "115", null), true, 113, 3);
029    assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10, 1);
030    assertResult(MeterRangeParser.parseElevation("10", "10", "1"), true, 10, 1);
031    assertResult(MeterRangeParser.parseElevation(null, "10000", "1"), true, 10000, 1);
032    assertResult(MeterRangeParser.parseElevation("4061987", "4061987", null), false, null, null,
033                 OccurrenceIssue.ELEVATION_UNLIKELY);
034
035    // check units are removed
036    assertResult(MeterRangeParser.parseElevation("1000m", null, "1"), true, 1000, 1,
037                 OccurrenceIssue.ELEVATION_NON_NUMERIC);
038    assertResult(MeterRangeParser.parseElevation("3280ft", null, "1"), true, 1000, 1,
039                  OccurrenceIssue.ELEVATION_NOT_METRIC, OccurrenceIssue.ELEVATION_NON_NUMERIC);
040
041    // check out of range
042    assertResult(MeterRangeParser.parseElevation("100000000000", null, "1"), false, null, 1,
043                 OccurrenceIssue.ELEVATION_UNLIKELY);
044
045    // nonsense
046    assertResult(MeterRangeParser.parseElevation("booya", "boom", "1"), false, null, 1,
047                 OccurrenceIssue.ELEVATION_NON_NUMERIC);
048  }

114  private void assertResult(ParseResult<?> pr, boolean success, Integer elevation, Integer precision, OccurrenceIssue ... issue) {
115    if (success) {
116      assertExpected(pr, new IntAccuracy(elevation, precision), ParseResult.CONFIDENCE.DEFINITE, issue);
117    } else {
118      assertFailed(pr);
119    }
120  }
121
122  private void assertExpected(ParseResult<?> pr, Object expected, ParseResult.CONFIDENCE c, OccurrenceIssue ... issue) {
123    assertNotNull(pr);
124    assertEquals(ParseResult.STATUS.SUCCESS, pr.getStatus());
125    assertEquals(c, pr.getConfidence());
126    assertNotNull(pr.getPayload());
127    assertEquals(expected, pr.getPayload());
128    if (issue == null) {
129      assertTrue(pr.getIssues().isEmpty());
130    } else {
131      assertEquals("Issue mismatch, found issues: " + pr.getIssues(), issue.length, pr.getIssues().size());
132      for (OccurrenceIssue iss : issue) {
133        assertTrue(pr.getIssues().contains(iss));
134      }
135    }
136  }
137
138  private void assertFailed(ParseResult<?> pr) {
139    assertNotNull(pr);
140    assertEquals(ParseResult.STATUS.FAIL, pr.getStatus());
141  }
