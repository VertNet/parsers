001package org.gbif.common.parsers.geospatial;
002
003
004import org.gbif.api.vocabulary.OccurrenceIssue;
005import org.gbif.common.parsers.core.ParseResult;
006
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010import static org.junit.Assert.assertNotNull;
011import static org.junit.Assert.assertTrue;
012
013public class CoordinateParseUtilsTest {
014
015  @Test
016  public void testParseLatLng() {
017    assertExpected(CoordinateParseUtils.parseLatLng("10.3", "99.99"), new LatLng(10.3, 99.99), ParseResult.CONFIDENCE.DEFINITE);
018    assertExpected(CoordinateParseUtils.parseLatLng("10", "10"), new LatLng(10, 10), ParseResult.CONFIDENCE.DEFINITE);
019    assertExpected(CoordinateParseUtils.parseLatLng("90", "180"), new LatLng(90, 180), ParseResult.CONFIDENCE.DEFINITE);
020    assertExpected(CoordinateParseUtils.parseLatLng("-90", "180"), new LatLng(-90, 180), ParseResult.CONFIDENCE.DEFINITE);
021    assertExpected(CoordinateParseUtils.parseLatLng("90", "-180"), new LatLng(90, -180), ParseResult.CONFIDENCE.DEFINITE);
022    assertExpected(CoordinateParseUtils.parseLatLng("-90", "-180"), new LatLng(-90, -180), ParseResult.CONFIDENCE.DEFINITE);
023    assertExpected(CoordinateParseUtils.parseLatLng("0", "0"), new LatLng(0, 0), ParseResult.CONFIDENCE.POSSIBLE, OccurrenceIssue.ZERO_COORDINATE);
024
025    // check swapped coords
026    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("100", "40"), OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
027    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("-100", "90"), OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
028
029    // check errors
030    assertErrored(CoordinateParseUtils.parseLatLng("tim", "tom"));
031    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("200", "200"), OccurrenceIssue.COORDINATES_OUT_OF_RANGE);
032    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("-200", "30"), OccurrenceIssue.COORDINATES_OUT_OF_RANGE);
033    assertFailedWithIssues(CoordinateParseUtils.parseLatLng("200", "30"), OccurrenceIssue.COORDINATES_OUT_OF_RANGE);
034  }
035
036  private void assertExpected(ParseResult<?> pr, Object expected, ParseResult.CONFIDENCE c, OccurrenceIssue ... issue) {
037    assertNotNull(pr);
038    assertEquals(ParseResult.STATUS.SUCCESS, pr.getStatus());
039    assertEquals(c, pr.getConfidence());
040    assertNotNull(pr.getPayload());
041    assertEquals(expected, pr.getPayload());
042    if (issue == null) {
043      assertTrue(pr.getIssues().isEmpty());
044    } else {
045      assertEquals(issue.length, pr.getIssues().size());
046      for (OccurrenceIssue iss : issue) {
047        assertTrue(pr.getIssues().contains(iss));
048      }
049    }
050  }
051
052  private void assertErrored(ParseResult<?> pr) {
053    assertNotNull(pr);
054    assertEquals(ParseResult.STATUS.ERROR, pr.getStatus());
055  }
056
057  private void assertFailed(ParseResult<?> pr) {
058    assertNotNull(pr);
059    assertEquals(ParseResult.STATUS.FAIL, pr.getStatus());
060  }
061
062  private void assertFailedWithIssues(ParseResult<LatLng> pr, OccurrenceIssue issue) {
063    assertFailed(pr);
064    assertEquals(1, pr.getIssues().size());
065    assertTrue(pr.getIssues().contains(issue));
066  }
067}
