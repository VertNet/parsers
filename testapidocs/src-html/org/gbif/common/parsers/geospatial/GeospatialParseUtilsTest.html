001package org.gbif.common.parsers.geospatial;
002
003
004import org.gbif.api.vocabulary.OccurrenceValidationRule;
005import org.gbif.common.parsers.ParseResult;
006
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010import static org.junit.Assert.assertNotNull;
011
012public class GeospatialParseUtilsTest {
013
014  @Test
015  public void testParseLatLng() {
016    assertExpected(GeospatialParseUtils.parseLatLng("10.3", "99.99"), new LatLngIssue(10.3, 99.99),
017      ParseResult.CONFIDENCE.DEFINITE);
018    assertExpected(GeospatialParseUtils.parseLatLng("10", "10"), new LatLngIssue(10, 10),
019      ParseResult.CONFIDENCE.DEFINITE);
020    assertExpected(GeospatialParseUtils.parseLatLng("90", "180"), new LatLngIssue(90, 180),
021      ParseResult.CONFIDENCE.DEFINITE);
022    assertExpected(GeospatialParseUtils.parseLatLng("-90", "180"), new LatLngIssue(-90, 180),
023      ParseResult.CONFIDENCE.DEFINITE);
024    assertExpected(GeospatialParseUtils.parseLatLng("90", "-180"), new LatLngIssue(90, -180),
025      ParseResult.CONFIDENCE.DEFINITE);
026    assertExpected(GeospatialParseUtils.parseLatLng("-90", "-180"), new LatLngIssue(-90, -180),
027      ParseResult.CONFIDENCE.DEFINITE);
028    assertExpected(GeospatialParseUtils.parseLatLng("0", "0"), new LatLngIssue(0, 0, OccurrenceValidationRule.ZERO_COORDINATE),
029      ParseResult.CONFIDENCE.POSSIBLE);
030
031    // check swapped coords
032    assertFailedWithIssues(GeospatialParseUtils.parseLatLng("100", "40"),
033                           OccurrenceValidationRule.PRESUMED_SWAPPED_COORDINATE);
034    assertFailedWithIssues(GeospatialParseUtils.parseLatLng("-100", "90"),
035                           OccurrenceValidationRule.PRESUMED_SWAPPED_COORDINATE);
036
037    // check errors
038    assertErrored(GeospatialParseUtils.parseLatLng("tim", "tom"));
039    assertFailedWithIssues(GeospatialParseUtils.parseLatLng("200", "200"), OccurrenceValidationRule.COORDINATES_OUT_OF_RANGE);
040    assertFailedWithIssues(GeospatialParseUtils.parseLatLng("-200", "30"), OccurrenceValidationRule.COORDINATES_OUT_OF_RANGE);
041    assertFailedWithIssues(GeospatialParseUtils.parseLatLng("200", "30"), OccurrenceValidationRule.COORDINATES_OUT_OF_RANGE);
042  }
043
044  @Test
045  public void testParseDepth() {
046    assertExpected(GeospatialParseUtils.parseDepth("10", "20", null), new IntPrecisionIssue(15, null),
047      ParseResult.CONFIDENCE.DEFINITE);
048    assertExpected(GeospatialParseUtils.parseDepth("10", "20", "1"), new IntPrecisionIssue(15, 1),
049      ParseResult.CONFIDENCE.DEFINITE);
050    assertExpected(GeospatialParseUtils.parseDepth("10", "10", "1"), new IntPrecisionIssue(10, 1),
051      ParseResult.CONFIDENCE.DEFINITE);
052    assertExpected(GeospatialParseUtils.parseDepth("10", null, "1"), new IntPrecisionIssue(10, 1),
053      ParseResult.CONFIDENCE.DEFINITE);
054
055    // check units are removed
056    assertExpected(GeospatialParseUtils.parseDepth("10m", null, "1"),
057      new IntPrecisionIssue(10, 1, OccurrenceValidationRule.DEPTH_NON_NUMERIC),
058      ParseResult.CONFIDENCE.DEFINITE);
059    assertExpected(GeospatialParseUtils.parseDepth("3.27ft", null, "1"),
060      new IntPrecisionIssue(1, 1,OccurrenceValidationRule.DEPTH_PRESUMED_IN_FEET, OccurrenceValidationRule.DEPTH_NON_NUMERIC),
061      ParseResult.CONFIDENCE.DEFINITE);
062
063    // check out of range
064    assertExpected(GeospatialParseUtils.parseDepth("100000000", null, "1"),
065      new IntPrecisionIssue(null, null, OccurrenceValidationRule.DEPTH_OUT_OF_RANGE),
066      ParseResult.CONFIDENCE.DEFINITE);
067
068    // nonsense
069    assertFailed(GeospatialParseUtils.parseDepth("booya", "boom", "1"));
070  }
071
072  @Test
073  public void testParseAltitude() {
074    assertExpected(GeospatialParseUtils.parseAltitude("10", "20", null), new IntPrecisionIssue(15, null),
075      ParseResult.CONFIDENCE.DEFINITE);
076    assertExpected(GeospatialParseUtils.parseAltitude("10", "20", "1"), new IntPrecisionIssue(15, 1),
077      ParseResult.CONFIDENCE.DEFINITE);
078    assertExpected(GeospatialParseUtils.parseAltitude("10", "10", "1"), new IntPrecisionIssue(10, 1),
079      ParseResult.CONFIDENCE.DEFINITE);
080    assertExpected(GeospatialParseUtils.parseAltitude("10", null, "1"), new IntPrecisionIssue(10, 1),
081      ParseResult.CONFIDENCE.DEFINITE);
082    assertExpected(GeospatialParseUtils.parseAltitude(null, "10000", "1"), new IntPrecisionIssue(10000, 1),
083      ParseResult.CONFIDENCE.DEFINITE);
084    assertExpected(GeospatialParseUtils.parseAltitude("4061987", "4061987", null),
085      new IntPrecisionIssue(null, null, OccurrenceValidationRule.ALTITUDE_OUT_OF_RANGE),
086      ParseResult.CONFIDENCE.DEFINITE);
087
088    // check units are removed
089    assertExpected(GeospatialParseUtils.parseAltitude("1000m", null, "1"),
090      new IntPrecisionIssue(1000, 1, OccurrenceValidationRule.ALTITUDE_NON_NUMERIC),
091      ParseResult.CONFIDENCE.DEFINITE);
092    assertExpected(GeospatialParseUtils.parseAltitude("3280ft", null, "1"),
093      new IntPrecisionIssue(1000, 1, OccurrenceValidationRule.ALTITUDE_PRESUMED_IN_FEET, OccurrenceValidationRule.ALTITUDE_NON_NUMERIC),
094      ParseResult.CONFIDENCE.DEFINITE);
095
096    // check out of range
097    assertExpected(GeospatialParseUtils.parseAltitude("100000000000", null, "1"),
098      new IntPrecisionIssue(null, 1, OccurrenceValidationRule.ALTITUDE_OUT_OF_RANGE),
099      ParseResult.CONFIDENCE.DEFINITE);
100
101    // nonsense
102    assertFailed(GeospatialParseUtils.parseAltitude("booya", "boom", "1"));
103  }
104
105  private void assertExpected(ParseResult<?> pr, Object expected, ParseResult.CONFIDENCE c) {
106    assertNotNull(pr);
107    assertEquals(ParseResult.STATUS.SUCCESS, pr.getStatus());
108    assertEquals(c, pr.getConfidence());
109    assertNotNull(pr.getPayload());
110    assertEquals(expected, pr.getPayload());
111  }
112
113  private void assertErrored(ParseResult<?> pr) {
114    assertNotNull(pr);
115    assertEquals(ParseResult.STATUS.ERROR, pr.getStatus());
116  }
117
118  private void assertFailed(ParseResult<?> pr) {
119    assertNotNull(pr);
120    assertEquals(ParseResult.STATUS.FAIL, pr.getStatus());
121  }
122
123  private void assertFailedWithIssues(ParseResult<LatLngIssue> pr, OccurrenceValidationRule issue) {
124    assertFailed(pr);
125    assertNotNull(pr.getPayload().getIssue());
126    assertEquals(issue, pr.getPayload().getIssue());
127  }
128}
