001package org.gbif.common.parsers.geospatial;
002
003
004import org.gbif.api.vocabulary.OccurrenceIssue;
005import org.gbif.common.parsers.core.ParseResult;
006
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010import static org.junit.Assert.assertNotNull;
011import static org.junit.Assert.assertTrue;
012
013public class GeospatialParseUtilsTest {
014
015  @Test
016  public void testParseLatLng() {
017    assertExpected(GeospatialParseUtils.parseLatLng("10.3", "99.99"), new LatLng(10.3, 99.99), ParseResult.CONFIDENCE.DEFINITE);
018    assertExpected(GeospatialParseUtils.parseLatLng("10", "10"), new LatLng(10, 10), ParseResult.CONFIDENCE.DEFINITE);
019    assertExpected(GeospatialParseUtils.parseLatLng("90", "180"), new LatLng(90, 180), ParseResult.CONFIDENCE.DEFINITE);
020    assertExpected(GeospatialParseUtils.parseLatLng("-90", "180"), new LatLng(-90, 180), ParseResult.CONFIDENCE.DEFINITE);
021    assertExpected(GeospatialParseUtils.parseLatLng("90", "-180"), new LatLng(90, -180), ParseResult.CONFIDENCE.DEFINITE);
022    assertExpected(GeospatialParseUtils.parseLatLng("-90", "-180"), new LatLng(-90, -180), ParseResult.CONFIDENCE.DEFINITE);
023    assertExpected(GeospatialParseUtils.parseLatLng("0", "0"), new LatLng(0, 0), ParseResult.CONFIDENCE.POSSIBLE, OccurrenceIssue.ZERO_COORDINATE);
024
025    // check swapped coords
026    assertFailedWithIssues(GeospatialParseUtils.parseLatLng("100", "40"), OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
027    assertFailedWithIssues(GeospatialParseUtils.parseLatLng("-100", "90"), OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
028
029    // check errors
030    assertErrored(GeospatialParseUtils.parseLatLng("tim", "tom"));
031    assertFailedWithIssues(GeospatialParseUtils.parseLatLng("200", "200"), OccurrenceIssue.COORDINATES_OUT_OF_RANGE);
032    assertFailedWithIssues(GeospatialParseUtils.parseLatLng("-200", "30"), OccurrenceIssue.COORDINATES_OUT_OF_RANGE);
033    assertFailedWithIssues(GeospatialParseUtils.parseLatLng("200", "30"), OccurrenceIssue.COORDINATES_OUT_OF_RANGE);
034  }
035
036  @Test
037  public void testParseDepth() {
038    assertExpected(GeospatialParseUtils.parseDepth("10", "20", null), new IntPrecision(15, null), ParseResult.CONFIDENCE.DEFINITE);
039    assertExpected(GeospatialParseUtils.parseDepth("10", "20", "1"), new IntPrecision(15, 1), ParseResult.CONFIDENCE.DEFINITE);
040    assertExpected(GeospatialParseUtils.parseDepth("10", "10", "1"), new IntPrecision(10, 1), ParseResult.CONFIDENCE.DEFINITE);
041    assertExpected(GeospatialParseUtils.parseDepth("10", null, "1"), new IntPrecision(10, 1), ParseResult.CONFIDENCE.DEFINITE);
042
043    // check units are removed
044    assertExpected(GeospatialParseUtils.parseDepth("10m", null, "1"),
045      new IntPrecision(10, 1), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.DEPTH_NON_NUMERIC);
046    assertExpected(GeospatialParseUtils.parseDepth("3.27ft", null, "1"),
047      new IntPrecision(1, 1), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.DEPTH_PRESUMED_IN_FEET, OccurrenceIssue.DEPTH_NON_NUMERIC);
048
049    // check out of range
050    assertExpected(GeospatialParseUtils.parseDepth("100000000", null, "1"),
051      new IntPrecision(null, null), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.DEPTH_OUT_OF_RANGE);
052
053    // nonsense
054    assertFailed(GeospatialParseUtils.parseDepth("booya", "boom", "1"));
055  }
056
057  @Test
058  public void testParseAltitude() {
059    assertExpected(GeospatialParseUtils.parseAltitude("10", "20", null), new IntPrecision(15, null),
060      ParseResult.CONFIDENCE.DEFINITE);
061    assertExpected(GeospatialParseUtils.parseAltitude("10", "20", "1"), new IntPrecision(15, 1),
062      ParseResult.CONFIDENCE.DEFINITE);
063    assertExpected(GeospatialParseUtils.parseAltitude("10", "10", "1"), new IntPrecision(10, 1),
064      ParseResult.CONFIDENCE.DEFINITE);
065    assertExpected(GeospatialParseUtils.parseAltitude("10", null, "1"), new IntPrecision(10, 1),
066      ParseResult.CONFIDENCE.DEFINITE);
067    assertExpected(GeospatialParseUtils.parseAltitude(null, "10000", "1"), new IntPrecision(10000, 1),
068      ParseResult.CONFIDENCE.DEFINITE);
069    assertExpected(GeospatialParseUtils.parseAltitude("4061987", "4061987", null),
070      new IntPrecision(null, null), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.ALTITUDE_OUT_OF_RANGE);
071
072    // check units are removed
073    assertExpected(GeospatialParseUtils.parseAltitude("1000m", null, "1"),
074      new IntPrecision(1000, 1),ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.ALTITUDE_NON_NUMERIC);
075    assertExpected(GeospatialParseUtils.parseAltitude("3280ft", null, "1"),
076      new IntPrecision(1000, 1), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.ALTITUDE_PRESUMED_IN_FEET, OccurrenceIssue.ALTITUDE_NON_NUMERIC);
077
078    // check out of range
079    assertExpected(GeospatialParseUtils.parseAltitude("100000000000", null, "1"),
080                   new IntPrecision(null, 1), ParseResult.CONFIDENCE.DEFINITE, OccurrenceIssue.ALTITUDE_OUT_OF_RANGE);
081
082    // nonsense
083    assertFailed(GeospatialParseUtils.parseAltitude("booya", "boom", "1"));
084  }
085
086  private void assertExpected(ParseResult<?> pr, Object expected, ParseResult.CONFIDENCE c, OccurrenceIssue ... issue) {
087    assertNotNull(pr);
088    assertEquals(ParseResult.STATUS.SUCCESS, pr.getStatus());
089    assertEquals(c, pr.getConfidence());
090    assertNotNull(pr.getPayload());
091    assertEquals(expected, pr.getPayload());
092    if (issue == null) {
093      assertTrue(pr.getIssues().isEmpty());
094    } else {
095      assertEquals(issue.length, pr.getIssues().size());
096      for (OccurrenceIssue iss : issue) {
097        assertTrue(pr.getIssues().contains(iss));
098      }
099    }
100  }
101
102  private void assertErrored(ParseResult<?> pr) {
103    assertNotNull(pr);
104    assertEquals(ParseResult.STATUS.ERROR, pr.getStatus());
105  }
106
107  private void assertFailed(ParseResult<?> pr) {
108    assertNotNull(pr);
109    assertEquals(ParseResult.STATUS.FAIL, pr.getStatus());
110  }
111
112  private void assertFailedWithIssues(ParseResult<LatLng> pr, OccurrenceIssue issue) {
113    assertFailed(pr);
114    assertEquals(1, pr.getIssues().size());
115    assertTrue(pr.getIssues().contains(issue));
116  }
117}
