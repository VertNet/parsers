001package org.gbif.common.parsers.date;
002
003import org.gbif.utils.file.FileUtils;
004import org.gbif.utils.file.csv.CSVReader;
005import org.gbif.utils.file.csv.CSVReaderFactory;
006
007import java.io.File;
008import java.io.IOException;
009
010import javax.annotation.Nullable;
011
012import com.google.common.base.Function;
013import org.junit.Test;
014
015import static org.junit.Assert.assertEquals;
016import static org.junit.Assert.assertNull;
017import static org.junit.Assert.assertTrue;
018import static org.junit.Assert.fail;
019
020/**
021 * Test for {@link TextualMonthDateTokenizer }.
022 */
023public class TextualMonthDateTokenizerTest {
024
025  private static final String TEST_FILE = "parse/date/TextualMonthDateTokenizerTests.txt";
026  private static final String COLUMN_SEPARATOR = ";";
027  private static final String COMMENT_MARKER = "#";
028
029  private static final int RAW_VAL_IDX = 0;
030  private static final int INT_4_IDX = 1;
031  private static final int TEXT_IDX = 2;
032  private static final int INT_2_IDX = 3;
033
034  private TextualMonthDateTokenizer DATE_TOKENIZER = new TextualMonthDateTokenizer();
035
036  @Test
037  public void testDateTokenizerFromFile(){
038
039    assertTestFile(TEST_FILE, new Function<String[], Void>(){
040
041      @Nullable
042      @Override
043      public Void apply(String[] row) {
044
045        String raw = row[RAW_VAL_IDX];
046        String int4Token = row[INT_4_IDX];
047        String textToken = row[TEXT_IDX];
048        String int2Token = row[INT_2_IDX];
049
050        TextualMonthDateTokenizer.DateTokens dateTokens = DATE_TOKENIZER.tokenize(raw);
051        assertEquals("Raw: " + raw, new TextualMonthDateTokenizer.DateToken(int2Token, TextualMonthDateTokenizer.TokenType.INT_2),
052                dateTokens.getToken(TextualMonthDateTokenizer.TokenType.INT_2));
053        assertEquals("Raw: " + raw, new TextualMonthDateTokenizer.DateToken(textToken, TextualMonthDateTokenizer.TokenType.TEXT),
054                dateTokens.getToken(TextualMonthDateTokenizer.TokenType.TEXT));
055        assertEquals("Raw: " + raw, new TextualMonthDateTokenizer.DateToken(int4Token, TextualMonthDateTokenizer.TokenType.INT_4),
056                dateTokens.getToken(TextualMonthDateTokenizer.TokenType.INT_4));
057
058        return null;
059      }
060    });
061  }
062
063  /**
064   * Utility function to run assertions received as Function on each rows of an input file.
065   *
066   * @param filepath
067   * @param assertRow
068   */
069  private void assertTestFile(String filepath, Function<String[], Void> assertRow) {
070    File testInputFile = FileUtils.getClasspathFile(filepath);
071    try{
072      CSVReader csv = CSVReaderFactory.build(testInputFile, COLUMN_SEPARATOR, true);
073      for (String[] row : csv) {
074        if (row == null || row[0].startsWith(COMMENT_MARKER)) {
075          continue;
076        }
077        assertRow.apply(row);
078      }
079    } catch (IOException e) {
080      fail("Problem reading testFile " + filepath + " " + e.getMessage());
081    }
082  }
083
084  @Test
085  public void testDateTokenizerWithNullAndEmpty(){
086    assertNull(DATE_TOKENIZER.tokenize(null));
087    assertNull(DATE_TOKENIZER.tokenize(""));
088  }
089
090  /**
091   * Test behavior when no textual month is provided.
092   * Month and Day will be returned as TokenType.INT_2, one of the 2 will fall under the discarded tokens.
093   * See {@link TextualMonthDateTokenizer } for more details.
094   */
095  @Test
096  public void testDateTokenizerWithISODate(){
097    TextualMonthDateTokenizer.DateTokens dateTokens = DATE_TOKENIZER.tokenize("2018-01-02");
098    assertEquals(new TextualMonthDateTokenizer.DateToken("2018", TextualMonthDateTokenizer.TokenType.INT_4),
099            dateTokens.getToken(TextualMonthDateTokenizer.TokenType.INT_4));
100    assertTrue(dateTokens.containsDiscardedTokens());
101  }
102}
