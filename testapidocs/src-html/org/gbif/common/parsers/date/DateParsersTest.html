001package org.gbif.common.parsers.date;
002
003import org.junit.Ignore;
004import org.junit.Test;
005import org.threeten.bp.YearMonth;
006import org.threeten.bp.format.DateTimeFormatter;
007import org.threeten.bp.format.DateTimeFormatterBuilder;
008import org.threeten.bp.format.DateTimeParseException;
009import org.threeten.bp.format.ResolverStyle;
010import org.threeten.bp.format.SignStyle;
011import org.threeten.bp.temporal.ChronoField;
012import org.threeten.bp.temporal.TemporalAccessor;
013
014import static org.junit.Assert.assertNotNull;
015
016/**
017 * Validate predefined parsers from {@link DateParsers}
018 */
019public class DateParsersTest {
020
021  //  -- DateParsers.ISO_YEAR --
022  @Test
023  public void testParsersISOYear(){
024    assertNotNull(DateParsers.ISO_YEAR.parse("2016"));
025  }
026
027  @Test(expected = DateTimeParseException.class)
028  public void testParsersISOYearException1(){
029    TemporalAccessor ta = DateParsers.ISO_YEAR.parse("2016-10-45");
030    YearMonth.from(ta);
031  }
032
033  @Test(expected = DateTimeParseException.class)
034  public void testParsersISOYearException2(){
035    DateParsers.ISO_YEAR.parse("16");
036  }
037
038
039  //  -- ISO_YEAR_MONTH --
040  @Test
041  public void testParsersISOYearMonth(){
042    DateParsers.ISO_YEAR_MONTH.parse("2016-05");
043  }
044
045  @Test(expected = DateTimeParseException.class)
046  public void testParsersISOYearMonthException1(){
047    assertNotNull(DateParsers.ISO_YEAR_MONTH.parse("2016"));
048  }
049
050  @Test(expected = DateTimeParseException.class)
051  public void testParsersISOYearMonthException2(){
052    DateParsers.ISO_YEAR_MONTH.parse("2016-07-25");
053  }
054
055  //  -- ISO_LOCAL_PARTIAL_DATE --
056  @Test
057  public void testIsoLocalPartialDateParser(){
058    DateTimeFormatter parser = DateParsers.ISO_LOCAL_PARTIAL_DATE;
059
060    TemporalAccessor ta = parser.parse("2016");
061    assertNotNull(ta);
062
063    ta = parser.parse("2016-07");
064    assertNotNull(ta);
065
066    ta = parser.parse("2016-07-25");
067    assertNotNull(ta);
068
069    //test wrong date that "works" https://github.com/ThreeTen/threetenbp/issues/49
070    ta = parser.parse("2016-40");
071    assertNotNull(ta);
072
073    //test wrong date
074   // ta = parser.parse("2016-40-10");
075  //  assertNotNull(ta);
076
077  }
078
079
080  // see https://github.com/ThreeTen/threetenbp/issues/49
081  @Test
082  public void testISOOptionalMonthOfYear(){
083    DateTimeFormatter isoOptionalDayOfMonthFormatter =
084            new DateTimeFormatterBuilder()
085                    .appendValue(ChronoField.YEAR, 4, 4, SignStyle.NEVER)
086                    .optionalStart()
087                    .appendLiteral('-')
088                    .appendValue(ChronoField.MONTH_OF_YEAR, 1, 2, SignStyle.NEVER)
089                    .optionalEnd()
090                    .toFormatter().withResolverStyle(ResolverStyle.STRICT);
091    TemporalAccessor ta = isoOptionalDayOfMonthFormatter.parse("2016-40");
092  }
093
094  @Ignore(" see https://github.com/ThreeTen/threetenbp/issues/49")
095  public void testISOOptionalDayOfMonth(){
096    DateTimeFormatter isoOptionalDayOfMonthFormatter =
097            new DateTimeFormatterBuilder()
098                    .appendValue(ChronoField.YEAR, 4, 4, SignStyle.NEVER)
099                    .appendLiteral('-')
100                    .appendValue(ChronoField.MONTH_OF_YEAR, 1, 2, SignStyle.NEVER)
101                    .optionalStart()
102                    .appendLiteral('-')
103                    .appendValue(ChronoField.DAY_OF_MONTH, 1, 2, SignStyle.NEVER)
104                    .optionalEnd()
105                    .toFormatter().withResolverStyle(ResolverStyle.STRICT);
106    isoOptionalDayOfMonthFormatter.parse("2016-10-40");
107  }
