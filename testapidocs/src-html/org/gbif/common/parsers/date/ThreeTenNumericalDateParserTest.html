003import org.gbif.common.parsers.core.ParseResult;

006import org.threeten.bp.LocalDate;

012import static org.junit.Assert.assertEquals;

014/**
015 *
016 */
017public class ThreeTenNumericalDateParserTest {
018
019  @Test
020  public void testParseAsLocalDateTime(){
021    ThreeTenNumericalDateParser parser = new ThreeTenNumericalDateParser();
022
023    //d-m-y
024    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21-12-1978").getPayload()));
025    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21-12-78").getPayload()));
026    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21.12.78").getPayload()));
027    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21.12.1978").getPayload()));
028    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21/12/1978").getPayload()));
029    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21121978").getPayload()));
030    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21_12_78").getPayload()));
031    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21\\12\\78").getPayload()));
032    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21\\12\\1978").getPayload()));
033    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21-12-78").getPayload()));
034    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21_12_1978").getPayload()));
035    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("21/12/78").getPayload()));
036
037    //y-m-d
038    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("1978-12-21").getPayload()));
039    assertEquals(LocalDate.of(1978, Month.DECEMBER, 1), LocalDate.from(parser.parse("1978-12-01").getPayload()));
040
041    assertEquals(LocalDateTime.of(1978, Month.DECEMBER, 21, 2, 12, 43), LocalDateTime.from(parser.parse("1978-12-21 02:12:43").getPayload()));
042    assertEquals(LocalDateTime.of(1978, Month.DECEMBER, 21, 2, 0, 0), LocalDateTime.from(parser.parse("1978-12-21T02").getPayload()));
043    assertEquals(LocalDateTime.of(1978, Month.DECEMBER, 21, 2, 12, 0), LocalDateTime.from(parser.parse("1978-12-21T02:12").getPayload()));
044    assertEquals(LocalDateTime.of(1978, Month.DECEMBER, 21, 2, 12, 43), LocalDateTime.from(parser.parse("1978-12-21T02:12:43").getPayload()));
045    assertEquals(LocalDateTime.of(1978, Month.DECEMBER, 21, 2, 12, 0), LocalDateTime.from(parser.parse("1978-12-21T0212").getPayload()));
046    assertEquals(LocalDateTime.of(1978, Month.DECEMBER, 21, 2, 12, 43), LocalDateTime.from(parser.parse("1978-12-21T021243").getPayload()));
047
048    //Z means 0 offset
049    assertEquals(LocalDateTime.of(1978, Month.DECEMBER, 21, 2, 12, 43), LocalDateTime.from(parser.parse("1978-12-21T02:12:43Z").getPayload()));
050    assertEquals(LocalDateTime.of(1978, Month.DECEMBER, 21, 2, 12, 43), LocalDateTime.from(parser.parse("1978-12-21T02:12:43+0100").getPayload()));
051    assertEquals(LocalDateTime.of(1978, Month.DECEMBER, 21, 2, 12, 43), LocalDateTime.from(parser.parse("1978-12-21T02:12:43+01:00").getPayload()));
052    assertEquals(LocalDateTime.of(1978, Month.DECEMBER, 21, 2, 12, 43), LocalDateTime.from(parser.parse("1978-12-21T02:12:43").getPayload()));
053
054    //month first :(
055    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12/21/1978").getPayload()));
056    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12211978").getPayload()));
057    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12\\21\\1978").getPayload()));
058    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12.21.1978").getPayload()));
059    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12-21-1978").getPayload()));
060    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12_21_1978").getPayload()));
061    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12/21/1978").getPayload()));
062
063    //month first with 2 digits years >_<
064    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("122178").getPayload()));
065    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12/21/78").getPayload()));
066    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12\\21\\78").getPayload()));
067    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12.21.78").getPayload()));
068    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12-21-78").getPayload()));
069    assertEquals(LocalDate.of(1978, Month.DECEMBER, 21), LocalDate.from(parser.parse("12_21_78").getPayload()));
070
071    //month/year alone
072    assertEquals(YearMonth.of(1978,12), YearMonth.from(parser.parse("1978-12").getPayload()));
073
074    //year alone
075    assertEquals(Year.of(1978), Year.from(parser.parse("1978").getPayload()));
076    assertEquals(Year.of(1978), Year.from(parser.parse("78").getPayload()));
077  }

080  @Test
081  public void testParseUndefinedLocalDateTime(){
082    ThreeTenNumericalDateParser parser = new ThreeTenNumericalDateParser();
083    assertEquals(ParseResult.STATUS.ERROR, parser.parse("120578").getStatus());
084  }
085
086  @Test
087  public void testBadDates(){
088    ThreeTenNumericalDateParser parser = new ThreeTenNumericalDateParser();
089    assertEquals(ParseResult.STATUS.ERROR, parser.parse("fsfgr/12/78").getStatus());
090    assertEquals(ParseResult.STATUS.ERROR, parser.parse("fsfgr").getStatus());
091    assertEquals(ParseResult.STATUS.ERROR, parser.parse("//").getStatus());
092    assertEquals(ParseResult.STATUS.ERROR, parser.parse("0-0-0").getStatus());
093    assertEquals(ParseResult.STATUS.ERROR, parser.parse("%$-s2").getStatus());
094    assertEquals(ParseResult.STATUS.ERROR, parser.parse("as-fds-sdf").getStatus());
095    assertEquals(ParseResult.STATUS.ERROR, parser.parse("88/88/88").getStatus());
096    assertEquals(ParseResult.STATUS.ERROR, parser.parse(" ").getStatus());
097    assertEquals(ParseResult.STATUS.ERROR, parser.parse("").getStatus());
098    assertEquals(ParseResult.STATUS.ERROR, parser.parse(null).getStatus());
099  }
