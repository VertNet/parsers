001package org.gbif.common.parsers.date;
002
003import org.gbif.common.parsers.ParseResult;
004
005import java.io.BufferedReader;
006import java.io.BufferedWriter;
007import java.io.FileInputStream;
008import java.io.FileReader;
009import java.io.FileWriter;
010import java.io.IOException;
011import java.io.InputStreamReader;
012import java.text.SimpleDateFormat;
013import java.util.Date;
014import java.util.regex.Pattern;
015import java.util.zip.GZIPInputStream;
016
017import org.apache.commons.lang3.StringUtils;
018
019/**
020 * A small utility to allow real world testing.
021 * Takes a tab delimited input file of day \t month \t year
022 * and outputs 7 columns of input fields, parsed fields and
023 * a constructed date.
024 * There is the complete distinct verbatim day,month,year fields
025 * found in raw_occurrence_record table in the src/test/resources
026 *
027 * @author tim
028 */
029public class ParseInputFile {
030
031  /**
032   * @param args inputFile outputFile skipCompleteRecords
033   */
034  public static void main(String[] args) throws IOException {
035    BufferedReader r = null;
036    if (args[0].endsWith(".gz")) {
037      r = new BufferedReader(new InputStreamReader(new GZIPInputStream(new FileInputStream(args[0]))));
038    } else {
039      r = new BufferedReader(new FileReader(args[0]));
040    }
041    BufferedWriter w = new BufferedWriter(new FileWriter(args[1]));
042    w.write("D-in\tM-in\tY-in\tD-out\tM-out\tY-out\tDate\n");
043    boolean skip = false;
044    try {
045      skip = Boolean.parseBoolean(args[2]);
046    } catch (RuntimeException e) {
047    }
048    Pattern tab = Pattern.compile("\t");
049    String line = null;
050    int lineCount = 0;
051    int ymdCount = 0;
052    int dateCount = 0;
053    SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy");
054    while ((line = r.readLine()) != null) {
055      String[] dmy = tab.split(line);
056      String d = null;
057      String m = null;
058      String y = null;
059      try {
060        d = dmy[0];
061      } catch (Exception e) {
062      }
063      ;
064      try {
065        m = dmy[1];
066      } catch (Exception e) {
067      }
068      ;
069      try {
070        y = dmy[2];
071      } catch (Exception e) {
072      }
073      ;
074
075      YearMonthDay ymd = DateParseUtils.normalize(y, m, d);
076      ParseResult<Date> pr = DateParseUtils.parse(ymd.getYear(), ymd.getMonth(), ymd.getDay());
077      Date date = null;
078      if (ParseResult.STATUS.SUCCESS == pr.getStatus()) {
079        date = pr.getPayload();
080      }
081
082      if (skip && date != null && StringUtils.isNotBlank(ymd.getDay()) && StringUtils.isNotBlank(ymd.getMonth())
083          && StringUtils.isNotBlank(ymd.getYear())) {
084        dateCount++;
085        ymdCount++;
086
087      } else {
088        w.write(d + "\t");
089        w.write(m + "\t");
090        w.write(y + "\t");
091        if (ymd.getDay() != null) w.write(ymd.getDay());
092        w.write("\t");
093        if (ymd.getMonth() != null) w.write(ymd.getMonth());
094        w.write("\t");
095        if (ymd.getYear() != null) w.write(ymd.getYear());
096        w.write("\t");
097
098        if (!ymd.representsNull()) ymdCount++;
099
100        if (ParseResult.STATUS.SUCCESS == pr.getStatus()) {
101          w.write(sdf.format(date));
102          dateCount++;
103        }
104        w.write("\n");
105      }
106
107
108      lineCount++;
109      if (lineCount % 10000 == 0) {
110        System.out.println("Processed[" + lineCount + "], ymdFound[" + ymdCount + "], dateFound[" + dateCount + "]");
111      }
112    }
113    System.out.println("Processed[" + lineCount + "], ymdFound[" + ymdCount + "], dateFound[" + dateCount + "]");
114    w.close();
115    r.close();
116  }
117}
