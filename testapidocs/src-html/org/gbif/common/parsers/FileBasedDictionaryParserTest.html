001package org.gbif.common.parsers;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006import static org.junit.Assert.assertNotNull;
007
008public class FileBasedDictionaryParserTest {
009
010  @Test
011  public void testParse() {
012    Parsable<String, String> dbp =
013      new FileBasedDictionaryParser(false, getClass().getResourceAsStream("/parse/simpleDictionary.txt"));
014    assertParseSuccess(dbp, "32", "Tim");
015    assertParseSuccess(dbp, "32", "TIM");
016    assertParseSuccess(dbp, "32", "tIm");
017    assertParseSuccess(dbp, "38", "Markus");
018    assertParseSuccess(dbp, "38", "MarKUS");
019    assertParseSuccess(dbp, "28", "Jose");
020
021    assertParseFailure(dbp, "Lars");
022  }
023
024  protected static void assertParseSuccess(Parsable<String, String> dbp, String expected, String input) {
025    ParseResult<String> parsed = dbp.parse(input);
026    assertNotNull(parsed);
027    assertEquals(expected, parsed.getPayload());
028    assertEquals(ParseResult.STATUS.SUCCESS, parsed.getStatus());
029    assertEquals(ParseResult.CONFIDENCE.DEFINITE, parsed.getConfidence());
030  }
031
032  protected static void assertParseFailure(Parsable<String, String> dbp, String input) {
033    ParseResult<String> parsed = dbp.parse(input);
034    assertEquals(ParseResult.STATUS.FAIL, parsed.getStatus());
035  }
036}
