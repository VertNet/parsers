001package org.gbif.common.parsers;

003import org.gbif.api.vocabulary.Country;
004import org.gbif.common.parsers.core.ParseResult;

006import java.io.BufferedReader;
007import java.io.IOException;
008import java.io.InputStreamReader;
009import java.util.Map;

011
012import com.google.common.base.Charsets;
013import com.google.common.collect.Maps;
014import org.junit.Test;
015
016import static org.junit.Assert.assertTrue;
017
018public class CountryParserTest extends EnumParserTest<Country> {
019
020  public CountryParserTest() {
021    super(CountryParser.getInstance());
022  }
023
024  /**
025   * Makes sure all Country enum values are parsed ok.
026   */
027  @Test
028  public void testParseAllEnumValues() {
029    for (Country c : Country.OFFICIAL_COUNTRIES) {
030      assertParseSuccess(c, c.getIso2LetterCode());
031      assertParseSuccess(c, c.getIso3LetterCode());
032      assertParseSuccess(c, c.getTitle());
033    }
034  }
035
036  /**
037   * Parse all unique occurrence values found for countries and make sure parsing doesnt get worse.
038   * If the test file is updated, values here need to be adjusted!
039   */
040  @Test
041  public void testOccurrenceValues() throws IOException {
042    final int CURRENT_COUNTRIES_PARSED = 246;
043    final int CURRENT_TESTS_SUCCESSFUL = 2976;
044
045
046    // remember number of successful parsed countries:
047    Map<Country, AtomicInteger> stats = Maps.newHashMap();
048    int values = 0;
049    int failed = 0;
050    int success = 0;
051    BufferedReader r = new BufferedReader(
052      new InputStreamReader(this.getClass().getResourceAsStream("/parse/countryname/occurrence_countries.txt"),
053        Charsets.UTF_8));
054    String line;
055    do {
056      line = r.readLine();
057      values++;
058      ParseResult<Country> parsed = parser.parse(line);
059      if (ParseResult.STATUS.SUCCESS == parsed.getStatus()) {
060        Country c = parsed.getPayload();
061        if (!stats.containsKey(c)) {
062          stats.put(c, new AtomicInteger(0));
063        }
064        stats.get(c).incrementAndGet();
065        success++;
066      } else {
067        failed++;
068      }
069    } while (line != null);
070
071    System.out.println(values + " tested values");
072    System.out.println(failed + " failed parse results");
073    System.out.println(success + " successful parse results");
074    System.out.println(stats.size() + " distinct countries parsed");
075    assertTrue(failed < values);
076    assertTrue(CURRENT_COUNTRIES_PARSED <= stats.size());
077    assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
078  }
079
080  @Test
081  public void testParseFail() {
082    assertParseFailure("[West Indian Ocean]");
083    assertParseFailure("Really great britain");
084    assertParseFailure("Borneo");
085  }
086
087  @Test
088  public void testParseSuccess() {
089    assertParseSuccess(Country.UNITED_KINGDOM, "Great Britain");
090    assertParseSuccess(Country.UNITED_KINGDOM, "Great Britain!!!");
091    assertParseSuccess(Country.UNITED_KINGDOM, "Great  Britain"); // Test collapsing of multiple whitespaces
092    assertParseSuccess(Country.UNITED_KINGDOM, " Great\tBritain "); // Test collapsing of other kinds of whitespace and trimming
093    assertParseSuccess(Country.DENMARK, "Denmark");
094    assertParseSuccess(Country.GERMANY, " Germany "); // Test trimming
095    assertParseSuccess(Country.AUSTRALIA, "off Australia");
096    assertParseSuccess(Country.AUSTRALIA, " off (australia)\t! \u0312 "); // all together now...
097    assertParseSuccess(Country.DENMARK, "DK");
098    assertParseSuccess(Country.DENMARK, "OFF DENMARK");
099    assertParseSuccess(Country.MONTENEGRO, "me");
100    assertParseSuccess(Country.SOUTH_SUDAN, "ss");
101    assertParseSuccess(Country.SOUTH_SUDAN, "SOUTH_SUDAN");
102    assertParseSuccess(Country.SOUTH_SUDAN, "south sudan");
103    assertParseSuccess(Country.MEXICO, "México");
104    assertParseSuccess(Country.BOSNIA_HERZEGOVINA, "BOSNIA");
105    assertParseSuccess(Country.GERMANY, "Germany");
106    assertParseSuccess(Country.GERMANY, "deutschland");
107    assertParseSuccess(Country.GERMANY, "De");
108  }
