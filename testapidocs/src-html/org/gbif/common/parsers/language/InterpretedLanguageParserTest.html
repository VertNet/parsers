001package org.gbif.common.parsers.language;
002
003
004import org.gbif.api.vocabulary.Language;
005
006import junit.framework.TestCase;
007import org.junit.Test;
008
009public class InterpretedLanguageParserTest extends TestCase {
010
011  @Test
012  public void testParse() {
013    // run a few basic tests to check it bootstraps and appears to work
014    InterpretedLanguageParser parser = InterpretedLanguageParser.getInstance();
015    assertEquals(Language.GERMAN, parser.parse("German").getPayload().getInterpreted());
016    assertEquals(Language.GERMAN, parser.parse("deutsch").getPayload().getInterpreted());
017    assertEquals(Language.GERMAN, parser.parse("de").getPayload().getInterpreted());
018    assertEquals(Language.GERMAN, parser.parse("ger").getPayload().getInterpreted());
019
020    assertEquals(Language.BASQUE, parser.parse("baq").getPayload().getInterpreted());
021  }
022
023}
