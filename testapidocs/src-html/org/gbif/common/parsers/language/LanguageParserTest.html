001package org.gbif.common.parsers.language;
002
003import org.gbif.common.parsers.FileBasedDictionaryParserTest;
004
005import org.junit.Test;
006
007public class LanguageParserTest extends FileBasedDictionaryParserTest {
008
009  @Test
010  public void testParse() {
011    // run a few basic tests to check it bootstraps and appears to work
012    LanguageParser parser = LanguageParser.getInstance();
013    assertParseFailure(parser, "[West Indian Ocean]");
014    assertParseSuccess(parser, "de", "German");
015    assertParseSuccess(parser, "de", "Deutsch");
016    assertParseSuccess(parser, "de", "deutsch");
017    assertParseSuccess(parser, "de", "GER");
018    assertParseSuccess(parser, "de", "DEU");
019    assertParseSuccess(parser, "de", "de");
020    assertParseSuccess(parser, "de", "De ");
021    assertParseSuccess(parser, "en", "en_US");
022    assertParseSuccess(parser, "en", "eng_US");
023    assertParseSuccess(parser, "en", "english");
024  }
025}
