017package org.gbif.common.parsers;
018
019import org.gbif.common.parsers.utils.MappingUtils;
020
021import java.util.HashMap;
022import java.util.Map;
023
024import org.junit.Test;
025
026import static org.junit.Assert.assertEquals;
027import static org.junit.Assert.assertNull;
028
029/**
030 * @author markus
031 */
032public class MappingUtilsTest {
033
034  @Test
035  public void testKingdoms() {
036    Map<String, String> kingdoms = new HashMap<String, String>() {
037      {
038        put("", null);
039        put("PLANT", "Plantae");
040        put("FUNGI AND LICHENS", "Fungi");
041        put("FUNGI", "Fungi");
042        put("9ANIMALI", "Animalia");
043      }
044    };
045
046    for (Map.Entry<String, String> taxon : kingdoms.entrySet()) {
047      assertEquals(taxon.getValue(), MappingUtils.mapKingdom(taxon.getKey()));
048    }
049
050    assertNull(MappingUtils.mapKingdom(null));
051    assertNull("Empty strings should return null", MappingUtils.mapKingdom(""));
052  }
053
054  @Test
055  public void testPhyla() {
056    Map<String, String> phyla = new HashMap<String, String>() {
057      {
058        put("", null);
059        put("PLATYHERMINTHES", "Platyhelminthes");
060        put("PlaTYHERMINTHES", "Platyhelminthes");
061        put("?? PORIFERA ??", "Porifera");
062        put("PRASINOPHYTA", "Prasinophyta");
063        put("PSILOPHYTA (PSILOTUM)", "Psilophyta");
064      }
065    };
066
067    for (Map.Entry<String, String> taxon : phyla.entrySet()) {
068      assertEquals(taxon.getValue(), MappingUtils.mapPhylum(taxon.getKey()));
069    }
070
071    assertNull(MappingUtils.mapPhylum(null));
072    assertNull("Empty strings should return null", MappingUtils.mapPhylum(""));
073  }
074}
