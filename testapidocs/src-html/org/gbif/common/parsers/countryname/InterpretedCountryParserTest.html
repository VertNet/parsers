001package org.gbif.common.parsers.countryname;
002
003import org.gbif.api.model.common.InterpretedEnum;
004import org.gbif.api.vocabulary.Country;
005import org.gbif.common.parsers.ParseResult;
006
007import java.io.BufferedReader;
008import java.io.IOException;
009import java.io.InputStreamReader;
010import java.util.Map;
011import java.util.concurrent.atomic.AtomicInteger;
012
013import com.google.common.base.Charsets;
014import com.google.common.collect.Maps;
015import junit.framework.TestCase;
016import org.junit.Test;
017
018public class InterpretedCountryParserTest extends TestCase {
019
020  private InterpretedCountryParser parser = InterpretedCountryParser.getInstance();
021
022  @Test
023  public void testParse() {
024    // run a few basic tests to check it bootstraps and appears to work
025    assertEquals(Country.GERMANY, parser.parse("Germany").getPayload().getInterpreted());
026    assertEquals(Country.GERMANY, parser.parse("deutschland").getPayload().getInterpreted());
027    assertEquals(Country.GERMANY, parser.parse("De").getPayload().getInterpreted());
028  }
029
030  /**
031   * Makes sure all Country enum values are parsed ok.
032   */
033  @Test
034  public void testParseAllEnumValues() {
035
036    // make sure all
037    for (Country c : Country.OFFICIAL_COUNTRIES) {
038      assertEquals(c, parser.parse(c.getIso2LetterCode()).getPayload().getInterpreted());
039      assertEquals(c, parser.parse(c.getIso3LetterCode()).getPayload().getInterpreted());
040      assertEquals(c, parser.parse(c.getTitle()).getPayload().getInterpreted());
041    }
042  }
043
044  /**
045   * Parse all unique occurrence values found for countries and make sure parsing doesnt get worse.
046   * If the test file is updated, values here need to be adjusted!
047   */
048  @Test
049  public void testOccurrenceValues() throws IOException {
050    final int CURRENT_COUNTRIES_PARSED = 246;
051    final int CURRENT_TESTS_SUCCESSFUL = 2976;
052
053
054    // remember number of successful parsed countries:
055    Map<Country, AtomicInteger> stats = Maps.newHashMap();
056    int values = 0;
057    int failed = 0;
058    int success = 0;
059    BufferedReader r = new BufferedReader(
060      new InputStreamReader(this.getClass().getResourceAsStream("/parse/countryname/occurrence_countries.txt"),
061        Charsets.UTF_8));
062    String line;
063    do {
064      line = r.readLine();
065      values++;
066      ParseResult<InterpretedEnum<String, Country>> parsed = parser.parse(line);
067      if (ParseResult.STATUS.SUCCESS == parsed.getStatus()) {
068        Country c = parsed.getPayload().getInterpreted();
069        if (!stats.containsKey(c)) {
070          stats.put(c, new AtomicInteger(0));
071        }
072        stats.get(c).incrementAndGet();
073        success++;
074      } else {
075        failed++;
076      }
077    } while (line != null);
078
079    System.out.println(values + " tested values");
080    System.out.println(failed + " failed parse results");
081    System.out.println(success + " successful parse results");
082    System.out.println(stats.size() + " distinct countries parsed");
083    assertTrue(failed < values);
084    assertTrue(CURRENT_COUNTRIES_PARSED <= stats.size());
085    assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
086  }
087}
