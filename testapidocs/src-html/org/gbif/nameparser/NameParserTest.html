111  private ParsedName buildParsedNameFromTabRow(String[] cols) {
112    ParsedName pn = null;
113    try {
114      pn = new ParsedName(NameType.fromString(cols[13]), StringUtils.trimToNull(cols[1]),
115          StringUtils.trimToNull(cols[2]), StringUtils.trimToNull(cols[3]), StringUtils.trimToNull(cols[4]),
116          NamePart.fromString(cols[12]), StringUtils.trimToNull(cols[10]), StringUtils.trimToNull(cols[6]),
117          StringUtils.trimToNull(cols[7]), StringUtils.trimToNull(cols[8]), StringUtils.trimToNull(cols[9]),
118          StringUtils.trimToNull(cols[5]), null, null, null, null);
119    } catch (ArrayIndexOutOfBoundsException e) {
120      LOG.error("scientific_names.txt file bogus, too little columns:\n{}", Joiner.on("|").useForNull("").join(cols));
121      throw e;
122    }
123    return pn;
124  }

871  @Test
872  public void testNameFile() throws Exception {
873    final int CURRENTLY_FAIL = 24;
874    final int CURRENTLY_FAIL_EXCL_AUTHORS = 19;
875
876    LOG.info("\n\nSTARTING FULL PARSER\n");
877    Reader reader = FileUtils.getInputStreamReader(streamUtils.classpathStream("scientific_names.txt"));
878    LineIterator iter = new LineIterator(reader);
879
880    int parseFails = 0;
881    int parseNoAuthors = 0;
882    int parseErrors = 0;
883    int cnt = 0;
884    int lineNum = 0;
885    long start = System.currentTimeMillis();
886
887    Splitter tabSplit = Splitter.on('\t');
888    while (iter.hasNext()) {
889      lineNum++;
890      String line = iter.nextLine();
891      if (lineNum == 1 || line == null || line.startsWith("#") || line.trim().isEmpty()) {
892        continue;
893      }
894      cnt++;
895
896      List<String> cols = tabSplit.splitToList(line);
897      String name = cols.get(0);
898      if (cols.size() < 14) {
899        LOG.warn("Too short line in names test file:");
900        LOG.warn(line);
901      }
902      ParsedName expected = buildParsedNameFromTabRow(cols.toArray(new String[cols.size()]));
903      ParsedName n;
904      try {
905        n = parser.parse(name, null);
906        if (!n.isAuthorsParsed()) {
907          parseNoAuthors++;
908        }
909        // copy scientific name, we dont wanna compare it, it might be slightly different
910        expected.setScientificName(n.getScientificName());
911        // remove SCIENTIFIC nametype as we only like to compare other values
912        if (NameType.SCIENTIFIC == n.getType()) {
913          n.setType(null);
914        }
915        if (!n.equals(expected)) {
916          parseErrors++;
917          LOG.warn("WRONG\t " + name + "\n  EXPECTED: " + expected + "\n  PARSED  : " + n);
918        }
919      } catch (Exception e) {
920        parseFails++;
921        LOG.warn("FAIL\t " + name + "\nEXPECTED: " + expected);
922      }
923    }
924    long end = System.currentTimeMillis();
925    int successfulParses = cnt - parseFails - parseErrors;
926    LOG.info("\n\nNames tested: " + cnt);
927    LOG.info("Names success: " + successfulParses);
928    LOG.info("Names parse fail: " + parseFails);
929    LOG.info("Names parse errors: " + parseErrors);
930    LOG.info("Names ignoring authors: " + parseNoAuthors);
931    LOG.info("Total time: " + (end - start));
932    LOG.info("Average per name: " + (((double) end - start) / cnt));
933
934    if ((parseFails + parseErrors) > CURRENTLY_FAIL) {
935      TestCase.fail(
936          "We are getting worse, not better. Currently failing: " + (parseFails + parseErrors) + ". We used to fail :"
937              + CURRENTLY_FAIL);
938    } else if (parseFails + parseErrors < CURRENTLY_FAIL) {
939      LOG.info("We are getting better! Used to fail : " + CURRENTLY_FAIL);
940    }
941    if ((parseFails + parseNoAuthors) > CURRENTLY_FAIL_EXCL_AUTHORS) {
942      TestCase.fail(
943          "We are getting worse, not better. Currently failing ignroing authors: " + (parseFails + parseNoAuthors)
944              + ". Was passing:" + CURRENTLY_FAIL_EXCL_AUTHORS);
945    } else if (parseFails + parseNoAuthors < CURRENTLY_FAIL_EXCL_AUTHORS) {
946      LOG.info("We are getting better! Used to fail without authors: " + CURRENTLY_FAIL_EXCL_AUTHORS);
947    }
948
949  }
