027  @Test
028  public void testFile() throws Exception {
029    LOG.info("\n\nSTARTING FULL PARSER\n");
030    Reader reader = Files.newReader(new File("/Users/markus/Downloads/col/scinames.txt"), Charset.forName("UTF8"));
031    LineIterator iter = new LineIterator(reader);
032
033    int parseFails = 0;
034    int parseNoAuthors = 0;
035    int cnt = 0;
036    final long start = System.currentTimeMillis();
037
038    while (iter.hasNext()) {
039      cnt++;
040      String name = iter.nextLine();
041      if (name == null || name.startsWith("#") || name.trim().isEmpty()) {
042        continue;
043      }
044      try {
045        final long pStart = System.currentTimeMillis();
046        ParsedName n = parser.parse(name, null);
047        final long duration = System.currentTimeMillis() - pStart;
048        if (!n.isAuthorsParsed()) {
049          parseNoAuthors++;
050        }
051        if (duration > 100) {
052          LOG.info("SLOOOOW {}ms\t{}", duration, name);
053        } else if (duration > 20) {
054          LOG.info("SLOW {}ms\t{}", duration, name);
055        }
056        // copy scientific name, we dont wanna compare it, it might be slightly different
057      } catch (UnparsableException e) {
058        parseFails++;
059        LOG.warn("FAIL {}\t{}", e.type, name);
060      }
061    }
062    long end = System.currentTimeMillis();
063    LOG.info("\n\nNames tested: " + cnt);
064    LOG.info("Names failed: {}", parseFails);
065    LOG.info("Names ignoring authors: {}", parseNoAuthors);
066    LOG.info("Total time: {}", (end - start));
067    LOG.info("Average per name: {}", (((double) end - start) / cnt));
068  }
