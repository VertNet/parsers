001package org.gbif.nameparser;
002
003import java.util.regex.Pattern;
004
005import org.junit.Assert;
006import org.junit.Test;
007
008import static org.junit.Assert.assertEquals;
009import static org.junit.Assert.assertNull;
010
011public class NormalizeUtilsTest {
012
013  @Test
014  public void testNormalizeCitation() {
015    assertEquals("Hallo", NormalizeUtils.normalizeCitation("   Hallo "));
016    assertEquals(
017      "Benois, R. 1964. Contribution a la connaissance des Halictus malgaches (Hym. Apidae). Revue Française d'Entomologie 31: 45 60.",
018      NormalizeUtils.normalizeCitation(
019        "Benois, R. 1964. Contribution a la connaissance des Halictus malgaches (Hym. Apidae). Revue Française d'Entomologie 31: 45 60. "));
020
021    Pattern PRO_PARTE_SPLITTER = Pattern.compile("\\|+");
022    String[] ids = PRO_PARTE_SPLITTER.split("123|456|783942|1|");
023    for (String id : ids) {
024      System.out.println(id);
025    }
026  }
027
028  @Test
029  public void testTrimToNull() {
030    assertEquals("aCcepTed", NormalizeUtils.trimToNull("aCcepTed "));
031    assertEquals("nuller", NormalizeUtils.trimToNull("nuller "));
032    assertNull(NormalizeUtils.trimToNull("NuLL "));
033    assertNull(NormalizeUtils.trimToNull(" "));
034    assertNull(NormalizeUtils.trimToNull(" NULL "));
035    assertNull(NormalizeUtils.trimToNull("       "));
036    assertNull(NormalizeUtils.trimToNull("\\N "));
037  }
038
039  @Test
040  public void testNormalizeTerm() {
041    assertEquals("accepted", NormalizeUtils.normalizeTerm("aCcepTed "));
042    assertEquals("accepted", NormalizeUtils.normalizeTerm("accepted"));
043  }
044
045  @Test
046  public void testReplaceUnicodeEntities() {
047    assertEquals("Markus&Pia", NormalizeUtils.replaceUnicodeEntities("Markus&Pia"));
048    assertEquals("Markus&Pia;", NormalizeUtils.replaceUnicodeEntities("Markus&Pia;"));
049    assertEquals("Markus & Pia ; ", NormalizeUtils.replaceUnicodeEntities("Markus & Pia ; "));
050    assertEquals("&#pia;", NormalizeUtils.replaceUnicodeEntities("&#pia;"));
051    assertEquals("&#12pia;", NormalizeUtils.replaceUnicodeEntities("&#12pia;"));
052    Assert.assertTrue(NormalizeUtils.replaceUnicodeEntities(null) == null);
053    assertEquals("лобан", NormalizeUtils.replaceUnicodeEntities("&#1083;&#1086;&#1073;&#1072;&#1085;"));
054    assertEquals("лобан", NormalizeUtils.replaceUnicodeEntities("&#x43b;&#x43e;&#x431;&#x430;&#x43d;"));
055    assertEquals("лобан", NormalizeUtils.replaceUnicodeEntities("&#x43B;&#x043e;&#x0431;&#x430;&#x43D;"));
056    //    assertEquals("лобан", NormalizeUtils.replaceUnicodeEntities("\\u043b\\u0u43e\\u0431\\u0430\\u043d"));
057    //    assertEquals("лобан", NormalizeUtils.replaceUnicodeEntities("\\u043B\\u0u43E\\u0431\\u0430\\u043D"));
058    //    assertEquals("\\u43Bобан", NormalizeUtils.replaceUnicodeEntities("\\u43B\\0u43E\\u0431\\u0430\\u043D"));
059    assertEquals("\\u43b\\u43e\\u431\\u430\\u43d",
060      NormalizeUtils.replaceUnicodeEntities("\\u43b\\u43e\\u431\\u430\\u43d"));
061  }
062}
