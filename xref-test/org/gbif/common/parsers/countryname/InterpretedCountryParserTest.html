54      // remember number of successful parsed countries:
55      Map<Country, AtomicInteger> stats = Maps.newHashMap();
56      int values = 0;
57      int failed = 0;
58      int success = 0;
59      BufferedReader r = new BufferedReader(
60        new InputStreamReader(this.getClass().getResourceAsStream("/parse/countryname/occurrence_countries.txt"),
61          Charsets.UTF_8));
62      String line;
63      do {
64        line = r.readLine();
65        values++;
66        ParseResult<InterpretedEnum<String, Country>> parsed = parser.parse(line);
67        if (ParseResult.STATUS.SUCCESS == parsed.getStatus()) {
68          Country c = parsed.getPayload().getInterpreted();
69          if (!stats.containsKey(c)) {
70            stats.put(c, new AtomicInteger(0));
71          }
72          stats.get(c).incrementAndGet();
73          success++;
74        } else {
75          failed++;
76        }
77      } while (line != null);
78  
79      System.out.println(values + " tested values");
80      System.out.println(failed + " failed parse results");
81      System.out.println(success + " successful parse results");
82      System.out.println(stats.size() + " distinct countries parsed");
83      assertTrue(failed < values);
84      assertTrue(CURRENT_COUNTRIES_PARSED <= stats.size());
85      assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
