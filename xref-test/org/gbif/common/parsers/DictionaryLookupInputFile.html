17  /**
18   * A utility that will use the named dictionary, and read an input file,
19   * emiting to the output file only those records that do not have
20   * dictionary values.
21   * In essence you use this to find out which terms from a list are not in the
22   * dictionary
23   *
24   * @author tim
25   */
26  public class DictionaryLookupInputFile {
27  
28    /**
29     * @param args dictionary (e.g. basisOfRecord, countryName), inputFile, outputFile
30     */
31    public static void main(String[] args) throws IOException {
32      Parsable<String, String> p = null;
33      if ("basisOfRecord".equalsIgnoreCase(args[0])) {
34        p = BasisOfRecordParser.getInstance();
35      } else if ("countryName".equalsIgnoreCase(args[0])) {
36        p = CountryNameParser.getInstance();
37      }
38  
39      BufferedReader r = new BufferedReader(new FileReader(args[1]));
40      BufferedWriter w = new BufferedWriter(new FileWriter(args[2]));
41      String line = null;
42      List<String> results = new ArrayList<String>();
43      while ((line = r.readLine()) != null) {
44        line = StringUtils.trimToNull(line);
45        if (line != null) {
46          ParseResult<String> x = p.parse(line);
47          if (ParseResult.STATUS.FAIL == x.getStatus()) {
48            results.add(line);
49          }
50        }
51      }
52      Collections.sort(results);
53      for (String s : results) {
54        w.write(s + "\n");
55      }
56  
57      r.close();
58      w.close();
59    }
