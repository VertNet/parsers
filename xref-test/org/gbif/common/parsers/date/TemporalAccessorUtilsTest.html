14  /**
15   * Unit test for {@link TemporalAccessorUtils}.
16   *
17   */
18  public class TemporalAccessorUtilsTest {
19  
20    private ZoneId EUROPE_CENTRAL_TIME = ZoneId.of("Europe/Paris");
21  
22    private long YEAR2000_EPOCH_UTC = 946684800000l;
23    private long YEAR2000_2H_3M_4S_EPOCH_UTC = 946692184000l;
24  
25    @Test
26    public void testToUTCDate(){
27      TemporalAccessor ta = LocalDate.of(2000,01,01);
28      Date date = TemporalAccessorUtils.toUTCDate(ta);
29      assertEquals(YEAR2000_EPOCH_UTC, date.getTime());
30  
31      ta = LocalDateTime.of(2000, 01, 01, 2, 3, 4);
32      date = TemporalAccessorUtils.toUTCDate(ta);
33      assertEquals(YEAR2000_2H_3M_4S_EPOCH_UTC, date.getTime());
34    }
35  
36    @Test
37    public void testToUTCDateTimeZone(){
38      ZonedDateTime testTime = ZonedDateTime.of(2000, 1, 1, 4 , 20, 0, 0, EUROPE_CENTRAL_TIME);
39      Date date = TemporalAccessorUtils.toUTCDate(testTime);
40  
41      assertEquals(testTime.toEpochSecond(), date.getTime()/1000);
42    }
43  
44  }
