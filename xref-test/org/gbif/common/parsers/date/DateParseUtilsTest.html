211   private void assertEquivalent(String expected, ParseResult<Date> found, ParseResult.CONFIDENCE c) {
212     if (ParseResult.STATUS.ERROR == found.getStatus()) {
213       LOGGER.warn("Unexpected error found in parsing", found.getError());
214     }
215     assertEquals(ParseResult.STATUS.SUCCESS, found.getStatus());
216     SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy");
217     LOGGER.info("Expected [{}]: {}", expected, sdf.format(found.getPayload()));
218     assertEquals(c, found.getConfidence());
219     assertNotNull(found.getPayload());
220     try {
221       assertEquals(DateUtils.parseDateStrictly(expected, new String[] {"dd/MM/yyyy"}), found.getPayload());
222     } catch (ParseException e) {
223       fail("Error in the test class.  Expected date is not in the correct format of dd/MM/yyyy:" + expected);
224     }
