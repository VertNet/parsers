37     * Parse all unique occurrence values found for countries and make sure parsing doesnt get worse.
38     * If the test file is updated, values here need to be adjusted!
39     */
40    @Test
41    public void testOccurrenceValues() throws IOException {
42      final int CURRENT_COUNTRIES_PARSED = 251; //246;
43      final int CURRENT_TESTS_SUCCESSFUL = 5067;// 2976;
44  
45  
46      // remember number of successful parsed countries:
47      Map<Country, AtomicInteger> stats = Maps.newHashMap();
48      int values = 0;
49      int failed = 0;
50      int success = 0;
51      BufferedReader r = new BufferedReader(
52        new InputStreamReader(this.getClass().getResourceAsStream("/parse/countryname/occurrence_countries.txt"),
53          Charsets.UTF_8));
54      String line;
55      do {
56        line = r.readLine();
57        values++;
58        ParseResult<Country> parsed = parser.parse(line);
59        if (ParseResult.STATUS.SUCCESS == parsed.getStatus()) {
60          Country c = parsed.getPayload();
61          if (!stats.containsKey(c)) {
62            stats.put(c, new AtomicInteger(0));
63          }
64          stats.get(c).incrementAndGet();
65          success++;
66        } else {
67          failed++;
68        }
69      } while (line != null);
70  
71      System.out.println(values + " tested values");
72      System.out.println(failed + " failed parse results");
73      System.out.println(success + " successful parse results");
74      System.out.println(stats.size() + " distinct countries parsed");
75      assertTrue(failed < values);
76      assertTrue(CURRENT_COUNTRIES_PARSED <= stats.size());
77      assertTrue(CURRENT_TESTS_SUCCESSFUL <= success);
78    }
