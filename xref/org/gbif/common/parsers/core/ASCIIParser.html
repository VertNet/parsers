44    public static ASCIIParser getInstance() {
45      synchronized (ASCIIParser.class) {
46        if (singletonObject == null) {
47          singletonObject = new ASCIIParser();
48        }
49      }
50      return singletonObject;
51    }

53    @Override
54    public ParseResult<String> parse(String input) {
55      if (Strings.isNullOrEmpty(input)) {
56        return ParseResult.fail();
57      } else {
58        StringBuilder sb = new StringBuilder();
59        for (char c : input.toCharArray()) {
60          foldToASCII(c, sb);
61        }
62        return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, sb.toString());
63      }
64    }
