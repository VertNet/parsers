15  /**
16   * Implementation to take String and algorithmically convert to Date objects.
17   */
18  class StringToDateParser implements Parsable<String, Date> {
19  
20    private static final Logger LOGGER = LoggerFactory.getLogger(StringToDateParser.class);
21  
22    protected static String[] allPatterns =
23      {"dd/MM/yy", "ddMMyy", "dd\\MM\\yy", "dd.MM.yy", "dd-MM-yy", "dd_MM_yy", "MM/dd/yy", "MMddyy", "MM\\dd\\yy",
24        "MM.dd.yy", "MM-dd-yy", "MM_dd_yy", "dd/MM/yyyy", "ddMMyyyy", "dd\\MM\\yyyy", "dd.MM.yyyy", "dd-MM-yyyy",
25        "dd_MM_yyyy", "MM/dd/yyyy", "MMddyyyy", "MM\\dd\\yyyy", "MM.dd.yyyy", "MM-dd-yyyy", "MM_dd_yyyy", "yyyy-MM-dd",
26        "yyyy-MM-dd HH:mm:ss", "yyyy-MM-dd'T'HH:mm:ssZZ", "yyyy-MM-dd'T'HH:mm:ss'Z'", "yyyy-MM"};
27  
28    private static final Pattern DOUBLE_ZERO_PATTERN = Pattern.compile("00");
29    private static final Pattern DOUBLE_ZERO_ONE_PATTERN = Pattern.compile("0101");
30  
31    @Override
32    public ParseResult<Date> parse(String input) {
33      if (input.length() == 6) { // looks likely to be a 2 digit year
34        // now try and get common 2 digit years
35        Date ddmm = strictParse(input, new String[] {"ddMMyy"});
36        Date mmdd = strictParse(input, new String[] {"MMddyy"});
37  
38        if (ddmm != null && mmdd == null) {
39          return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, ddmm);
40        } else if (ddmm == null && mmdd != null) {
41          return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, mmdd);
42        } else if (ddmm != null) {
43          // dilema - could be ddMM or MMdd
44          // since we see 50% split on "MM/dd/yyyy" "dd/MM/yyyy" but
45          // it is much more common to see yyyyMMdd than yyyyddMM we
46          // assume ddmm format
47          // Hint: consider looking at several records and inferring
48          // the format by looking at outliers on 1-31 and 1-12 range
49          return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, ddmm);
50        } else {
51          // ignore and continue
52        }
53      } else {
54        // starts with the most commonly seen dates, before dropping back into
55        // a catch as many as possible algorithm
56        Date ddmm = strictParse(input, new String[] {"dd/MM/yy", "dd/MM/yyyy", "yyyyMMdd", "yyyy-MM-dd", "ddMMyyyy"});
57        Date mmdd = strictParse(input, new String[] {"MM/dd/yy", "MM/dd/yyyy", "yyyyddMM", "yyyy-dd-MM", "MMddyyyy"});
58  
59        if (ddmm != null && mmdd == null) {
60          return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, ddmm);
61        } else if (ddmm == null && mmdd != null) {
62          return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, mmdd);
63        } else if (ddmm != null) {
64          // dilemma - could be ddMM or MMdd
65          // since we see 50% split on "MM/dd/yyyy" "dd/MM/yyyy" but
66          // it is much more common to see yyyyMMdd than yyyyddMM we
67          // assume ddmm format
68          // Hint: consider looking at several records and inferring
69          // the format by looking at outliers on 1-31 and 1-12 range
70          return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, ddmm);
71        } else {
72          // ignore and continue
73        }
74  
75      }
76      // now try and catch something that looks reasonable
77      try {
78        Date d;
79        try {
80          d = fallbackParse(input, allPatterns);
81        } catch (ParseException e) {
82          // sometime we see 00 instead of 01 for days and months
83          // but if it has not separators, we might have just made
84          // 00000000 become 01010101
85          // 00/00/0000 should become 01/01/0000 though - see below
86          input = DOUBLE_ZERO_PATTERN.matcher(input).replaceAll("01");
87          input = DOUBLE_ZERO_ONE_PATTERN.matcher(input).replaceAll("00"); // so catch stupidity
88          d = fallbackParse(input, allPatterns);
89        }
90        Calendar calendar = Calendar.getInstance();
91        calendar.setTime(d);
92        LOGGER.debug("Day[{}] Month[{}] Year[{}]",
93          new Object[] {calendar.get(Calendar.DAY_OF_MONTH), calendar.get(Calendar.DAY_OF_MONTH),
94            calendar.get(Calendar.YEAR)});
95        if (d != null && !(calendar.get(Calendar.DAY_OF_MONTH) == 1 // remember above 0 becomes 1
96                           && calendar.get(Calendar.MONTH) == 1 && calendar.get(Calendar.YEAR) == 0)) {
97          // possible as this is not an exact sciensce by any means
98          return ParseResult.success(ParseResult.CONFIDENCE.POSSIBLE, d);
99        } else {
100         return ParseResult.fail();
101       }
102     } catch (ParseException e) {
103       return ParseResult.fail();
104     }
105   }

107   /**
108    * Tries strict and then lenient parsing of the date
109    *
110    * @param input    To parse
111    * @param patterns to use in parsing
112    *
113    * @return The date if possible
114    *
115    * @throws ParseException On error
116    */
117   Date fallbackParse(String input, String[] patterns) throws ParseException {
118     Date d;
119     try {
120       d = DateUtils.parseDateStrictly(input, patterns);
121     } catch (RuntimeException e) {
122       // try leniently
123       d = DateUtils.parseDate(input, patterns);
124     }
125     return d;
126   }

128   /**
129    * Parses strictly swallowing errors
130    *
131    * @param input    to parse
132    * @param patterns to use in parsing
133    *
134    * @return null on error or the date
135    */
136   Date strictParse(String input, String[] patterns) {
137     try {
138       return DateUtils.parseDateStrictly(input, patterns);
139     } catch (ParseException e) {
140       return null;
141     }
142   }
