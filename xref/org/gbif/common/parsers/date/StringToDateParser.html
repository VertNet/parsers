41    /**
42     * Verifies if day and year cannot be confused with the month and returns appropriate confidence level.
43     */
44    private static ParseResult.CONFIDENCE checkMonthDay(Date d, boolean verifyYear) {
45      if (d == null) {
46        return ParseResult.CONFIDENCE.DEFINITE;
47      }
48  
49      Calendar cal = Calendar.getInstance();
50      cal.setTime(d);
51      int day = cal.get(Calendar.DAY_OF_MONTH);
52      int month = cal.get(Calendar.MONTH) + 1;
53  
54      if (day > 12 || day == month) {
55        if (verifyYear) {
56          int year = cal.get(Calendar.YEAR);
57          int yearInCentury = year % 100;
58          if (yearInCentury > 31) {
59            return ParseResult.CONFIDENCE.DEFINITE;
60          } else if (year < 1700) {
61            return ParseResult.CONFIDENCE.POSSIBLE;
62          }
63          return ParseResult.CONFIDENCE.PROBABLE;
64        }
65        return ParseResult.CONFIDENCE.DEFINITE;
66      }
67      // day and month might have been confused
68      return ParseResult.CONFIDENCE.POSSIBLE;
69    }

71    /**
72     * Parses a numerical date or datetime string trying various formats.
73     * Parsing is done based on the string length and delimiters found in the input primarily.
74     * There is no cleaning done that removes question marks.
75     * Only superflous whitespace is removed.
76     */
77    @Override
78    public ParseResult<Date> parse(String input) {
79  
80      final int len = input.length();
81      Date d = null;
82      ParseResult.CONFIDENCE confidence = ParseResult.CONFIDENCE.DEFINITE;
83  
84      if (input.contains("年")) {
85        d = strictParse(input, ASIAN);
86  
87      } if (StringUtils.isNumeric(input)) {
88        if (len<=4) {
89          // year only
90          d = strictParse(input, "yy", "yyyy");
91  
92        } else if (len==6) {
93          // no delimiter, year as 2 digits
94          d = strictParse(input, CHAR6_ONLY_DIGITS);
95          confidence = checkMonthDay(d, true);
96          confidence = confidence == ParseResult.CONFIDENCE.DEFINITE ? ParseResult.CONFIDENCE.PROBABLE : confidence;
97        } else if (len==8) {
98          // no delimiter, year as 2 digits
99          d = strictParse(input, CHAR8_ONLY_DIGITS);
100         confidence = checkMonthDay(d, true);
101       }
102 
103     } else {
104       // find what delimiter exists how often
105       Character del = null;
106       int delimCount = 0;
107       for (char c : input.toCharArray()) {
108         if (!Character.isDigit(c)) {
109           delimCount++;
110           if (del == null) {
111             del = c;
112           } else if (!del.equals(c)) {
113             // different delimiters in use - we dont understand this at all
114             return ParseResult.fail();
115           }
116         }
117       }
118 
119       if (Character.isLetter(del)) {
120         confidence = ParseResult.CONFIDENCE.POSSIBLE;
121       }
122 
123       // check if full year is given
124       boolean fullYear = FULL_YEAR.matcher(input).find();
125 
126       if (delimCount == 1) {
127         // year & month only
128         if (fullYear) {
129           String[] formats = new String[]{"yyyy" + del + "MM", "MM" + del + "yyyy"};
130           d = strictParse(input, formats);
131 
132         } else {
133           String[] formats = new String[]{"yy" + del + "MM", "MM" + del + "yy"};
134           d = strictParse(input, formats);
135           confidence = ParseResult.CONFIDENCE.POSSIBLE;
136         }
137       } else if (delimCount == 2) {
138         if (fullYear) {
139           // first try very common formats (main US, European, Chinese, etc)
140           d = strictParse(input, FREQUENT_FULL);
141           if (d == null) {
142             String[] formats = new String[]{"yyyy" + del + "MM"+ del + "dd", "dd" + del + "MM" + del + "yyyy", "MM" + del + "dd" + del + "yyyy"};
143             d = strictParse(input, formats);
144             confidence = checkMonthDay(d, false);
145           }
146 
147         } else {
148           String[] formats;
149           if (del=='-') {
150             // iso formats use dash and start with years
151             formats = new String[]{"yy" + del + "MM" + del + "dd", "dd" + del + "MM" + del + "yy", "MM" + del + "dd" + del + "yy"};
152           } else {
153             formats = new String[]{"dd" + del + "MM" + del + "yy", "MM" + del + "dd" + del + "yy", "yy" + del + "MM" + del + "dd"};
154           }
155           d = strictParse(input, formats);
156           confidence = checkMonthDay(d, true);
157         }
158 
159       } else {
160         // more than 3 delimiters, with time?
161         d = strictParse(input, TIME_FORMATS);
162       }
163     }
164 
165     if (d != null) {
166       return ParseResult.success(confidence, d);
167     }
168     return ParseResult.fail();
