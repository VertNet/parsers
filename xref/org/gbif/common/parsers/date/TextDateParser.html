14  /**
15   * Main class to parse a date represented as a single String or as date parts into a {@link TemporalAccessor}.
16   * If the String contains letters, the {@link TextualMonthDateTokenizer} and the {@link DatePartsNormalizer} will be used.
17   * Otherwise, {@link ThreeTenNumericalDateParser} will be used.
18   */
19  public class TextDateParser implements Parsable<TemporalAccessor> {
20  
21    //private static final Pattern ISO_TIME_MARKER =  Pattern.compile("\\dT\\d");
22    //private static final Pattern AT_LEAST_ONE_LETTER =  Pattern.compile("[a-zA-Z]+");
23    //This regex is not complete and will NOT handle date when the time zone is provided as text GMT
24    private static final Pattern NUMERICAL_DATE_PATTERN =  Pattern.compile("[^a-zA-Z]+[\\dT\\d]?[^a-zA-Z]+[Z]?$");
25    private static final TextualMonthDateTokenizer TEXT_MONTH_TOKENIZER = new TextualMonthDateTokenizer();
26    private static final ThreeTenNumericalDateParser THREETEN_NUMERICAL_PARSER = ThreeTenNumericalDateParser.getParser();
27  
28    public static void main(String[] args){
29      TextDateParser dateParser = new TextDateParser();
30      ParseResult<TemporalAccessor> ta = dateParser.parse("2nd jan. 1999");
31  
32      LocalDate localDate = LocalDate.from(ta.getPayload());
33      //System.out.println(localDate);
34  
35      ta = dateParser.parse("1999", "jan.", "2");
36      localDate = LocalDate.from(ta.getPayload());
37      System.out.println(localDate);
38    }

40    @Override
41    public ParseResult<TemporalAccessor> parse(String input) {
42  
43      if(StringUtils.isBlank(input)){
44        return ParseResult.fail();
45      }
46  
47      // Check if the input text contains only punctuations and numbers
48      // Also accept the T marker (e.g. 1978-12-21T02:12) from the ISO format
49      // We could also simply try to parse it but it is probably not optimal
50      //if(!AT_LEAST_ONE_LETTER.matcher(input).find() || ISO_TIME_MARKER.matcher(input).find()) {
51      if(NUMERICAL_DATE_PATTERN.matcher(input).matches()) {
52        return THREETEN_NUMERICAL_PARSER.parse(input, DateFormatHint.NONE);
53      }
54  
55      TextualMonthDateTokenizer.DateTokens dt = TEXT_MONTH_TOKENIZER.tokenize(input);
56      // for now we only handle cases where we can find year, month, day with confidence.
57      if(!dt.containsDiscardedTokens() && dt.size() == 3){
58        DatePartsNormalizer.NormalizedYearMonthDay normalizedYearMonthDay = DatePartsNormalizer.normalize(
59                dt.getToken(TextualMonthDateTokenizer.TokenType.INT_4).getToken(),
60                dt.getToken(TextualMonthDateTokenizer.TokenType.TEXT).getToken(),
61                dt.getToken(TextualMonthDateTokenizer.TokenType.INT_2).getToken());
62  
63        //no handling for partial dates with textual month for now
64        if(normalizedYearMonthDay.getYear() != null &&
65                normalizedYearMonthDay.getMonth() != null &&
66                normalizedYearMonthDay.getDay() != null){
67          try {
68            return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE,
69                    (TemporalAccessor)LocalDate.of(normalizedYearMonthDay.getYear(),
70                            normalizedYearMonthDay.getMonth(), normalizedYearMonthDay.getDay()));
71          }
72          catch (DateTimeException ignore){
73            //simply ignore bad dates
74          }
75        }
76      }
77      return ParseResult.fail();
78    }
79  
80    /**
81     * Parse date parts into a TemporalAccessor.
82     * The {@link DatePartsNormalizer} will be applied on raw data.
83     *
84     * @param year
85     * @param month
86     * @param day
87     * @return
88     */
89    public ParseResult<TemporalAccessor> parse(String year, String month, String day) {
90      DatePartsNormalizer.NormalizedYearMonthDay normalizedYearMonthDay = DatePartsNormalizer.normalize(
91              year, month, day);
92  
93      // TODO when we can get the day but not the month in NormalizedYearMonthDay the parsing fails
94      // we could actually keep the year by calling ThreeTenNumericalDateParser.parse(normalizedYearMonthDay.getYear(),
95      // null, null); the CONFIDENCE would be set at PROBABLE.
96  
97      ParseResult<TemporalAccessor> parseResult = ThreeTenNumericalDateParser.parse(normalizedYearMonthDay.getYear(),
98              normalizedYearMonthDay.getMonth(), normalizedYearMonthDay.getDay());
99  
100     //If we got a successful parsing BUT a part of the date was discarded we reduce confidence.
101     if(parseResult.isSuccessful() && normalizedYearMonthDay.containsDiscardedPart()){
102       return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, parseResult.getPayload());
103     }
104 
105     return parseResult;
106   }
