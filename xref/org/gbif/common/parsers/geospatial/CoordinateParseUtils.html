21    /**
22     * This parses string representations of latitude and longitude values. It tries its best to interpret the values and
23     * indicates any problems in its result as {@link org.gbif.api.vocabulary.OccurrenceIssue}.
24     * When the {@link ParseResult.STATUS} is FAIL the payload will be null and one or more issues should be set
25     * in {@link ParseResult#getIssues()}.
26     *
27     * Coordinate precision will be 5 decimals at most, any more precise values will be rounded.
28     *
29     * @param latitude  The decimal latitude
30     * @param longitude The decimal longitude
31     *
32     * @return The parse result
33     */
34    public static ParseResult<LatLng> parseLatLng(final String latitude, final String longitude) {
35      return new Parsable<LatLng>() {
36        @Override
37        public ParseResult<LatLng> parse(String v) {
38          Double lat;
39          Double lng;
40          try {
41            lat = roundTo5decimals(Double.parseDouble(latitude));
42            lng = roundTo5decimals(Double.parseDouble(longitude));
43          } catch (NumberFormatException e) {
44            return ParseResult.error(e);
45          }
46  
47          // 0,0 is too suspicious
48          if (Double.compare(lat, 0) == 0 && Double.compare(lng, 0) == 0) {
49            return ParseResult
50              .success(ParseResult.CONFIDENCE.POSSIBLE, new LatLng(0, 0), OccurrenceIssue.ZERO_COORDINATE);
51          }
52  
53          // if everything falls in range
54          if (Double.compare(lat, 90) <= 0 && Double.compare(lat, -90) >= 0 && Double.compare(lng, 180) <= 0
55              && Double.compare(lng, -180) >= 0) {
56            return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new LatLng(lat, lng));
57          }
58  
59          // if lat is out of range, but in range of the lng,
60          // assume swapped coordinates.
61          // note that should we desire to trust the following records, we would need to clear the flag for the records to
62          // appear in
63          // search results and maps etc. however, this is logic decision, that goes above the capabilities of this method
64          if (Double.compare(lat, 90) > 0 || Double.compare(lat, -90) < 0) {
65  
66            // try and swap
67            if (Double.compare(lng, 90) <= 0 && Double.compare(lng, -90) >= 0 && Double.compare(lat, 180) <= 0
68                && Double.compare(lat, -180) >= 0) {
69              return ParseResult.fail(new LatLng(lat, lng), OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
70            }
71          }
72  
73          // then something is out of range
74          return ParseResult.fail(OccurrenceIssue.COORDINATES_OUT_OF_RANGE);
75        }
76      }.parse(null);
77    }
78  
79    // round to 5 decimals (~1m precision) since no way we're getting anything legitimately more precise
80    private static Double roundTo5decimals(Double x) {
81      return x == null ? null : Math.round(x * Math.pow(10, 5)) / Math.pow(10, 5);
82    }
