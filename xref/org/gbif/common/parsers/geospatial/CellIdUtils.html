27    /**
28     * Determines the cell id for the Lat / Long provided.
29     *
30     * @param latitude  Which may be null
31     * @param longitude Which may be null
32     *
33     * @return The cell id for the Lat Long pair
34     *
35     * @throws UnableToGenerateCellIdException
36     *          Should the lat long be null or invalid
37     */
38    public static int toCellId(Double latitude, Double longitude) throws UnableToGenerateCellIdException {
39      LOGGER.debug("Getting cell for [{},{}]", latitude, longitude);
40  
41      if (latitude == null || latitude < MIN_LATITUDE || latitude > MAX_LATITUDE || longitude < MIN_LONGITUDE
42          || longitude > MAX_LONGITUDE) {
43        throw new UnableToGenerateCellIdException(
44          "Latitude[" + latitude + "], Longitude[" + longitude + "] cannot be converted to a cell id");
45      } else {
46        int la = getCellIdFor(latitude);
47        int lo = getMod360CellIdFor(longitude);
48        return la + lo;
49      }
50    }

66    /**
67     * Returns true if the supplied cell id lies in the bounding box demarcated by the min and max cell ids supplied.
68     */
69    public static boolean isCellIdInBoundingBox(int cellId, int minCellId, int maxCellId) throws Exception {
70      return cellId >= minCellId && cellId <= (maxCellId - 361) && (cellId % 360) >= (minCellId % 360)
71             && (cellId % 360) <= ((maxCellId - 361) % 360);

74    /**
75     * Determines the centi cell id for the given values
76     *
77     * @param latitude  Which may be null
78     * @param longitude Which may be null
79     *
80     * @return The centi cell id within the cell for the lat long
81     *
82     * @throws UnableToGenerateCellIdException
83     *          Shoudl the lat long be null or invalid
84     */
85    public static int toCentiCellId(Double latitude, Double longitude) throws UnableToGenerateCellIdException {
86      if (latitude == null || latitude < MIN_LATITUDE || latitude > MAX_LATITUDE || longitude < MIN_LONGITUDE
87          || longitude > MAX_LONGITUDE) {
88        throw new UnableToGenerateCellIdException(
89          "Latitude[" + latitude + "], Longitude[" + longitude + "] cannot be " + "converted to a centi cell id");
90      } else {
91  
92        //get decimal value for up to 4 decimal places
93        //17.2-> 172000 -> 2000
94        int la = Math.abs((int) (latitude * 10000) % 10000);
95        if (latitude < 0) la = 10000 - la;
96        la = (la / 1000) % 10;
97        int lo = Math.abs((int) (longitude * 10000) % 10000);
98        if (longitude < 0) lo = 10000 - lo;
99        lo = (lo / 1000) % 10;
100 
101       int centiCellId = (la * 10) + lo;
102       return Math.abs(centiCellId);
103     }
104   }

108    *
109    * @param cellId To return the lat long box of

111    * @return The box
112    */
113   public static LatLngBoundingBox toBoundingBox(int cellId) {
114     int longitude = (cellId % 360) - MAX_LONGITUDE;
115     int latitude = MIN_LATITUDE;
116     if (cellId > 0) {
117       latitude = new Double(Math.floor(cellId / 360)).intValue() - MAX_LATITUDE;
118     }
119     return new LatLngBoundingBox(longitude, latitude, longitude + 1, latitude + 1);

122   /**
123    * Returns the box of the given cell and centi cell An attempt has been made to avoid rounding errors with doubles,
124    * but may need revisited
125    *
126    * @param cellId      To return the lat long box of
127    * @param centiCellId within the box
128    *
129    * @return The box
130    */
131   public static LatLngBoundingBox toBoundingBox(int cellId, int centiCellId) {
132     int longitudeX10 = 10 * ((cellId % 360) - MAX_LONGITUDE);
133     int latitudeX10 = -900;
134     if (cellId > 0) {
135       latitudeX10 = 10 * (new Double(Math.floor(cellId / 360)).intValue() - MAX_LATITUDE);
136     }
137 
138     double longOffset = (centiCellId % 10);
139     double latOffset = 0;
140     if (centiCellId > 0) {
141       latOffset = centiCellId / 10;
142     }
143 
144     double minLatitude = (latitudeX10 + latOffset) / 10;
145     double minLongitude = (longitudeX10 + longOffset) / 10;
146     double maxLatitude = (latitudeX10 + latOffset + 1) / 10;
147     double maxlongitude = (longitudeX10 + longOffset + 1) / 10;
148     return new LatLngBoundingBox(minLongitude, minLatitude, maxlongitude, maxLatitude);
149   }

151   /**
152    * Gets the list of cells that are enclosed within the bounding box. Cells that are partially enclosed are returned
153    * also
154    * TODO implement this properly, the current version will include cells that are partially included on the bottom and
155    * left, but not the top and right
156    *
157    * @return The cells that are enclosed by the bounding box
158    *
159    * @throws UnableToGenerateCellIdException
160    *          if the lat longs are invalid
161    */
162   public static Set<Integer> getCellsEnclosedBy(double minLat, double maxLat, double minLong, double maxLong)
163     throws UnableToGenerateCellIdException {
164     if (minLat < MIN_LATITUDE) minLat = MIN_LATITUDE;
165     if (maxLat > MAX_LATITUDE) maxLat = MAX_LATITUDE;
166     if (minLong < MIN_LONGITUDE) minLong = MIN_LONGITUDE;
167     if (maxLong > MAX_LONGITUDE) maxLong = MAX_LONGITUDE;
168 
169     LOGGER.debug("Establishing cells enclosed by: {}:{}   {}:{}", new Object[] {minLat, maxLat, minLong, maxLong});
170 
171     int lower = toCellId(minLat, minLong);
172     int upper = toCellId(maxLat, maxLong);
173 
174     LOGGER.debug("Unprocessed cells: {} -> {}", lower, upper);
175 
176     // if the BB upper right corner is on a grid, then it needs flagged
177     if (Math.ceil(maxLong) == Math.floor(maxLong)) {
178       LOGGER.debug("Longitude lies on a boundary");
179       upper -= 1;
180     }
181     if (Math.ceil(maxLat) == Math.floor(maxLat)) {
182       LOGGER.debug("Latitude lies on a boundary");
183       upper -= 360;
184     }
185 
186     LOGGER.debug("Getting cells contained in {} to {}", lower, upper);
187 
188     int omitLeft = lower % 360;
189     int omitRight = upper % 360;
190     if (omitRight == 0) omitRight = 360;
191     Set<Integer> cells = new HashSet<Integer>();
192     for (int i = lower; i <= upper; i++) {
193       if (i % 360 >= omitLeft && i % 360 <= omitRight) {
194         cells.add(i);
195       }
196     }
197     return cells;
198   }

201    * Return a min cell id and a max cell id for this bounding box.
202    *
203    * @return the minCellId in int[0] and maxCellId in int[1]
204    */
205   public static int[] getMinMaxCellIdsForBoundingBox(double minLongitude, double minLatitude, double maxLongitude,
206     double maxLatitude) throws UnableToGenerateCellIdException {
207 
208     int minCellId = toCellId(minLatitude, minLongitude);
209     int maxCellId = toCellId(maxLatitude, maxLongitude);
210 
211     if (Math.ceil(maxLatitude) == Math.floor(maxLatitude) && Math.ceil(maxLongitude) == Math.floor(maxLongitude)
212         && maxLongitude != 180f && maxLatitude != 90f && maxCellId > 0) {
213 
214       //the maxLongitude,maxLatitude point is on a cell intersection, hence the maxCellId should be
215       // -361 the maxCellId CellIdUtils will give us i.e. the cell that is
216       // 1 below and 1 to the left of the cell id retrieved
217       //unless it is the 64799 cell.
218       maxCellId = maxCellId - 361;
219     }
220 
221     return new int[] {minCellId, maxCellId};
222   }

313   private static boolean isBoundingBoxCentiCell(double minLongitude, double minLatitude, double maxLongitude,
314     double maxLatitude) {
315     double width = maxLongitude > minLongitude ? maxLongitude - minLongitude : minLongitude - maxLongitude;
316     double height = maxLatitude > minLatitude ? maxLatitude - minLatitude : minLatitude - maxLatitude;
317     return Precision.round(height, 1) == 0.1f && Precision.round(width, 1) == 0.1f;
318   }
