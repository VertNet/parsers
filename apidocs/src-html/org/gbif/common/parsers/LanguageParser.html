001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.Language;
004import org.gbif.common.parsers.core.EnumParser;
005
006import java.io.BufferedReader;
007import java.io.IOException;
008import java.io.InputStreamReader;
009import java.util.Iterator;
010
011import com.google.common.base.Charsets;
012import com.google.common.base.Splitter;
013
014/**
015 * Singleton implementation of the case insensitive iso 639-1 language dictionary
016 * that uses the org.gbif.common.api.Language enumeration.
017 * Parsing results will be upper cased 2 letter codes.
018 * Understood inputs are 2 or 3 letter (both terminological and bibliographical) iso codes
019 * and natural language names given in any of the iso languages.
020 */
021public class LanguageParser extends EnumParser<Language> {
022
023  private static LanguageParser singletonObject = null;
024  private static final Splitter TAB_SPLITTER = Splitter.on('\t').trimResults();
025  private static final Splitter SEMICOLON_SPLITTER = Splitter.on(';').trimResults();
026
027
028  private LanguageParser() {
029    super(Language.class, false);
030    try {
031      BufferedReader r = new BufferedReader(
032        new InputStreamReader(LanguageParser.class.getResourceAsStream("/dictionaries/parse/iso-639-1.txt"),
033          Charsets.UTF_8));
034      String line;
035      do {
036        line = r.readLine();
037        if (line != null) {
038          Iterator<String> cols = TAB_SPLITTER.split(line).iterator();
039          // iso 2 letter code
040          String iso = cols.next().toLowerCase();
041          Language lang = Language.fromIsoCode(iso);
042          if (lang == null) {
043            log.warn("ISO code {} not part of our language enumeration", iso);
044            continue;
045          }
046          // ignore iso 3 letter code - handled by enum already
047          cols.next();
048          // ignore english name - handled by enum already
049          cols.next();
050          // ignore native name - handled by enum already
051          cols.next();
052          // make use of all translations!
053          for (String s : SEMICOLON_SPLITTER.split(cols.next())) {
054            add(s, lang);
055          }
056        }
057      } while (line != null);
058    } catch (IOException e) {
059      log.error("Cannot initiate language parser: {}", e.getMessage());
060      throw new IllegalStateException("Cannot initiate language parser", e);
061    }
062
063    // also make sure we have all enum values mapped
064    for (Language r : Language.values()) {
065      add(r.name(), r);
066      add(r.getTitleEnglish(), r);
067      add(r.getTitleNative(), r);
068      add(r.getIso2LetterCode(), r);
069      add(r.getIso3LetterCode(), r);
070    }
071  }
072
073  @Override
074  protected String normalize(String value) {
075    if (value != null) {
076      /**
077       * A language string could come in as a locale like "en_US" or if it was constructed improperly "eng_US", so
078       * extract only the part before the underscore". Only if it contains an "_" is parsing attempted.
079       * Whether it actually represents an iso 369 language code is left for the language parser to determine.
080       */
081      if (value.contains("_")) {
082        int index = value.indexOf("_");
083        return super.normalize(value.substring(0, index));
084      }
085      return super.normalize(value);
086    }
087    return null;
088  }
089
090  public static LanguageParser getInstance() {
091    synchronized (LanguageParser.class) {
092      if (singletonObject == null) {
093        singletonObject = new LanguageParser();
094      }
095    }
096    return singletonObject;
097  }
098}
