001package org.gbif.common.parsers;
002
003import org.gbif.api.vocabulary.Language;
004import org.gbif.common.parsers.core.EnumParser;
005
006import java.io.BufferedReader;
007import java.io.IOException;
008import java.io.InputStreamReader;
009import java.util.Iterator;
010
011import com.google.common.base.CharMatcher;
012import com.google.common.base.Charsets;
013import com.google.common.base.Splitter;
014import com.google.common.base.Strings;
015
016/**
017 * Singleton implementation of the case insensitive iso 639-1 language dictionary
018 * that uses the org.gbif.common.api.Language enumeration.
019 * Parsing results will be upper cased 2 letter codes.
020 * Understood inputs are 2 or 3 letter (both terminological and bibliographical) iso codes
021 * and natural language names given in any of the iso languages.
022 */
023public class LanguageParser extends EnumParser<Language> {
024
025  private static LanguageParser singletonObject = null;
026  private static final CharMatcher LETTER_MATCHER = CharMatcher.JAVA_LETTER.or(CharMatcher.WHITESPACE).precomputed();
027  private static final CharMatcher WHITESPACE_MATCHER = CharMatcher.WHITESPACE.precomputed();
028  private static final Splitter TAB_SPLITTER = Splitter.on('\t').trimResults();
029  private static final Splitter SEMICOLON_SPLITTER = Splitter.on(';').trimResults();
030
031
032  private LanguageParser() {
033    super(Language.class, true);
034    try {
035      BufferedReader r = new BufferedReader(
036        new InputStreamReader(LanguageParser.class.getResourceAsStream("/dictionaries/parse/iso-639-1.txt"),
037          Charsets.UTF_8));
038      String line;
039      do {
040        line = r.readLine();
041        if (line != null) {
042          Iterator<String> cols = TAB_SPLITTER.split(line).iterator();
043          // iso 2 letter code
044          String iso = cols.next().toLowerCase();
045          Language lang = Language.fromIsoCode(iso);
046          if (lang == null) {
047            log.warn("ISO code {} not part of our language enumeration", iso);
048            continue;
049          }
050          // ignore iso 3 letter code - handled by enum already
051          cols.next();
052          // ignore english name - handled by enum already
053          cols.next();
054          // ignore native name - handled by enum already
055          cols.next();
056          // make use of all translations!
057          for (String s : SEMICOLON_SPLITTER.split(cols.next())) {
058            add(s, lang);
059          }
060        }
061      } while (line != null);
062    } catch (IOException e) {
063      log.error("Cannot initiate language parser: {}", e.getMessage());
064      throw new IllegalStateException("Cannot initiate language parser", e);
065    }
066
067    // also make sure we have all enum values mapped
068    for (Language r : Language.values()) {
069      add(r.name(), r);
070      add(r.getTitleEnglish(), r);
071      add(r.getTitleNative(), r);
072      add(r.getIso2LetterCode(), r);
073      add(r.getIso3LetterCode(), r);
074    }
075  }
076
077  /**
078   * A language string could come in as a locale like "en_US" or if it was constructed improperly "eng_US", so
079   * extract only the part before the underscore". Only if it contains an "_" is parsing attempted.
080   * Whether it actually represents an iso 369 language code is left for the language parser to determine.
081   *
082   * @param input language string (possibly representing a Locale)
083   *
084   * @return parsed string if it contained an underscore, unchanged string if it didn't, or null if it was empty or
085   *         null
086   *         to begin with
087   */
088  private String extractLanguageFromLocale(String input) {
089    if (!Strings.isNullOrEmpty(input) && input.contains("_")) {
090      int index = input.indexOf("_");
091      return input.substring(0, index);
092    }
093    return input;
094  }
095
096  @Override
097  protected String normalize(String value) {
098    if (value != null) {
099      String cleaned = extractLanguageFromLocale(value);
100      cleaned = LETTER_MATCHER.retainFrom(cleaned);
101      cleaned = WHITESPACE_MATCHER.trimAndCollapseFrom(cleaned, ' ');
102      cleaned = Strings.emptyToNull(cleaned);
103      return super.normalize(cleaned);
104    }
105    return null;
106  }
107
108  public static LanguageParser getInstance() {
109    synchronized (LanguageParser.class) {
110      if (singletonObject == null) {
111        singletonObject = new LanguageParser();
112      }
113    }
114    return singletonObject;
115  }
116}
