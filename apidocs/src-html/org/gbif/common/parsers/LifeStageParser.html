001package org.gbif.common.parsers;
002
003import org.gbif.api.util.VocabularyUtils;
004import org.gbif.api.vocabulary.LifeStage;
005import org.gbif.common.parsers.core.EnumParser;
006
007import com.google.common.base.CharMatcher;
008import com.google.common.base.Strings;
009
010/**
011 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/continents.txt.
012 */
013public class LifeStageParser extends EnumParser<LifeStage> {
014
015  private static LifeStageParser singletonObject = null;
016
017  private static final CharMatcher LETTER_MATCHER = CharMatcher.JAVA_LETTER.or(CharMatcher.WHITESPACE).precomputed();
018  private static final CharMatcher WHITESPACE_MATCHER = CharMatcher.WHITESPACE.precomputed();
019
020
021  private LifeStageParser() {
022    super(LifeStage.class, false);
023    // also make sure we have all official iso countries mapped
024    for (LifeStage c : LifeStage.values()) {
025      add(c.name(), c);
026    }
027    // use dict file last
028    init(LifeStageParser.class.getResourceAsStream("/dictionaries/parse/life_stage.txt"));
029  }
030
031  @Override
032  protected String normalize(String value) {
033    if (value != null) {
034      String cleaned = LETTER_MATCHER.retainFrom(value);
035      cleaned = WHITESPACE_MATCHER.trimAndCollapseFrom(cleaned, ' ');
036      cleaned = Strings.emptyToNull(cleaned);
037      return super.normalize(cleaned);
038    }
039    return null;
040  }
041
042  @Override
043  protected LifeStage fromDictFile(String value) {
044    try {
045      return (LifeStage) VocabularyUtils.lookupEnum(value, LifeStage.class);
046    } catch (RuntimeException e) {
047      return null;
048    }
049  }
050
051  public static LifeStageParser getInstance() {
052    synchronized (LifeStageParser.class) {
053      if (singletonObject == null) {
054        singletonObject = new LifeStageParser();
055      }
056    }
057    return singletonObject;
058  }
059
060}
