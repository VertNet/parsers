001package org.gbif.common.parsers.rank;
002
003import org.gbif.api.vocabulary.Rank;
004import org.gbif.common.parsers.InterpretedEnumParser;
005import org.gbif.common.parsers.Parsable;
006import org.gbif.common.parsers.language.InterpretedLanguageParser;
007
008
009/**
010 * Interpreted enum parser version of TypeStatusParser.
011 */
012public class InterpretedRankParser extends InterpretedEnumParser<Rank> {
013
014  private static InterpretedRankParser singletonObject = null;
015
016  private InterpretedRankParser(Class<Rank> clazz, Parsable<String, String> parser) {
017    super(clazz, parser);
018  }
019
020  public static InterpretedRankParser getInstance()
021    throws ClassCastException, AbstractMethodError, ArithmeticException, ArrayIndexOutOfBoundsException {
022    synchronized (InterpretedLanguageParser.class) {
023      if (singletonObject == null) {
024        singletonObject = new InterpretedRankParser(Rank.class, RankParser.getInstance());
025      }
026    }
027    return singletonObject;
028  }
029
030}
