001package org.gbif.common.parsers.date;
002
003import javax.annotation.Nullable;
004
005import com.google.common.base.Strings;
006import org.apache.commons.lang3.builder.EqualsBuilder;
007import org.apache.commons.lang3.builder.HashCodeBuilder;
008import org.apache.commons.lang3.builder.ToStringBuilder;
009import org.apache.commons.lang3.builder.ToStringStyle;
010
011/**
012 * @deprecated replaced by AtomizedLocalDate + threeten
013 */
014@Deprecated
015public class YearMonthDay {
016
017  protected String year;
018  protected String month;
019  protected String day;
020
021  public YearMonthDay(String year, String month, String day) {
022    setYear(year);
023    setMonth(month);
024    setDay(day);
025  }
026
027  public YearMonthDay() {
028  }
029
030  public String getYear() {
031    return year;
032  }
033
034  /**
035   * @return day value or null, catching exceptions
036   */
037  public Integer getIntegerYear() {
038    return parseInt(year);
039  }
040
041  private static Integer parseInt(String x) {
042    if (!Strings.isNullOrEmpty(x)) {
043      try {
044        return Integer.valueOf(x);
045      } catch (NumberFormatException e) {
046        // swallow
047      }
048    }
049    return null;
050  }
051
052  public void setYear(@Nullable String year) {
053    this.year = year;
054    if (year != null) {
055      for (int i = year.length(); i < 4; i++) {
056        this.year = "0" + year;
057      }
058    }
059  }
060
061  public String getMonth() {
062    return month;
063  }
064
065  /**
066   * @return day value or null, catching exceptions
067   */
068  public Integer getIntegerMonth() {
069    return parseInt(month);
070  }
071
072  public void setMonth(@Nullable String month) {
073    this.month = month;
074    if (month != null) {
075      for (int i = month.length(); i < 2; i++) {
076        this.month = "0" + month;
077      }
078    }
079  }
080
081  public String getDay() {
082    return day;
083  }
084
085  /**
086   * @return day value or null, catching exceptions
087   */
088  public Integer getIntegerDay() {
089    return parseInt(day);
090  }
091
092  public void setDay(@Nullable String day) {
093    this.day = day;
094    if (day != null) {
095      for (int i = day.length(); i < 2; i++) {
096        this.day = "0" + day;
097      }
098    }
099  }
100
101  /**
102   * @return true if year month and day are all null
103   */
104  public boolean representsNull() {
105    return year == null && month == null && day == null;
106  }
107
108  @Override
109  public int hashCode() {
110    return new HashCodeBuilder().append(year).append(month).append(day).toHashCode();
111  }
112
113  @Override
114  public boolean equals(Object obj) {
115    if (obj instanceof YearMonthDay) {
116      YearMonthDay ymd = (YearMonthDay) obj;
117      return new EqualsBuilder().append(year, ymd.getYear()).append(month, ymd.getMonth()).append(day, ymd.getDay())
118        .isEquals();
119    } else {
120      return false;
121    }
122  }
123
124  @Override
125  public String toString() {
126    return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE).append("year", year).append("month", month)
127      .append("day", day).toString();
128  }
129}
