001package org.gbif.common.parsers.date;
002
003import javax.annotation.Nullable;
004
005import org.apache.commons.lang3.builder.EqualsBuilder;
006import org.apache.commons.lang3.builder.HashCodeBuilder;
007import org.apache.commons.lang3.builder.ToStringBuilder;
008import org.apache.commons.lang3.builder.ToStringStyle;
009
010public class YearMonthDay {
011
012  protected String year;
013  protected String month;
014  protected String day;
015
016  public YearMonthDay(String year, String month, String day) {
017    setYear(year);
018    setMonth(month);
019    setDay(day);
020  }
021
022  public YearMonthDay() {
023  }
024
025  public String getYear() {
026    return year;
027  }
028
029  public Integer getIntegerYear() {
030    return year == null ? null : Integer.valueOf(year);
031  }
032
033  public void setYear(@Nullable String year) {
034    this.year = year;
035    if (year != null) {
036      for (int i = year.length(); i < 4; i++) {
037        this.year = "0" + year;
038      }
039    }
040  }
041
042  public String getMonth() {
043    return month;
044  }
045
046  public Integer getIntegerMonth() {
047    return month == null ? null : Integer.valueOf(month);
048  }
049
050  public void setMonth(@Nullable String month) {
051    this.month = month;
052    if (month != null) {
053      for (int i = month.length(); i < 2; i++) {
054        this.month = "0" + month;
055      }
056    }
057  }
058
059  public String getDay() {
060    return day;
061  }
062
063  public void setDay(@Nullable String day) {
064    this.day = day;
065    if (day != null) {
066      for (int i = day.length(); i < 2; i++) {
067        this.day = "0" + day;
068      }
069    }
070  }
071
072  /**
073   * @return true if year month and day are all null
074   */
075  public boolean representsNull() {
076    return year == null && month == null && day == null;
077  }
078
079  @Override
080  public int hashCode() {
081    return new HashCodeBuilder().append(year).append(month).append(day).toHashCode();
082  }
083
084  @Override
085  public boolean equals(Object obj) {
086    if (obj instanceof YearMonthDay) {
087      YearMonthDay ymd = (YearMonthDay) obj;
088      return new EqualsBuilder().append(year, ymd.getYear()).append(month, ymd.getMonth()).append(day, ymd.getDay())
089        .isEquals();
090    } else {
091      return false;
092    }
093  }
094
095  @Override
096  public String toString() {
097    return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE).append("year", year).append("month", month)
098      .append("day", day).toString();
099  }
100}
