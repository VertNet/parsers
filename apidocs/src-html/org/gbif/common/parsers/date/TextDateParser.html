001package org.gbif.common.parsers.date;
002
003import org.gbif.common.parsers.core.Parsable;
004import org.gbif.common.parsers.core.ParseResult;
005import org.gbif.common.parsers.date.threeten.ThreeTenNumericalDateParser;
006
007import java.util.regex.Pattern;
008
009import org.apache.commons.lang3.StringUtils;
010import org.threeten.bp.DateTimeException;
011import org.threeten.bp.LocalDate;
012import org.threeten.bp.temporal.TemporalAccessor;
013
014/**
015 * Main class to parse a date represented as a single String or as date parts into a {@link TemporalAccessor}.
016 * If the String contains letters, the {@link TextualMonthDateTokenizer} and the {@link DatePartsNormalizer} will be used.
017 * Otherwise, {@link ThreeTenNumericalDateParser} will be used.
018 */
019public class TextDateParser implements Parsable<TemporalAccessor> {
020
021  //private static final Pattern ISO_TIME_MARKER =  Pattern.compile("\\dT\\d");
022  //private static final Pattern AT_LEAST_ONE_LETTER =  Pattern.compile("[a-zA-Z]+");
023  //This regex is not complete and will NOT handle date when the time zone is provided as text GMT
024  private static final Pattern NUMERICAL_DATE_PATTERN =  Pattern.compile("[^a-zA-Z]+[\\dT\\d]?[^a-zA-Z]+[Z]?$");
025  private static final TextualMonthDateTokenizer TEXT_MONTH_TOKENIZER = TextualMonthDateTokenizer.newInstance();
026  private static final ThreeTenNumericalDateParser THREETEN_NUMERICAL_PARSER = ThreeTenNumericalDateParser.newInstance();
027  private static final DatePartsNormalizer DATE_PARTS_NORMALIZER = DatePartsNormalizer.newInstance();
028
029  @Override
030  public ParseResult<TemporalAccessor> parse(String input) {
031
032    if(StringUtils.isBlank(input)){
033      return ParseResult.fail();
034    }
035
036    // Check if the input text contains only punctuations and numbers
037    // Also accept the T marker (e.g. 1978-12-21T02:12) from the ISO format
038    // We could also simply try to parse it but it is probably not optimal
039    //if(!AT_LEAST_ONE_LETTER.matcher(input).find() || ISO_TIME_MARKER.matcher(input).find()) {
040    if(NUMERICAL_DATE_PATTERN.matcher(input).matches()) {
041      return THREETEN_NUMERICAL_PARSER.parse(input, DateFormatHint.NONE);
042    }
043
044    TextualMonthDateTokenizer.DateTokens dt = TEXT_MONTH_TOKENIZER.tokenize(input);
045    // for now we only handle cases where we can find year, month, day with confidence.
046    if(!dt.containsDiscardedTokens() && dt.size() == 3){
047      DatePartsNormalizer.NormalizedYearMonthDay normalizedYearMonthDay = DATE_PARTS_NORMALIZER.normalize(
048              dt.getToken(TextualMonthDateTokenizer.TokenType.INT_4).getToken(),
049              dt.getToken(TextualMonthDateTokenizer.TokenType.TEXT).getToken(),
050              dt.getToken(TextualMonthDateTokenizer.TokenType.INT_2).getToken());
051
052      //no handling for partial dates with textual month for now
053      if(normalizedYearMonthDay.getYear() != null &&
054              normalizedYearMonthDay.getMonth() != null &&
055              normalizedYearMonthDay.getDay() != null){
056        try {
057          return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE,
058                  (TemporalAccessor)LocalDate.of(normalizedYearMonthDay.getYear(),
059                          normalizedYearMonthDay.getMonth(), normalizedYearMonthDay.getDay()));
060        }
061        catch (DateTimeException ignore){
062          //simply ignore bad dates
063        }
064      }
065    }
066    return ParseResult.fail();
067  }
068
069  /**
070   * Parse date parts into a TemporalAccessor.
071   * The {@link DatePartsNormalizer} will be applied on raw data.
072   *
073   * @param year
074   * @param month
075   * @param day
076   * @return
077   */
078  public ParseResult<TemporalAccessor> parse(String year, String month, String day) {
079    DatePartsNormalizer.NormalizedYearMonthDay normalizedYearMonthDay = DATE_PARTS_NORMALIZER.normalize(
080            year, month, day);
081
082    // TODO when we can get the day but not the month in NormalizedYearMonthDay the parsing fails
083    // we could actually keep the year by calling ThreeTenNumericalDateParser.parse(normalizedYearMonthDay.getYear(),
084    // null, null); the CONFIDENCE would be set at PROBABLE.
085
086    ParseResult<TemporalAccessor> parseResult = ThreeTenNumericalDateParser.parse(normalizedYearMonthDay.getYear(),
087            normalizedYearMonthDay.getMonth(), normalizedYearMonthDay.getDay());
088
089    //If we got a successful parsing BUT a part of the date was discarded we reduce confidence.
090    if(parseResult.isSuccessful() && normalizedYearMonthDay.containsDiscardedPart()){
091      return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, parseResult.getPayload());
092    }
093
094    return parseResult;
095  }
096
097}
