001package org.gbif.common.parsers.date;
002
003import org.threeten.bp.format.DateTimeFormatter;
004import org.threeten.bp.format.DateTimeFormatterBuilder;
005import org.threeten.bp.format.ResolverStyle;
006import org.threeten.bp.format.SignStyle;
007import org.threeten.bp.temporal.ChronoField;
008
009/**
010 * Factories for date parsing related instances.
011 */
012public class DateParsers {
013
014  /**
015   * Do not use for now, see  DateParsersTest
016   * Different with DateTimeFormatter.ISO_LOCAL_DATE is that DAY and MONTH are optionals
017   */
018  public static final DateTimeFormatter ISO_LOCAL_PARTIAL_DATE =
019          new DateTimeFormatterBuilder()
020                  .appendValue(ChronoField.YEAR, 4, 4, SignStyle.NEVER)
021                  .optionalStart().appendLiteral('-')
022                  .appendValue(ChronoField.MONTH_OF_YEAR, 1, 2, SignStyle.NEVER)
023                  .optionalStart().appendLiteral('-')
024                  .appendValue(ChronoField.DAY_OF_MONTH, 1, 2, SignStyle.NEVER)
025                  .optionalEnd()
026                  .optionalEnd()
027                  .toFormatter().withResolverStyle(ResolverStyle.STRICT);
028
029  /**
030   * {@link DateTimeFormatter} for ISO Year (4 digits)
031   */
032  public static final DateTimeFormatter ISO_YEAR =
033          new DateTimeFormatterBuilder()
034                  .appendValue(ChronoField.YEAR, 4, 4, SignStyle.NEVER)
035                  .toFormatter().withResolverStyle(ResolverStyle.STRICT);
036
037  /**
038   * {@link DateTimeFormatter} for ISO Year-MonthOfYear (4 digits year)
039   */
040  public static final DateTimeFormatter ISO_YEAR_MONTH =
041          new DateTimeFormatterBuilder()
042                  .appendValue(ChronoField.YEAR, 4, 4, SignStyle.NEVER)
043                  .appendLiteral('-')
044                  .appendValue(ChronoField.MONTH_OF_YEAR, 1, 2, SignStyle.NEVER)
045                  .toFormatter().withResolverStyle(ResolverStyle.STRICT);
046
047  /**
048   * Return a pre-configured {@link TemporalParser} instance.
049   */
050  public static TemporalParser defaultTemporalParser() {
051    return new TextDateParser();
052  }
053
054  /**
055   * Get a new instance of the default implementation of TemporalParser that handles
056   * numerical dates.
057   */
058  public static TemporalParser defaultNumericalDateParser() {
059    return ThreeTenNumericalDateParser.newInstance();
060  }
061
062  /**
063   * Do not use for now, see  DateParsersTest
064   * Returns a new instance of DateTimeParser with support for partial local dates LocalDate, YearMonth and Year
065   * @return DateTimeParser
066   */
067  public static DateTimeParser isoLocalPartialDateParser(){
068    return new DateTimeParser(ISO_LOCAL_PARTIAL_DATE, null, DateFormatHint.YMD, 4);
069  }
070
071}
