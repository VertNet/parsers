001package org.gbif.common.parsers.date;
002
003import java.util.Date;
004import javax.annotation.Nullable;
005
006import com.google.common.base.Optional;
007import org.threeten.bp.DateTimeUtils;
008import org.threeten.bp.LocalDate;
009import org.threeten.bp.LocalDateTime;
010import org.threeten.bp.Year;
011import org.threeten.bp.YearMonth;
012import org.threeten.bp.ZoneId;
013import org.threeten.bp.ZoneOffset;
014import org.threeten.bp.ZonedDateTime;
015import org.threeten.bp.temporal.ChronoField;
016import org.threeten.bp.temporal.TemporalAccessor;
017import org.threeten.bp.temporal.TemporalQueries;
018
019/**
020 * Utility methods to work with {@link TemporalAccessor}
021 *
022 */
023public class TemporalAccessorUtils {
024
025  public static ZoneId UTC_ZONE_ID = ZoneOffset.UTC;
026
027  /**
028   * Transform a {@link TemporalAccessor} to a {@link java.util.Date}.
029   * If the provided {@link TemporalAccessor} contains offset(timezone) information it will be used.
030   * See {@link #toDate(TemporalAccessor, boolean)} for more details.
031   *
032   * @param temporalAccessor
033   * @return  the Date object or null if a Date object can not be created
034   */
035  public static Date toDate(TemporalAccessor temporalAccessor) {
036    return toDate(temporalAccessor, false);
037  }
038
039  /**
040   * Transform a {@link TemporalAccessor} to a {@link java.util.Date}.
041   *
042   * For {@link YearMonth}, the {@link java.util.Date} will represent the first day of the month.
043   * For {@link Year}, the {@link java.util.Date} will represent the first day of January.
044   *
045   * Remember that a {@link Date} object will always display the date in the current timezone.
046   *
047   * @param temporalAccessor
048   * @param ignoreOffset in case offset information is available in the provided {@link TemporalAccessor}, should it
049   *                     be used ?
050   * @return the Date object or null if a Date object can not be created
051   */
052  public static Date toDate(TemporalAccessor temporalAccessor, boolean ignoreOffset) {
053    if(temporalAccessor == null){
054      return null;
055    }
056
057    if(!ignoreOffset && temporalAccessor.isSupported(ChronoField.OFFSET_SECONDS)){
058      return DateTimeUtils.toDate(temporalAccessor.query(ZonedDateTime.FROM).toInstant());
059    }
060
061    if(temporalAccessor.isSupported(ChronoField.SECOND_OF_DAY)){
062      return DateTimeUtils.toDate(temporalAccessor.query(LocalDateTime.FROM).atZone(UTC_ZONE_ID).toInstant());
063    }
064
065    // this may return null in case of partial dates
066    LocalDate localDate = temporalAccessor.query(TemporalQueries.localDate());
067
068    // try YearMonth
069    if(localDate == null && temporalAccessor.isSupported(ChronoField.MONTH_OF_YEAR)) {
070      YearMonth yearMonth = YearMonth.from(temporalAccessor);
071      localDate = yearMonth.atDay(1);
072    }
073
074    // try Year
075    if(localDate == null && temporalAccessor.isSupported(ChronoField.YEAR)) {
076      Year year = Year.from(temporalAccessor);
077      localDate = year.atDay(1);
078    }
079
080    if (localDate != null) {
081      return DateTimeUtils.toDate(localDate.atStartOfDay(UTC_ZONE_ID).toInstant());
082    }
083
084    return null;
085  }
086
087  /**
088   * The idea of "best resolution" TemporalAccessor is to get the TemporalAccessor that offers more resolution than
089   * the other but they must NOT contradict.
090   * e.g. 2005-01 and 2005-01-01 will return 2005-01-01.
091   *
092   * Note that if one of the 2 parameters is null the other one will be considered having the best resolution
093   *
094   * @param ta1
095   * @param ta2
096   * @return never null
097   */
098  public static Optional<? extends TemporalAccessor> getBestResolutionTemporalAccessor(@Nullable TemporalAccessor ta1,
099                                                                                       @Nullable TemporalAccessor ta2){
100    //handle nulls combinations
101    if(ta1 == null && ta2 == null){
102      return Optional.absent();
103    }
104    if(ta1 == null){
105      return Optional.of(ta2);
106    }
107    if(ta2 == null){
108      return Optional.of(ta1);
109    }
110
111    AtomizedLocalDate ymd1 = AtomizedLocalDate.fromTemporalAccessor(ta1);
112    AtomizedLocalDate ymd2 = AtomizedLocalDate.fromTemporalAccessor(ta2);
113
114    // If they both provide the year, it must match
115    if(ymd1.getYear() != null && ymd2.getYear() != null && !ymd1.getYear().equals(ymd2.getYear())){
116      return Optional.absent();
117    }
118    // If they both provide the month, it must match
119    if(ymd1.getMonth() != null && ymd2.getMonth() != null && !ymd1.getMonth().equals(ymd2.getMonth())){
120      return Optional.absent();
121    }
122    // If they both provide the day, it must match
123    if(ymd1.getDay() != null && ymd2.getDay() != null && !ymd1.getDay().equals(ymd2.getDay())){
124      return Optional.absent();
125    }
126
127    if(ymd1.getResolution() > ymd2.getResolution()){
128      return Optional.of(ta1);
129    }
130
131    return Optional.of(ta2);
132  }
133
134}
