021/**
022 * The ThreeTenNumericalDateParserBuilder can build objects directly (build(..) methods) or return an instance
023 * of itself to create more complex object.
024 *
025 */
026public class ThreeTenNumericalDateParserBuilder {
027
028  //the letter 'u' in all the patterns refers to YEAR as opposed to 'y' who refers to YEAR_OF_ERA
029  private final static String YEAR_2_DIGITS_PATTERN_SUFFIX = "uu";
030  private final static String IS_YEAR_2_DIGITS_PATTERN = "^.+[^u]"+YEAR_2_DIGITS_PATTERN_SUFFIX+"$";
031
032  private ThreeTenNumericalDateParserBuilder(){}
033

035   * Get a new builder to create a list of ThreeTenDateTimeParser.

040    return new ThreeTenDateParserListBuilder();
041  }

051  /**
052   * Build a single ThreeTenDateTimeParser.
053   *
054   * @param pattern
055   * @param hint
056   * @return
057   */
058  public static ThreeTenDateTimeParser build(@NotNull String pattern, @NotNull DateFormatHint hint){
059    Preconditions.checkNotNull(pattern);
060    Preconditions.checkNotNull(hint);
061
062    int minLength = getMinimumStringLengthForPattern(pattern);
063    DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern(pattern).withResolverStyle(ResolverStyle.STRICT);
064    return new ThreeTenDateTimeParser(dateTimeFormatter, null, hint, minLength);
065  }
066
067  /**
068   * Build a single ThreeTenDateTimeParser with support for separator normalization.
069   *
070   * @param pattern
071   * @param hint
072   * @param separator
073   * @param alternativeSeparators
074   * @return
075   */
076  public static ThreeTenDateTimeParser build(String pattern, DateFormatHint hint, String separator,
077                                             String alternativeSeparators){
078    Preconditions.checkNotNull(pattern);
079    Preconditions.checkNotNull(hint);
080    Preconditions.checkArgument(StringUtils.isNotBlank(separator), "separator must NOT be blank");
081    Preconditions.checkArgument(StringUtils.isNotBlank(alternativeSeparators), "alternativeSeparators must NOT be blank");
082
083    DateTimeSeparatorNormalizer dateTimeNormalizer = new DateTimeSeparatorNormalizer(CharMatcher.anyOf(alternativeSeparators), separator);
084    DateTimeFormatter dateTimeFormatter = DateTimeFormatter.ofPattern(pattern).withResolverStyle(ResolverStyle.STRICT);
085    int minLength = getMinimumStringLengthForPattern(pattern);
086    return new ThreeTenDateTimeParser(dateTimeFormatter, dateTimeNormalizer, hint, minLength);
087  }
088
089  /**
090   *
091   * Build a single ThreeTenDateTimeParser from a baseYear.
092   *
093   *
094   * @param pattern pattern that includes a 2 digits year (-uu)
095   * @param hint
096   * @param baseYear
097   * @return
098   */
099  public static ThreeTenDateTimeParser build(String pattern, DateFormatHint hint, Year baseYear){
100    int minLength = getMinimumStringLengthForPattern(pattern);
101    DateTimeFormatter dateTimeFormatter = build2DigitsYearDateTimeFormatter(pattern, baseYear);
102    return new ThreeTenDateTimeParser(dateTimeFormatter, null, hint, minLength);
103  }
104
105  /**
106   *
107   * Build a single ThreeTenDateTimeParser from a baseYear with support for separator normalization.
108   *
109   * @param pattern pattern that includes a 2 digits year (-uu)
110   * @param hint
111   * @param separator
112   * @param alternativeSeparators
113   * @param baseYear
114   * @return
115   */
116  public static ThreeTenDateTimeParser build(String pattern, DateFormatHint hint, String separator,
117                                             String alternativeSeparators, Year baseYear){
118    Preconditions.checkNotNull(pattern);
119    Preconditions.checkNotNull(hint);
120    Preconditions.checkArgument(StringUtils.isNotBlank(separator), "separator must NOT be blank");
121    Preconditions.checkArgument(StringUtils.isNotBlank(alternativeSeparators), "alternativeSeparators must NOT be blank");
122
123    DateTimeSeparatorNormalizer dateTimeNormalizer = new DateTimeSeparatorNormalizer(CharMatcher.anyOf(alternativeSeparators), separator);
124    DateTimeFormatter dateTimeFormatter = build2DigitsYearDateTimeFormatter(pattern, baseYear);
125    int minLength = getMinimumStringLengthForPattern(pattern);
126    return new ThreeTenDateTimeParser(dateTimeFormatter, dateTimeNormalizer, hint, minLength);
127  }
128
129
130
131  /**
132   * From a {@link }DateTimeFormatter} pattern in String, get the minimum String length required for an input String to apply
133   * the pattern. This is used to quickly discard DateTimeFormatter simply based on String length of the input.
134   * Minimum length is the length of the pattern String minus the optional section(s) and quotes.
135   *
136   * @param pattern
137   * @return
138   */
139  private static int getMinimumStringLengthForPattern(String pattern){
140    pattern = ThreeTenNumericalDateParser.OPTIONAL_PATTERN_PART.matcher(pattern).replaceAll("").replaceAll("'", "");
141    return pattern.length();
142  }
143
144  private static DateTimeFormatter build2DigitsYearDateTimeFormatter(String pattern, Year baseYear){
145    Preconditions.checkState(pattern.matches(IS_YEAR_2_DIGITS_PATTERN) || pattern.equals(YEAR_2_DIGITS_PATTERN_SUFFIX),
146            "build2DigitsYearDateTimeFormatter can only be used for patterns with 2 digit year");
147    pattern = StringUtils.removeEnd(pattern, YEAR_2_DIGITS_PATTERN_SUFFIX);
148    return new DateTimeFormatterBuilder().append(DateTimeFormatter.ofPattern(pattern))
149            .appendValueReduced(ChronoField.YEAR, 2, 2, baseYear.getValue()).parseStrict().toFormatter();
150  }

152  /**
153   * Builder used to build a List of {@link ThreeTenDateTimeParser}.
154   */
155  public static class ThreeTenDateParserListBuilder {
156    private final List<ThreeTenDateTimeParser> dateTimeParsers = Lists.newArrayList();
157
158    public ThreeTenDateParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint){
159      dateTimeParsers.add(ThreeTenNumericalDateParserBuilder.build(pattern, hint));
160      return this;
161    }
162
163    /**
164     *
165     * @param pattern
166     * @param hint
167     * @param separator
168     * @param alternativeSeparators separator used in the pattern that should be used as replacement for alternativeSeparators
169     * @return
170     */
171    public ThreeTenDateParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint,
172                                                                   String separator, String alternativeSeparators){
173      dateTimeParsers.add(ThreeTenNumericalDateParserBuilder.build(pattern, hint, separator, alternativeSeparators));
174      return this;
175    }
176
177    public ThreeTenDateParserListBuilder append2DigitsYearDateTimeParser(String pattern, DateFormatHint hint,
178                                                                              Year baseYear){
179      int minLength = getMinimumStringLengthForPattern(pattern);
180      DateTimeFormatter dateTimeFormatter = build2DigitsYearDateTimeFormatter(pattern, baseYear);
181      dateTimeParsers.add(new ThreeTenDateTimeParser(dateTimeFormatter, null, hint, minLength));
182      return this;
183    }
184
185    public ThreeTenDateParserListBuilder append2DigitsYearDateTimeParser(String pattern, DateFormatHint hint,
186                                                                              String separator, String alternativeSeparators,
187                                                                              Year baseYear){
188      //get length before removing year part
189      int minLength = getMinimumStringLengthForPattern(pattern);
190      DateTimeSeparatorNormalizer dateTimeNormalizer = new DateTimeSeparatorNormalizer(CharMatcher.anyOf(alternativeSeparators), separator);
191      DateTimeFormatter dateTimeFormatter = build2DigitsYearDateTimeFormatter(pattern, baseYear);
192      dateTimeParsers.add(new ThreeTenDateTimeParser(dateTimeFormatter, dateTimeNormalizer, hint, minLength));
193      return this;
194    }
195
196    public List<ThreeTenDateTimeParser> build(){
197      return ImmutableList.copyOf(dateTimeParsers);
198    }
199  }

201  /**
202   * More specific builder Builder used to build a {@link ThreeTenDateTimeMultiParser}.
203   */
204  public static class ThreeTenDateMultiParserListBuilder {
205    private ThreeTenDateTimeParser preferred;
206    private List<ThreeTenDateTimeParser> otherParsers = Lists.newArrayList();
207
208    public ThreeTenDateMultiParserListBuilder preferredDateTimeParser(String pattern, DateFormatHint hint){
209      preferred = ThreeTenNumericalDateParserBuilder.build(pattern, hint);
210      return this;
211    }
212
213    public ThreeTenDateMultiParserListBuilder preferredDateTimeParser(String pattern, DateFormatHint hint, Year year){
214      preferred = ThreeTenNumericalDateParserBuilder.build(pattern, hint, year);
215      return this;
216    }
217
218    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint){
219      otherParsers.add(ThreeTenNumericalDateParserBuilder.build(pattern, hint));
220      return this;
221    }
222
223    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint, Year year){
224      otherParsers.add(ThreeTenNumericalDateParserBuilder.build(pattern, hint, year));
225      return this;
226    }
227
228    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint,
229                                                               String separator, String alternativeSeparators){
230      otherParsers.add(ThreeTenNumericalDateParserBuilder.build(pattern, hint, separator, alternativeSeparators));
231      return this;
232    }
233
234    public ThreeTenDateMultiParserListBuilder appendDateTimeParser(String pattern, DateFormatHint hint,
235                                                               String separator, String alternativeSeparators, Year year){
236      otherParsers.add(ThreeTenNumericalDateParserBuilder.build(pattern, hint, separator, alternativeSeparators, year));
237      return this;
238    }
239
240    /**
241     * Ensure the builder is used with content we expect.
242     * Currently (this could change) we should only have one ThreeTenDateTimeParser per DateFormatHint.
243     *
244     * @throws IllegalStateException
245     */
246    private void validate() throws IllegalStateException {
247      Set<DateFormatHint> hints = Sets.newHashSet();
248      if(preferred != null){
249        hints.add(preferred.getHint());
250      }
251
252      for(ThreeTenDateTimeParser parser : otherParsers){
253        if(!hints.add(parser.getHint())){
254          throw new IllegalStateException("DateFormatHint can only be used once in a ThreeTenDateTimeMultiParser." +
255                  "[" + parser.getHint() + "]");
256        }
257      }
258    }
259
260    public ThreeTenDateTimeMultiParser build() throws IllegalStateException {
261      validate();
262      return new ThreeTenDateTimeMultiParser(preferred, otherParsers);
263    }
264  }
