036/**
037 * Numerical DateParser based on threetenbp (JSR310 backport) library which also means it is almost ready for Java 8.
038 * This is a numerical Date Parser which means it is not responsible to parse dates that contains text for representing
039 * a part of the dates (e.g. January 1 1980)
040 *
041 * Months are in numerical value starting at 1 for January.
042 *
043 * Note that LocalDateTime and LocalDate are TimeZone agnostic.
044 *
045 * Be aware that LocalDate and LocalDateTime doesn't map correctly to Date object for all dates before the
046 * Gregorian cut-off date (1582-10-15). To transform a such date use GregorianCalendar by setting the date according
047 * to the TemporalAccessor you got back from that class.
048 *
049 * Thread-Safe after creation.
050 *
051 */
052public class ThreeTenNumericalDateParser implements Parsable<TemporalAccessor> {
053
054  private static final Logger LOGGER = LoggerFactory.getLogger(ThreeTenNumericalDateParser.class);
055
056  public static final Pattern OPTIONAL_PATTERN_PART = Pattern.compile("\\[.*\\]");
057
058  // ISO 8601 specifies a Unicode minus (CHAR_MINUS), with a hyphen (CHAR_HYPHEN) as an alternative.
059  public static final char CHAR_HYPHEN = '\u002d'; // Unicode hyphen, U+002d, char '-'
060  public static final char CHAR_MINUS = '\u2212'; // Unicode minus, U+2212, char '−'
061
062  private static final Map<DateFormatHint, List<ThreeTenDateTimeParser>> FORMATTERS_BY_HINT = Maps.newHashMap();
063
064  // DateTimeFormatter includes some ISO parsers but just to make it explicit we define our own
065  private static final DateTimeFormatter ISO_PARSER = (new DateTimeFormatterBuilder()
066          .appendValue(ChronoField.YEAR, 2, 4, SignStyle.NEVER)
067          .optionalStart().appendLiteral('-')
068          .appendValue(ChronoField.MONTH_OF_YEAR, 1, 2, SignStyle.NEVER)
069          .optionalStart().appendLiteral('-')
070          .appendValue(ChronoField.DAY_OF_MONTH, 1, 2, SignStyle.NEVER))
071          .optionalEnd()
072          .optionalEnd()
073          .toFormatter().withResolverStyle(ResolverStyle.STRICT);
074
075  //brackets [] represent optional section of the pattern
076  //separator is a CHAR_HYPHEN
077  private static final List<ThreeTenDateTimeParser> BASE_PARSER_LIST = ImmutableList.copyOf(
078          ThreeTenNumericalDateParserBuilder.newParserListBuilder()
079                  .appendDateTimeParser("uuuuMMdd", DateFormatHint.YMD)
080                  .appendDateTimeParser("uuuu-M-d[ HH:mm:ss]", DateFormatHint.YMDT, String.valueOf(CHAR_HYPHEN),
081                          String.valueOf(CHAR_MINUS) + ".")
082                  .appendDateTimeParser("uuuu-M-d'T'HH[:mm[:ss]]", DateFormatHint.YMDT)
083                  .appendDateTimeParser("uuuu-M-d'T'HHmm[ss]", DateFormatHint.YMDT)
084                  .appendDateTimeParser("uuuu-M-d'T'HH:mm:ssZ", DateFormatHint.YMDT)
085                  .appendDateTimeParser("uuuu-M-d'T'HH:mm:ssxxx", DateFormatHint.YMDT) //covers 1978-12-21T02:12:43+01:00
086                  .appendDateTimeParser("uuuu-M-d'T'HH:mm:ss'Z'", DateFormatHint.YMDT)
087                  .appendDateTimeParser("uuuu-M", DateFormatHint.YM)
088                  .appendDateTimeParser("uuuu", DateFormatHint.Y)
089                  .appendDateTimeParser("uuuu年MM月dd日", DateFormatHint.HAN)
090                  .appendDateTimeParser("uuuu年M月d日", DateFormatHint.HAN)
091                  .appendDateTimeParser("uu年M月d日", DateFormatHint.HAN)
092                  .build()
093  );

095  // Possibly ambiguous dates will record an error in case more than one pattern can be applied
096  private static final List<ThreeTenDateTimeMultiParser> MULTIPARSER_PARSER_LIST = ImmutableList.of(
097          ThreeTenNumericalDateParserBuilder.newMultiParserListBuilder()
098                  .preferredDateTimeParser("d.M.uuuu", DateFormatHint.DMY) //DE, DK, NO
099                  .appendDateTimeParser("M.d.uuuu", DateFormatHint.MDY)
100                  .build(),
101          // the followings are mostly derived of the difference between FR,GB,ES (DMY) format and US format (MDY)
102          ThreeTenNumericalDateParserBuilder.newMultiParserListBuilder()
103                  .appendDateTimeParser("d/M/uuuu", DateFormatHint.DMY, "/", String.valueOf(CHAR_HYPHEN) + String.valueOf(CHAR_MINUS))
104                  .appendDateTimeParser("M/d/uuuu", DateFormatHint.MDY, "/", String.valueOf(CHAR_HYPHEN) + String.valueOf(CHAR_MINUS))
105                  .build(),
106          ThreeTenNumericalDateParserBuilder.newMultiParserListBuilder()
107                  .appendDateTimeParser("ddMMuuuu", DateFormatHint.DMY)
108                  .appendDateTimeParser("MMdduuuu", DateFormatHint.MDY)
109                  .build(),
110          // the followings are not officially supported by any countries but are sometimes used
111          ThreeTenNumericalDateParserBuilder.newMultiParserListBuilder()
112                  .appendDateTimeParser("d\\M\\uuuu", DateFormatHint.DMY, "\\", "_")
113                  .appendDateTimeParser("M\\d\\uuuu", DateFormatHint.MDY, "\\", "_")
114                  .build()
115  );

128      for(ThreeTenDateTimeParser parser : parserAmbiguity.getAllParsers()) {
129        //TODO: when updated to Java 8 FORMATTERS_BY_HINT.putIfAbsent(parser.getHint(), new ArrayList<ThreeTenDateTimeParser>());

132        }
133        // end TODO
134        FORMATTERS_BY_HINT.get(parser.getHint()).add(parser);
135      }
136    }

170  private ThreeTenNumericalDateParser(Year baseYear) {
171
172    Preconditions.checkState(baseYear.getValue() <= LocalDate.now().getYear(), "Base year is less or equals to" +
173              " the current year");
174
175    Map<DateFormatHint, List<ThreeTenDateTimeParser>> formattersByHint = Maps.newHashMap(FORMATTERS_BY_HINT);
176
177    List<ThreeTenDateTimeMultiParser> multiParserList = Lists.newArrayList(MULTIPARSER_PARSER_LIST);
178    multiParserList.addAll(Lists.newArrayList(
179            ThreeTenNumericalDateParserBuilder.newMultiParserListBuilder()
180                    .preferredDateTimeParser("d.M.uu", DateFormatHint.DMY, baseYear) //DE, DK, NO
181                    .appendDateTimeParser("M.d.uu", DateFormatHint.MDY, baseYear)
182                    .build(),
183            ThreeTenNumericalDateParserBuilder.newMultiParserListBuilder()
184                    .appendDateTimeParser("d/M/uu", DateFormatHint.DMY, "/",
185                            String.valueOf(CHAR_HYPHEN) + String.valueOf(CHAR_MINUS), baseYear)
186                    .appendDateTimeParser("M/d/uu", DateFormatHint.MDY, "/",
187                            String.valueOf(CHAR_HYPHEN) + String.valueOf(CHAR_MINUS), baseYear)
188                    .build(),
189            ThreeTenNumericalDateParserBuilder.newMultiParserListBuilder()
190                    .appendDateTimeParser("ddMMuu", DateFormatHint.DMY, baseYear)
191                    .appendDateTimeParser("MMdduu", DateFormatHint.MDY, baseYear)
192                    .build(),
193            ThreeTenNumericalDateParserBuilder.newMultiParserListBuilder()
194                    .appendDateTimeParser("d\\M\\uu", DateFormatHint.DMY, "\\", "_", baseYear)
195                    .appendDateTimeParser("M\\d\\uu", DateFormatHint.MDY, "\\", "_", baseYear)
196                    .build()
197    ));
198
199    for(ThreeTenDateTimeMultiParser multiParser : multiParserList){
200      for(ThreeTenDateTimeParser parser : multiParser.getAllParsers()) {
201        //TODO: when updated to Java 8 formattersByHint.putIfAbsent(parser.getHint(), new ArrayList<ThreeTenDateTimeParser>());
202        if(!formattersByHint.containsKey(parser.getHint())){
203          formattersByHint.put(parser.getHint(), new ArrayList<ThreeTenDateTimeParser>());
204        }
205        // end TODO
206        formattersByHint.get(parser.getHint()).add(parser);
207      }
208    }
209
210    this.activeMultiParserList = ImmutableList.copyOf(multiParserList);
211    this.activeFormattersByHint = ImmutableMap.copyOf(formattersByHint);
212  }

214  @Override
215  public ParseResult<TemporalAccessor> parse(String input) {
216    return parse(input, DateFormatHint.NONE);
217  }

219  /**
220   * Parse year, month, day strings as a TemporalAccessor.
221   *
222   * @param year
223   * @param month
224   * @param day
225   * @return
226   */
227  public static ParseResult<TemporalAccessor> parse(@Nullable String year, @Nullable String month, @Nullable String day) {
228
229    // avoid possible misinterpretation when month is not provided (but day is)
230    if(StringUtils.isBlank(month) && StringUtils.isNotBlank(day)){
231      return ParseResult.fail();
232    }
233
234    String date = Joiner.on(CHAR_HYPHEN).skipNulls().join(Strings.emptyToNull(year), Strings.emptyToNull(month),
235            Strings.emptyToNull(day));
236    TemporalAccessor tp = tryParse(date, ISO_PARSER, null);
237
238    if(tp != null){
239      return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, tp);
240    }
241    return ParseResult.fail();
242  }
243
244  /**
245   * Parse year, month, day integers as a TemporalAccessor.
246   *
247   * @param year
248   * @param month
249   * @param day
250   * @return
251   */
252  public static ParseResult<TemporalAccessor> parse(@Nullable Integer year, @Nullable Integer month, @Nullable Integer day) {
253
254    // avoid possible misinterpretation when month is not provided (but day is)
255    if(month == null && day != null){
256      return ParseResult.fail();
257    }
258
259    String date = Joiner.on(CHAR_HYPHEN).skipNulls().join(year, month, day);
260    TemporalAccessor tp = tryParse(date, ISO_PARSER, null);
261
262    if(tp != null){
263      return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, tp);
264    }
265    return ParseResult.fail();
266  }

268  /**
269   * Parse a date represented as a single String into a TemporalAccessor.
270   *
271   * @param input
272   * @param hint help to speed up the parsing and possibly return a better confidence
273   * @return
274   */
275  public ParseResult<TemporalAccessor> parse(String input, DateFormatHint hint) {
276
277    if(StringUtils.isBlank(input)){
278      return ParseResult.fail();
279    }
280    // make sure hint is never null
281    if(hint == null){
282      hint = DateFormatHint.NONE;
283    }
284
285    List<ThreeTenDateTimeParser> parserList = activeFormattersByHint.containsKey(hint) ? activeFormattersByHint.get(hint) : BASE_PARSER_LIST;
286
287    // First attempt: find a match with definite confidence
288    TemporalAccessor parsedTemporalAccessor;
289    for(ThreeTenDateTimeParser parser : parserList){
290      parsedTemporalAccessor = parser.parse(input);
291      if(parsedTemporalAccessor != null){
292        return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, parsedTemporalAccessor);
293      }
294    }
295
296    // if a format hint was provided we already tried all possible format
297    if( hint != DateFormatHint.NONE){
298      return ParseResult.fail();
299    }
300
301    // Second attempt: find one or multiple match(es) in the list of ThreeTenDateTimeMultiParser
302    int numberOfPossiblyAmbiguousMatch = 0;
303    TemporalAccessor lastParsedSuccess = null;
304    TemporalAccessor lastParsedPreferred = null;
305    // Is that all results are equal (the represent the same TemporalAccessor) in case there is no preferred result defined
306    boolean lastParsedSuccessOtherResultsEqual = false;
307    ThreeTenDateTimeMultiParser.MultipleParseResult result;
308
309    // here we do not stop when we find a match, we try them all to check for a possible ambiguity
310    for(ThreeTenDateTimeMultiParser parserAmbiguity : activeMultiParserList){
311      result = parserAmbiguity.parse(input);
312      numberOfPossiblyAmbiguousMatch += result.getNumberParsed();
313
314      if(result.getNumberParsed() > 0){
315        lastParsedSuccess = result.getResult();
316
317        // make sure to log in case lastParsedSuccessEqual already equals true
318        if (lastParsedSuccessOtherResultsEqual) {
319          LOGGER.warn("Issue with ThreeTenDateTimeMultiParser configuration: Input {} produces more results even " +
320                  "if lastParsedSuccessEqual is set to true.", input);
321        }
322        lastParsedSuccessOtherResultsEqual = false;
323
324        if(result.getPreferredResult() != null){
325          if(lastParsedPreferred != null){
326            LOGGER.warn("Issue with ThreeTenDateTimeMultiParser configuration: Input {} produces 2 preferred results", input);
327          }
328          lastParsedPreferred = result.getPreferredResult();
329        }
330        else{
331          //if we have no PreferredResult but all results represent the same TemporalAccessor
332          if(result.getOtherResults() != null && result.getOtherResults().size() > 1) {
333            lastParsedSuccessOtherResultsEqual = allEquals(result.getOtherResults());
334          }
335        }
336      }
337    }
338
339    //if there is only one pattern that can be applied it is not ambiguous
340    if(numberOfPossiblyAmbiguousMatch == 1){
341      return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, lastParsedSuccess);
342    }
343    else if ( numberOfPossiblyAmbiguousMatch > 1 ){
344
345      //if all the possible ambiguities are equal, there is no ambiguity
346      if(lastParsedSuccessOtherResultsEqual){
347        return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, lastParsedSuccess);
348      }
349
350      //check if we have result from the preferred parser
351      if(lastParsedPreferred != null){
352        return ParseResult.success(ParseResult.CONFIDENCE.PROBABLE, lastParsedPreferred);
353      }
354    }
355
356    LOGGER.debug("Number of matches for {} : {}", input, numberOfPossiblyAmbiguousMatch);
357    return ParseResult.fail();
358  }

360  /**
361   * Utility private method to avoid throwing a runtime exception when the formatter can not parse the String.
362   * TODO normalizer is call too often maybe this class should not take it and only try to parse
363   * @param input
364   * @param formatter
365   * @param normalizer
366   * @return
367   */
368  private static TemporalAccessor tryParse(String input, DateTimeFormatter formatter, DateTimeSeparatorNormalizer normalizer){
369
370    if(normalizer != null){
371      input = normalizer.normalize(input);
372    }
373    try {
374      return formatter.parseBest(input, ZonedDateTime.FROM, LocalDateTime.FROM, LocalDate.FROM, YearMonth.FROM, Year.FROM);
375    }
376    catch (DateTimeParseException dpe){}
377    return null;
378  }

380  /**
381   * Check if all the TemporalAccessor of the list are equal.
382   * If the list contains 0 element is will return false, if the list contains 1 element it will return true.
383   * @param taList
384   * @return
385   */
386  private static boolean allEquals(List<TemporalAccessor> taList){
387
388    if(taList == null || taList.isEmpty()){
389      return false;
390    }
391
392    TemporalAccessor reference = null;
393    boolean allEqual = false;
394    for(TemporalAccessor ta: taList){
395      if(reference == null){
396        reference = ta;
397      }
398      allEqual = reference.equals(ta);
399    }
400    return allEqual;
401  }
