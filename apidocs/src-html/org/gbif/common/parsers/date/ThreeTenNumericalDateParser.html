027 * WORK IN PROGRESS
028 *
029 * Numerical DateParser based on threetenbp (JSR310 backport) library which also means it is almost ready for Java 8.
030 * This is a numerical Date Parser which means it is not responsible to parse dates that contains text for representing
031 * a part of the dates (e.g. January 1 1980)

039  //Should be mutually exclusive
040  //Hints are given to the parser to help to select the right sets of DateTimeFormatter
041  public enum DateParsingHint {YMDT, YMD, DMY, MDY, YM, Y, ASIAN, NONE};
042
043  private final static Map<DateParsingHint, List<InternalDateTimeParser>> FORMATTERS_HINT = Maps.newHashMap();

045  //the letter 'u' in all the patterns refers to YEAR as opposed to 'y' who refers to YEAR_OF_ERA

047  private final static String IS_YEAR_2_DIGITS_PATTERN = "^.+[^u]"+YEAR_2_DIGITS_PATTERN_SUFFIX+"$";

100  private static final List<InternalDateTimeParser> DATE_FORMATTERS;
101
102  static{
103    DATE_FORMATTERS = Lists.newArrayList();
104
105    for(InternalDateTimeParser parser : PATTERNS){
106      DATE_FORMATTERS.add(parser);
107      //keep a reference to the formatter by mapping it with his hint key
108      if(!FORMATTERS_HINT.containsKey(parser.getHint())){
109        FORMATTERS_HINT.put(parser.getHint(), new ArrayList<InternalDateTimeParser>());
110      }
111      FORMATTERS_HINT.get(parser.getHint()).add(parser);
112    }
113  }
114
115  private static InternalDateTimeParser buildDateTimeParser(String pattern, DateParsingHint hint){
116    return InternalDateTimeParser.of(DateTimeFormatter.ofPattern(pattern).withResolverStyle(ResolverStyle.STRICT),
117            pattern, hint);
118  }
119
120  private static InternalDateTimeParser buildPossiblyAmbiguousDateTimeParser(String pattern, DateParsingHint hint){
121    DateTimeFormatter dateTimeFormatter = null;
122
123    if(pattern.matches(IS_YEAR_2_DIGITS_PATTERN) || pattern.equals(YEAR_2_DIGITS_PATTERN_SUFFIX)){
124      dateTimeFormatter = buildWith2DigitYear(StringUtils.removeEnd(pattern, YEAR_2_DIGITS_PATTERN_SUFFIX));
125    }
126    else{
127      dateTimeFormatter = DateTimeFormatter.ofPattern(pattern).withResolverStyle(ResolverStyle.STRICT);
128    }
129    return InternalDateTimeParser.ofPossiblyAmbiguous(dateTimeFormatter, pattern, hint);
130  }

132
133  /**
134   * By default in JSR310 and Java 8, dates with 2 digits are 2000 based.
135   * This method will APPEND a 2 digit year in the pattern.
136   *
137   * @param pattern must NOT include the year in the pattern (e.g. MM-dd- will create a formatter for MM-dd-uu)
138   * @return
139   */
140  private static DateTimeFormatter buildWith2DigitYear(String pattern){
141    return new DateTimeFormatterBuilder().append(DateTimeFormatter.ofPattern(pattern)).appendValueReduced(
142                    ChronoField.YEAR, 2, 2, Year.now().getValue() - 80).parseStrict().toFormatter();
143  }

145  @Override
146  public ParseResult<TemporalAccessor> parse(String input) {
147    return parse(input, DateParsingHint.NONE);
148  }
149
150  /**
151   *
152   * @param input
153   * @param hint
154   * @return
155   */
156  public static ParseResult<TemporalAccessor> parse(String input, DateParsingHint hint) {
157
158    if(StringUtils.isBlank(input)){
159      return ParseResult.error();
160    }
161
162    //if there is no separator we can ignore all InternalDateTimeParser that use separator
163    boolean usesDelimiter = !StringUtils.isNumeric(input);
164
165    List<InternalDateTimeParser> parserList = FORMATTERS_HINT.containsKey(hint) ? FORMATTERS_HINT.get(hint) : DATE_FORMATTERS;
166
167    int numberOfPossiblyAmbiguousMatch = 0;
168    TemporalAccessor lastParsedTemporalAccessorSuccess = null;
169    TemporalAccessor parsedTemporalAccessor = null;
170    for(InternalDateTimeParser parser : parserList){
171      if(usesDelimiter || !usesDelimiter && !parser.isWithDelimiters()){
172        parsedTemporalAccessor = tryParse(input, parser.getFormatter());
173        if(parsedTemporalAccessor != null){
174          if(parser.isPossiblyAmbiguous()){
175            numberOfPossiblyAmbiguousMatch++;
176            lastParsedTemporalAccessorSuccess = parsedTemporalAccessor;
177          }
178          else {
179            return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, parsedTemporalAccessor);
180          }
181        }
182      }
183    }
184
185    //if there is only one pattern that can be applied it is not ambiguous
186    if(numberOfPossiblyAmbiguousMatch == 1){
187      return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, lastParsedTemporalAccessorSuccess);
188    }
189
190    LOGGER.debug("Number of matches for {} : {}", input, numberOfPossiblyAmbiguousMatch);
191
192    return ParseResult.error();
193  }
194
195  /**
196   * Utility private method to avoid throwing a runtime exception when the formatter can not parse the String.
197   *
198   * @param input
199   * @param formatter
200   * @return
201   */
202  private static TemporalAccessor tryParse(String input, DateTimeFormatter formatter){
203    try {
204      return formatter.parseBest(input, LocalDateTime.FROM, LocalDate.FROM, YearMonth.FROM, Year.FROM);
205    }
206    catch (DateTimeParseException dpe){}
207    return null;
208  }

211  private static class InternalDateTimeParser {
212
213    private DateTimeFormatter formatter;
214    private String pattern;
215    private DateParsingHint hint;
216
217    private boolean possiblyAmbiguous;
218    private boolean withDelimiters;
219
220    /**
221     *
222     * @param formatter
223     * @param pattern
224     * @param hint
225     */
226    private InternalDateTimeParser(DateTimeFormatter formatter, String pattern, DateParsingHint hint){
227      this(formatter, pattern,  hint, false);
228    }
229
230    private InternalDateTimeParser(DateTimeFormatter formatter, String pattern, DateParsingHint hint, boolean possiblyAmbiguous){
231      this.formatter = formatter;
232      this.pattern = pattern;
233      this.hint = hint;
234      this.possiblyAmbiguous = possiblyAmbiguous;
235
236      this.withDelimiters = !StringUtils.isAlpha(pattern) || hint == DateParsingHint.ASIAN;
237    }
238
239    public static InternalDateTimeParser of(DateTimeFormatter formatter, String pattern, DateParsingHint hint){
240      return new InternalDateTimeParser(formatter, pattern, hint);
241    }
242
243    public static InternalDateTimeParser ofPossiblyAmbiguous(DateTimeFormatter formatter, String pattern, DateParsingHint hint){
244      return new InternalDateTimeParser(formatter, pattern, hint, true);
245    }
246
247    public DateTimeFormatter getFormatter(){
248      return formatter;
249    }
250
251    public String getPattern() {
252      return pattern;
253    }
254
255    public DateParsingHint getHint() {
256      return hint;
257    }
258
259    public boolean isPossiblyAmbiguous() {
260      return possiblyAmbiguous;
261    }
262
263    public boolean isWithDelimiters() {
264      return withDelimiters;
265    }
266
267  }
