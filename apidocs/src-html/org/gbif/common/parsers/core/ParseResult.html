001package org.gbif.common.parsers.core;
002
003import org.gbif.api.vocabulary.OccurrenceIssue;
004
005import java.util.Collection;
006import java.util.EnumSet;
007import java.util.Set;
008
009import com.google.common.base.Preconditions;
010import com.google.common.base.Predicates;
011import com.google.common.collect.Iterables;
012import com.google.common.collect.Lists;
013import org.apache.commons.lang3.builder.ToStringBuilder;
014import org.apache.commons.lang3.builder.ToStringStyle;
015
016/**
017 * This represents the response of a generic parse operation.
018 */
019public class ParseResult<T> {
020
021  // the result code of the parse operation
022  public enum STATUS {
023    SUCCESS, FAIL, ERROR
024  }
025
026  // the confidence of the result, applicable if status=SUCCESS
027  public enum CONFIDENCE {
028    DEFINITE, PROBABLE, POSSIBLE
029  }
030
031  // the details of the response
032  protected STATUS status;
033  protected CONFIDENCE confidence;
034  protected T payload;
035  protected Throwable error;
036  private final Set<OccurrenceIssue> issues = EnumSet.noneOf(OccurrenceIssue.class);
037
038  /**
039   * @param <T1>       The generic type of the payload
040   * @param confidence The confidence in the result
041   * @param payload    The payload of the parse result
042   *
043   * @return The new ParseResult which has no error and status of SUCCESS
044   */
045  public static <T1> ParseResult<T1> success(CONFIDENCE confidence, T1 payload) {
046    return new ParseResult<T1>(STATUS.SUCCESS, confidence, payload, null, null);
047  }
048
049  public static <T1> ParseResult<T1> success(CONFIDENCE confidence, T1 payload, OccurrenceIssue issue) {
050    return new ParseResult<T1>(STATUS.SUCCESS, confidence, payload, null, Lists.newArrayList(issue));
051  }
052
053  public static <T1> ParseResult<T1> success(CONFIDENCE confidence, T1 payload, Collection<OccurrenceIssue> issues) {
054    return new ParseResult<T1>(STATUS.SUCCESS, confidence, payload, null, issues);
055  }
056  /**
057   * @return A new parse response with only the status set to FAIL
058   */
059  public static <T1> ParseResult<T1> fail() {
060    return new ParseResult<T1>(STATUS.FAIL, null, null, null, null);
061  }
062
063  public static <T1> ParseResult<T1> fail(OccurrenceIssue issue) {
064    return new ParseResult<T1>(STATUS.FAIL, null, null, null, Lists.newArrayList(issue));
065  }
066
067  public static <T1> ParseResult<T1> fail(Collection<OccurrenceIssue> issues) {
068    return new ParseResult<T1>(STATUS.FAIL, null, null, null, issues);
069  }
070
071  /**
072   * This creates a ParseResult indicating a parse failure but it also has a payload. Depending on the context this
073   * may provide additional information about the failure.
074   *
075   * @param payload the payload of the parse result
076   * @param <T1>    the generic type of the payload
077   *
078   * @return the new parse response which has a status of FAIL and an additional payload.
079   */
080  public static <T1> ParseResult<T1> fail(T1 payload, OccurrenceIssue issue) {
081    return new ParseResult<T1>(STATUS.FAIL, null, payload, null, Lists.newArrayList(issue));
082  }
083
084  /**
085   * @return A new parse response configured to indicate an error
086   */
087  public static <T1> ParseResult<T1> error() {
088    return new ParseResult<T1>(STATUS.ERROR, null, null, null, null);
089  }
090
091  /**
092   * @param cause The cause of the error
093   *
094   * @return A new parse response configured with error and the cause
095   */
096  public static <T1> ParseResult<T1> error(Throwable cause) {
097    return new ParseResult<T1>(STATUS.ERROR, null, null, cause, null);
098  }
099
100  /**
101   * Forces all fields to be provided
102   *
103   * @param status     The status of the response
104   * @param confidence The confidence in the result
105   * @param payload    The payload of the response
106   * @param error      The error in the response
107   * @param issues     Optional issues to report
108   */
109  public ParseResult(STATUS status, CONFIDENCE confidence, T payload, Throwable error, Collection<OccurrenceIssue> issues) {
110    this.status = status;
111    this.confidence = confidence;
112    this.payload = payload;
113    this.error = error;
114    if (issues != null) {
115      // add non nulls only
116      Iterables.addAll(this.issues, Iterables.filter(issues, Predicates.notNull()));
117    }
118  }
119
120  public STATUS getStatus() {
121    return status;
122  }
123
124  public CONFIDENCE getConfidence() {
125    return confidence;
126  }
127
128  public T getPayload() {
129    return payload;
130  }
131
132  public Throwable getError() {
133    return error;
134  }
135
136  public Set<OccurrenceIssue> getIssues() {
137    return issues;
138  }
139
140  public void addIssue(OccurrenceIssue issue) {
141    Preconditions.checkNotNull(issue);
142    issues.add(issue);
143  }
144
145  /**
146   * Returns true if {@link #getStatus()} returns SUCCESS.
147   *
148   * @return true if the parse operation was successful.
149   */
150  public boolean isSuccessful() {
151    return status == STATUS.SUCCESS;
152  }
153
154  @Override
155  public String toString() {
156    return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE)
157      .append("status", getStatus())
158      .append("confidence", getConfidence())
159      .append("payload", getPayload())
160      .append("issues", getIssues())
161      .append("error", getError()).toString();
162  }
163}
