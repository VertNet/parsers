001package org.gbif.common.parsers.core;
002
003import java.io.BufferedReader;
004import java.io.IOException;
005import java.io.InputStream;
006import java.io.InputStreamReader;
007import java.util.Iterator;
008import java.util.NoSuchElementException;
009import java.util.regex.Pattern;
010
011import com.google.common.base.Charsets;
012import org.apache.commons.lang3.StringUtils;
013
014/**
015 * A very simple Dictionary backed by a tab delimited file.
016 */
017public abstract class FileBasedDictionaryParser<T> extends DictionaryBackedParser<T> {
018
019  public FileBasedDictionaryParser(boolean caseSensitive) {
020    super(caseSensitive);
021  }
022
023  protected void init(InputStream input) {
024    init(new Source(input));
025  }
026
027  protected abstract T fromDictFile(String value);
028
029  /**
030   * An iterator over a well formed tab file.
031   * Should the file be poorly formed expect runtime exceptions.
032   */
033  class Source implements Iterator<KeyValue<String, T>> {
034
035    private final BufferedReader r;
036    private final Pattern tab = Pattern.compile("\t");
037    private String line = null;
038
039    Source(InputStream file) {
040      r = new BufferedReader(new InputStreamReader(file, Charsets.UTF_8));
041    }
042
043    @Override
044    public boolean hasNext() {
045      if (line != null) {
046        return true;
047      }
048
049      try {
050        // we discard bad lines
051        boolean valid = false;
052        do {
053          line = r.readLine();
054          if (line != null) {
055            valid = tab.split(line).length == 2;
056          }
057        } while (!valid && line != null);
058      } catch (IOException ignored) {
059        close();
060        return false;
061      }
062
063      if (line == null) {
064        close();
065        return false;
066      } else {
067        return true;
068      }
069    }
070
071    @Override
072    public KeyValue<String, T> next() {
073      if (!hasNext()) {
074        throw new NoSuchElementException();
075      }
076
077      String[] atoms = tab.split(line);
078      line = null;
079      return new KeyValue<String, T>(StringUtils.trimToNull(atoms[0]), fromDictFile(StringUtils.trimToNull(atoms[1])));
080    }
081
082    @Override
083    public void remove() {
084    }
085
086    public void close() {
087      if (r != null) {
088        try {
089          r.close();
090        } catch (IOException ignored) {
091        }
092      }
093    }
094  }
095}
