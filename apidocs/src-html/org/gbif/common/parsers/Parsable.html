001package org.gbif.common.parsers;
002
003/**
004 * Generic interface to allow multiple parser implementations.
005 *
006 * @param <TIN>  The input type of the parse operation
007 * @param <TOUT> The output type of the parse operation
008 */
009public interface Parsable<TIN, TOUT> {
010
011  /**
012   * Tries to parse the input and returns a {@link ParseResult} object.
013   * This should never return <code>null</code> as parse errors will be indicated in the returned object.
014   *
015   * @param input To parse
016   *
017   * @return The output result of the operation
018   */
019  ParseResult<TOUT> parse(TIN input);
020}
