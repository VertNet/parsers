001package org.gbif.common.parsers;
002
003import org.gbif.api.model.common.InterpretedEnum;
004import org.gbif.common.parsers.ParseResult.CONFIDENCE;
005import org.gbif.common.parsers.ParseResult.STATUS;
006
007/**
008 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/rank.txt.
009 */
010public abstract class InterpretedEnumParser<T extends Enum<T>> implements Parsable<String, InterpretedEnum<String, T>> {
011
012  private final Parsable<String, String> parser;
013  private final Class<T> clazz;
014
015  protected InterpretedEnumParser(Class<T> clazz, Parsable<String, String> parser) {
016    this.parser = parser;
017    this.clazz = clazz;
018  }
019
020  protected T toEnum(String parsedValue) {
021    return T.valueOf(clazz, parsedValue);
022  }
023
024  @Override
025  public ParseResult<InterpretedEnum<String, T>> parse(String input) {
026    ParseResult<String> preferred = parser.parse(input);
027    if (preferred.isSuccessful()) {
028      try {
029        InterpretedEnum<String, T> result = new InterpretedEnum<String, T>(input, toEnum(preferred.getPayload()));
030        return new ParseResult<InterpretedEnum<String, T>>(STATUS.SUCCESS, preferred.getConfidence(), result, null);
031      } catch (Exception e) {
032        return new ParseResult<InterpretedEnum<String, T>>(STATUS.FAIL, CONFIDENCE.DEFINITE, null, e);
033      }
034    }
035
036    return new ParseResult<InterpretedEnum<String, T>>(STATUS.FAIL, CONFIDENCE.DEFINITE, null, null);
037  }
038}
