001package org.gbif.common.parsers.utils;
002
003import org.gbif.api.vocabulary.Rank;
004
005import java.util.HashMap;
006import java.util.Map;
007
008public class RankUtils {
009
010  private static final Map<Rank, PortalTaxonomyRank> CLB_TO_PORTAL = new HashMap<Rank, PortalTaxonomyRank>();
011
012  private RankUtils() {
013    throw new UnsupportedOperationException("Can't initialize class");
014  }
015
016  static {
017    CLB_TO_PORTAL.put(Rank.KINGDOM, PortalTaxonomyRank.KINGDOM);
018    CLB_TO_PORTAL.put(Rank.SUBKINGDOM, PortalTaxonomyRank.SUBKINGDOM);
019    CLB_TO_PORTAL.put(Rank.SUPERPHYLUM, PortalTaxonomyRank.SUPERPHYLUM);
020    CLB_TO_PORTAL.put(Rank.PHYLUM, PortalTaxonomyRank.PHYLUM);
021    CLB_TO_PORTAL.put(Rank.SUBPHYLUM, PortalTaxonomyRank.SUBPHYLUM);
022    CLB_TO_PORTAL.put(Rank.SUPERCLASS, PortalTaxonomyRank.SUPERCLASS);
023    CLB_TO_PORTAL.put(Rank.CLASS, PortalTaxonomyRank.CLASS);
024    CLB_TO_PORTAL.put(Rank.SUBCLASS, PortalTaxonomyRank.SUBCLASS);
025    CLB_TO_PORTAL.put(Rank.SUPERORDER, PortalTaxonomyRank.SUPERORDER);
026    CLB_TO_PORTAL.put(Rank.ORDER, PortalTaxonomyRank.ORDER);
027    CLB_TO_PORTAL.put(Rank.SUBORDER, PortalTaxonomyRank.SUBORDER);
028    CLB_TO_PORTAL.put(Rank.INFRAORDER, PortalTaxonomyRank.INFRAORDER);
029    CLB_TO_PORTAL.put(Rank.SUPERFAMILY, PortalTaxonomyRank.SUPERFAMILY);
030    CLB_TO_PORTAL.put(Rank.FAMILY, PortalTaxonomyRank.FAMILY);
031    CLB_TO_PORTAL.put(Rank.SUBFAMILY, PortalTaxonomyRank.SUBFAMILY);
032    CLB_TO_PORTAL.put(Rank.TRIBE, PortalTaxonomyRank.TRIBE);
033    CLB_TO_PORTAL.put(Rank.SUBTRIBE, PortalTaxonomyRank.SUBTRIBE);
034    CLB_TO_PORTAL.put(Rank.GENUS, PortalTaxonomyRank.GENUS);
035    CLB_TO_PORTAL.put(Rank.SUBGENUS, PortalTaxonomyRank.SUBGENUS);
036    CLB_TO_PORTAL.put(Rank.SECTION, PortalTaxonomyRank.SECTION);
037    CLB_TO_PORTAL.put(Rank.SUBSECTION, PortalTaxonomyRank.SUBSECTION);
038    CLB_TO_PORTAL.put(Rank.SERIES, PortalTaxonomyRank.SERIES);
039    CLB_TO_PORTAL.put(Rank.SUBSERIES, PortalTaxonomyRank.SUBSERIES);
040    CLB_TO_PORTAL.put(Rank.SPECIES, PortalTaxonomyRank.SPECIES);
041    CLB_TO_PORTAL.put(Rank.INFRASPECIFIC_NAME, PortalTaxonomyRank.INFRASPECIFIC);
042    CLB_TO_PORTAL.put(Rank.SUBSPECIES, PortalTaxonomyRank.SUBSPECIES);
043    CLB_TO_PORTAL.put(Rank.VARIETY, PortalTaxonomyRank.VARIETY);
044    CLB_TO_PORTAL.put(Rank.FORM, PortalTaxonomyRank.FORM);
045    CLB_TO_PORTAL.put(Rank.CULTIVAR, PortalTaxonomyRank.CULTIVAR);
046    CLB_TO_PORTAL.put(Rank.UNRANKED, PortalTaxonomyRank.UNKNOWN);
047  }
048
049  /**
050   * Returns the Portal Rank Enumeration for the given CLB Rank.
051   * A few are not mapped: Subvariety, Strain, Suprageneric, etc.
052   *
053   * @param clbRank the CLB Rank to lookup
054   *
055   * @return the corresponding Portal rank or null if it couldn't be found
056   */
057  public static PortalTaxonomyRank getPortalRank(Rank clbRank) {
058    return CLB_TO_PORTAL.get(clbRank);
059  }
060
061}
