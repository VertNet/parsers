001package org.gbif.common.parsers.utils;
002
003import java.util.UUID;
004
005/**
006 * Utilities to parse an external dataset synthesized key.
007 * @deprecated please use {@link org.gbif.api.util.DatasetKey} instead.
008 */
009@Deprecated
010public class ExternalDatasetKeyUtils {
011
012  private static final int SOURCE_KEY_INDEX = 0;
013  private static final int DATASET_ID_INDEX = 1;
014  private static final int UUID_LENGTH = 36;
015  private static final int SYNTHESIZED_KEY_SEPARATOR_LENGTH = 1;
016  private static final int SYNTHESIZED_KEY_SEPARATOR_START_INDEX = 36;
017  private static final int SYNTHESIZED_KEY_DATASET_ID_START_INDEX = 37;
018
019  /**
020   * Parses the synthesized Dataset key into atoms or returns null if not parsable.
021   * The key is in the format: sourceKey:datasetId where the sourceKey must be a valid UUID. A UUID always has a length
022   * of 36 characters (32 digits, 5 groups, and 4 hyphens) like 550e8533-e29b-41d4-a716-446655441111 for example. The
023   * datasetId must just be a non empty string.
024   *
025   * @param key synthesized Dataset key
026   *
027   * @return array of strings, first entry is the sourceKey, second is the datasetId (both unvalidated), or an empty
028   *         array if the key was null or its length did not qualify as a valid external dataset key
029   */
030  private static String[] parseIntoAtoms(String key) {
031    if (key != null && key.length() > UUID_LENGTH + SYNTHESIZED_KEY_SEPARATOR_LENGTH) {
032      return new String[] {key.substring(0, SYNTHESIZED_KEY_SEPARATOR_START_INDEX),
033        key.substring(SYNTHESIZED_KEY_DATASET_ID_START_INDEX)};
034    }
035    return new String[] {};
036  }
037
038  /**
039   * Parses the synthesized Dataset key's sourceKey or returns null if not parsable.
040   * The sourceKey must be a valid UUID.
041   *
042   * @param key synthesized Dataset key
043   *
044   * @return sourceKey or null if not parsable or if it was not a valid UUID
045   */
046  public static UUID parseSourceKey(String key) {
047    String[] atoms = parseIntoAtoms(key);
048    try {
049      return UUID.fromString(atoms[SOURCE_KEY_INDEX]);
050    } catch (Exception e) {
051      return null; // nonsense provided
052    }
053  }
054
055  /**
056   * Parses the synthesized Dataset key's datasetId or returns null if not parsable.
057   * The datasetId is a String of any non empty string. Additionally, the sourceKey must be a valid UUID.
058   *
059   * @param key synthesized Dataset key
060   *
061   * @return datasetID or null if not parsable or sourceKey was not a valid UUID
062   */
063  public static String parseDatasetId(String key) {
064    String[] atoms = parseIntoAtoms(key);
065    try {
066      UUID.fromString(atoms[SOURCE_KEY_INDEX]);
067      return atoms[DATASET_ID_INDEX].isEmpty() ? null : atoms[DATASET_ID_INDEX];
068    } catch (Exception e) {
069      return null; // nonsense provided
070    }
071  }
072}
