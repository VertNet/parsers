001package org.gbif.common.parsers;
002
003import java.io.BufferedReader;
004import java.io.IOException;
005import java.io.InputStream;
006import java.io.InputStreamReader;
007import java.util.Iterator;
008import java.util.NoSuchElementException;
009import java.util.regex.Pattern;
010
011import com.google.common.base.Charsets;
012import org.apache.commons.lang3.StringUtils;
013
014/**
015 * A very simple Dictionary backed by a tab delimited file.
016 */
017public class FileBasedDictionaryParser extends DictionaryBackedParser<String, String> {
018
019  public FileBasedDictionaryParser(boolean caseSensitive, final InputStream... inputs) {
020    super(caseSensitive);
021    for (InputStream input : inputs) {
022      init(new Source(input));
023    }
024  }
025
026
027  /**
028   * An iterator over a well formed tab file. Should the file be poorly formed expect runtime exceptions.
029   *
030   * @author tim
031   */
032  static class Source implements Iterator<KeyValue<String, String>> {
033
034    private final BufferedReader r;
035    private final Pattern tab = Pattern.compile("\t");
036    private String line = null;
037
038    Source(InputStream file) {
039      r = new BufferedReader(new InputStreamReader(file, Charsets.UTF_8));
040    }
041
042    @Override
043    public boolean hasNext() {
044      if (line != null) {
045        return true;
046      }
047
048      try {
049        // we discard bad lines
050        boolean valid = false;
051        do {
052          line = r.readLine();
053          if (line != null) {
054            valid = tab.split(line).length == 2;
055          }
056        } while (!valid && line != null);
057      } catch (IOException ignored) {
058        close();
059        return false;
060      }
061
062      if (line == null) {
063        close();
064        return false;
065      } else {
066        return true;
067      }
068    }
069
070    @Override
071    public KeyValue<String, String> next() {
072      if (!hasNext()) {
073        throw new NoSuchElementException();
074      }
075
076      String[] atoms = tab.split(line);
077      line = null;
078      return new KeyValue<String, String>(StringUtils.trimToNull(atoms[0]), StringUtils.trimToNull(atoms[1]));
079    }
080
081    @Override
082    public void remove() {
083    }
084
085    public void close() {
086      if (r != null) {
087        try {
088          r.close();
089        } catch (IOException ignored) {
090        }
091      }
092    }
093  }
094}
