001package org.gbif.common.parsers.geospatial;
002
003/**
004 * This enumerates all the possible issues we can detect with latitude and longitude pairs.
005 */
006public enum GeospatialIssue {
007
008  NO_ISSUES(0x00),
009
010  /**
011   * Set if latitude appears to have the wrong sign
012   */
013  PRESUMED_NEGATED_LATITUDE(0x01),
014
015  /**
016   * Set if longitude appears to have the wrong sign
017   */
018  PRESUMED_NEGATED_LONGITUDE(0x02),
019
020  /**
021   * Set if latitude and longitude appears to have been switched.
022   */
023  PRESUMED_INVERTED_COORDINATES(0x04),
024
025  /**
026   * Set if coordinates are (0,0).
027   */
028  ZERO_COORDINATES(0x08),
029
030  /**
031   * Set if coordinates not in range -90 to 90 and -180 to 180.
032   */
033  COORDINATES_OUT_OF_RANGE(0x10),
034
035  /**
036   * Set if country name is not understood
037   */
038  COUNTRY_COORDINATE_MISMATCH(0x20);
039
040  private final int issue;
041
042  GeospatialIssue(int issue) {
043    this.issue = issue;
044  }
045
046  public int getIssueCode() {
047    return issue;
048  }
049}
