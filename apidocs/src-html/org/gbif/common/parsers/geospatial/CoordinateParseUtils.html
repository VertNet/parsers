001package org.gbif.common.parsers.geospatial;
002
003import org.gbif.api.vocabulary.OccurrenceIssue;
004import org.gbif.common.parsers.core.Parsable;
005import org.gbif.common.parsers.core.ParseResult;
006
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010/**
011 * Utilities for assisting in the parsing of latitude and longitude strings into Decimals.
012 */
013public class CoordinateParseUtils {
014
015  private CoordinateParseUtils() {
016    throw new UnsupportedOperationException("Can't initialize class");
017  }
018
019  private static final Logger LOG = LoggerFactory.getLogger(CoordinateParseUtils.class);
020
021  /**
022   * This parses string representations of latitude and longitude values. It tries its best to interpret the values and
023   * indicates any problems in its result as {@link org.gbif.api.vocabulary.OccurrenceIssue}.
024   * When the {@link ParseResult.STATUS} is FAIL the payload will be null and one or more issues should be set
025   * in {@link ParseResult#getIssues()}.
026   *
027   * Coordinate precision will be 5 decimals at most, any more precise values will be rounded.
028   *
029   * @param latitude  The decimal latitude
030   * @param longitude The decimal longitude
031   *
032   * @return The parse result
033   */
034  public static ParseResult<LatLng> parseLatLng(final String latitude, final String longitude) {
035    return new Parsable<LatLng>() {
036      @Override
037      public ParseResult<LatLng> parse(String v) {
038        Double lat;
039        Double lng;
040        try {
041          lat = roundTo5decimals(Double.parseDouble(latitude));
042          lng = roundTo5decimals(Double.parseDouble(longitude));
043        } catch (NumberFormatException e) {
044          return ParseResult.error(e);
045        }
046
047        // 0,0 is too suspicious
048        if (Double.compare(lat, 0) == 0 && Double.compare(lng, 0) == 0) {
049          return ParseResult
050            .success(ParseResult.CONFIDENCE.POSSIBLE, new LatLng(0, 0), OccurrenceIssue.ZERO_COORDINATE);
051        }
052
053        // if everything falls in range
054        if (Double.compare(lat, 90) <= 0 && Double.compare(lat, -90) >= 0 && Double.compare(lng, 180) <= 0
055            && Double.compare(lng, -180) >= 0) {
056          return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new LatLng(lat, lng));
057        }
058
059        // if lat is out of range, but in range of the lng,
060        // assume swapped coordinates.
061        // note that should we desire to trust the following records, we would need to clear the flag for the records to
062        // appear in
063        // search results and maps etc. however, this is logic decision, that goes above the capabilities of this method
064        if (Double.compare(lat, 90) > 0 || Double.compare(lat, -90) < 0) {
065
066          // try and swap
067          if (Double.compare(lng, 90) <= 0 && Double.compare(lng, -90) >= 0 && Double.compare(lat, 180) <= 0
068              && Double.compare(lat, -180) >= 0) {
069            return ParseResult.fail(new LatLng(lat, lng), OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE);
070          }
071        }
072
073        // then something is out of range
074        return ParseResult.fail(OccurrenceIssue.COORDINATES_OUT_OF_RANGE);
075      }
076    }.parse(null);
077  }
078
079  // round to 5 decimals (~1m precision) since no way we're getting anything legitimately more precise
080  private static Double roundTo5decimals(Double x) {
081    return x == null ? null : Math.round(x * Math.pow(10, 5)) / Math.pow(10, 5);
082  }
083}
