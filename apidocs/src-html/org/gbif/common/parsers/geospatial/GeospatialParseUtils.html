097  /**
098   * This parses string representations of latitude and longitude values. It tries its best to interpret the values and
099   * indicates any problems in its result as {@link org.gbif.api.vocabulary.OccurrenceIssue}.
100   * When the {@link ParseResult.STATUS} is either SUCCESS or FAIL the payload will always be non-null and the
101   * {@link LatLngIssue#getIssue()} method should return any issues there were. In case the issue is ERROR the payload
102   * will usually be {@code null}.
103   *
104   * @param latitude  The verbatim latitude
105   * @param longitude The verbatim longitude
106   *
107   * @return The parse result
108   */
109  public static ParseResult<LatLngIssue> parseLatLng(final String latitude, final String longitude) {
110    return new Parsable<Void, LatLngIssue>() {
111      @Override
112      public ParseResult<LatLngIssue> parse(Void v) {
113        Double lat;
114        Double lng;
115        try {
116          lat = Double.parseDouble(latitude);
117          lng = Double.parseDouble(longitude);
118        } catch (NumberFormatException e) {
119          return ParseResult.error(e);
120        }
121
122        // 0,0 is too suspicious
123        if (Double.compare(lat, 0) == 0 && Double.compare(lng, 0) == 0) {
124          return ParseResult
125            .success(ParseResult.CONFIDENCE.POSSIBLE, new LatLngIssue(0, 0, OccurrenceIssue.ZERO_COORDINATE));
126        }
127
128        // if everything falls in range
129        if (Double.compare(lat, 90) <= 0 && Double.compare(lat, -90) >= 0 && Double.compare(lng, 180) <= 0
130            && Double.compare(lng, -180) >= 0) {
131          return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new LatLngIssue(lat, lng));
132        }
133
134        // if lat is out of range, but in range of the lng,
135        // assume swapped coordinates.
136        // note that should we desire to trust the following records, we would need to clear the flag for the records to
137        // appear in
138        // search results and maps etc. however, this is logic decision, that goes above the capabilities of this method
139        if (Double.compare(lat, 90) > 0 || Double.compare(lat, -90) < 0) {
140
141          // try and swap
142          if (Double.compare(lng, 90) <= 0 && Double.compare(lng, -90) >= 0 && Double.compare(lat, 180) <= 0
143              && Double.compare(lat, -180) >= 0) {
144            return ParseResult.fail(new LatLngIssue(lat, lng, OccurrenceIssue.PRESUMED_SWAPPED_COORDINATE));
145          }
146        }
147
148        // then something is out of range
149        return ParseResult.fail(new LatLngIssue(OccurrenceIssue.COORDINATES_OUT_OF_RANGE));
150      }
151    }.parse(null);
152  }

154  /**
155   * Attempts to parse the depth information provided, and return the depth in metres. This unit (metres) is
156   * defined by the Darwin Core standard for depth and used in our Occurrence class.
157   *
158   * @param minimum           verbatim minimum depth
159   * @param maximum           verbatim maximum depth
160   * @param precisionAsString verbatim precision
161   *
162   * @return The parse result
163   */
164  public static ParseResult<IntPrecisionIssue> parseDepth(final String minimum, final String maximum,
165    final String precisionAsString) {
166    return new Parsable<Void, IntPrecisionIssue>() {
167      @Override
168      public ParseResult<IntPrecisionIssue> parse(Void v) {
169
170        Set<OccurrenceIssue> issues = Sets.newHashSet();
171
172        ParseStringToDouble source = new ParseStringToDouble(precisionAsString);
173        getDepthMeasurement(source);
174        Double precision = source.parsed;
175        issues.addAll(source.issues);
176
177        source = new ParseStringToDouble(minimum);
178        getDepthMeasurement(source);
179        Double min = source.parsed;
180        issues.addAll(source.issues);
181
182        source = new ParseStringToDouble(maximum);
183        getDepthMeasurement(source);
184        Double max = source.parsed;
185        issues.addAll(source.issues);
186
187        LOGGER.debug("minVerbatim[{}], minParsed[{}], maxVerbatim[{}], maxParsed[{}], precisionVerbatim[{}], "
188                     + "precisionParsed[{}]", new Object[] {minimum, min, maximum, max, precisionAsString, precision});
189
190        if (min == null && max == null) {
191          // both are null
192          return ParseResult.fail();
193        }
194
195        min = min == null ? max : min;
196        max = max == null ? min : max;
197
198        if (precision != null) {
199          min -= precision;
200          max += precision;
201        }
202
203        Double depth = max + min == 0 ? 0.0d : (max + min) / 2;
204
205        // record the number of record with depth 0
206        if (min > max) {
207          issues.add(OccurrenceIssue.DEPTH_MIN_MAX_SWAPPED);
208        }
209
210        // record the number of record with depth 0
211        if (depth > OUT_OF_RANGE_DEPTH) {
212          issues.add(OccurrenceIssue.DEPTH_OUT_OF_RANGE);
213        }
214
215        long depthInMetres = Math.round(depth);
216        if (depthInMetres <= MAX_RECORD_DEPTH_IN_METRES
217            && depthInMetres >= MIN_RECORD_DEPTH_IN_METERS) {
218          if (precision != null) {
219            return ParseResult
220              .success(ParseResult.CONFIDENCE.DEFINITE,
221                       new IntPrecisionIssue(roundedInt(depth), roundedInt(precision), issues));
222          } else {
223            return ParseResult
224              .success(ParseResult.CONFIDENCE.DEFINITE,
225              new IntPrecisionIssue(roundedInt(depth), null, issues));
226          }
227        }
228
229        // we get no value during parsing
230        return ParseResult.success(ParseResult.CONFIDENCE.DEFINITE, new IntPrecisionIssue(null, null, issues));
231      }
232
233    }.parse(null);
234  }

236  private static Integer roundedInt(Double x) {
237    if (x == null) return null;
238
239    Long xl = Math.round(x);
240    return xl.intValue();
241  }

243  /**
244   * Attempts to parse the depth information provided, and return the altitude in meters. This unit (metres) is defined
245   * by the previous implementation of the portal and this implementation is a port of the existing code
246   *
247   * @param minimum           verbatim minimum altitude
248   * @param maximum           verbatim maximum altitude
249   * @param precisionAsString verbatim precision
250   *
251   * @return The parse result
252   */
253  public static ParseResult<IntPrecisionIssue> parseAltitude(final String minimum, final String maximum,
254    final String precisionAsString) {
255    return new Parsable<Void, IntPrecisionIssue>() {
256      @Override
257      public ParseResult<IntPrecisionIssue> parse(Void v) {
258
259        Set<OccurrenceIssue> issues = EnumSet.noneOf(OccurrenceIssue.class);
260
261        ParseStringToDouble source = new ParseStringToDouble(minimum);
262        getAltitudeMeasurement(source);
263        Double min = source.parsed;
264        issues.addAll(source.issues);
265
266        source = new ParseStringToDouble(maximum);
267        getAltitudeMeasurement(source);
268        Double max = source.parsed;
269        issues.addAll(source.issues);
270
271        // parse the altitude precision
272        Double precision = null;
273        if (precisionAsString != null) {
274          String altitudePrecision = removeMeasurementMarkers(precisionAsString);
275
276          source = new ParseStringToDouble(altitudePrecision);
277          getAltitudeMeasurement(source);
278          precision = source.parsed;
279          issues.addAll(source.issues);
280
281          try {
282            if (precision != null && min != null && max != null) {
283              min -= precision;
284              max += precision;
285            }
286          } catch (NumberFormatException e) {
287            // TODO: Log invalid or unparsable altitude
288          }
289        }
290        LOGGER.debug("minVerbatim[{}], minParsed[{}], maxVerbatim[{}], maxParsed[{}], precisionVerbatim[{}], "
291                     + "precisionParsed[{}]", new Object[] {minimum, min, maximum, max, precisionAsString, precision});
292
293        // determine the altitude
294        Integer altitude = null;
295        if (max != null && min != null) {
296          // avoid divide by 0
297          if (max + min == 0) {
298            altitude = 0;
299
300          } else if (max == 0 && min > 0) {
301            // if max is supplied as 0 and min is above
302            altitude = roundedInt(min);
303          } else {
304            // average of the 2
305            altitude = roundedInt((max + min) / 2);
306          }
307        } else if (min != null) {
308          // if only min altitude supplied, use this
309          altitude = roundedInt(min);
310
311        } else if (max != null) {
312          // if only max altitude supplied, use this
313          altitude = roundedInt(max);
314        }
315
316        // record the number of records with altitude 0
317        if (altitude != null && altitude == 0) {
318          // TODO: Log "number of records marker with altitude 0"
319        }
320
321        // record the number of records with altitude out of range
322        if (altitude != null && (altitude > OUT_OF_RANGE_MAX_ALTITUDE || altitude < OUT_OF_RANGE_MIN_ALTITUDE)) {
323          issues.add(OccurrenceIssue.ALTITUDE_OUT_OF_RANGE);
324        }
325
326        // record the number of records with erroneous altitudes
327        if (altitude != null && (altitude == -9999 || altitude == 9999)) {
328          issues.add(OccurrenceIssue.ALTITUDE_UNLIKELY);
329        }
330
331        // record the number of records with min/max altitude transposed
332        if (min != null && max != null && min > max && max != 0) {
333          issues.add(OccurrenceIssue.ALTITUDE_MIN_MAX_SWAPPED);
334        }
335
336        Integer precisionAsInt = null;
337        if (precision != null) {
338          precisionAsInt = roundedInt(precision);
339        }
340        if (altitude != null && altitude <= MAX_TO_RECORD_ALTITUDE_IN_METRES && altitude >= MIN_TO_RECORD_ALTITUDE_IN_METRES) {
341          return ParseResult
342            .success(ParseResult.CONFIDENCE.DEFINITE, new IntPrecisionIssue(altitude, precisionAsInt, issues));
343        } else if (altitude != null) {
344          return ParseResult
345            .success(ParseResult.CONFIDENCE.DEFINITE, new IntPrecisionIssue(null, precisionAsInt, issues));
346        } else {
347          return ParseResult.fail();
348        }
349      }
350    }.parse(null);
351  }

353  /**
354   * Retrieve an altitude measurement, reporting any faults in the data. Handles ranges and converts feet to metres.
355   *
356   * @param source String to parse
357   */
358  private static void getAltitudeMeasurement(ParseStringToDouble source) {
359    if (source.verbatim == null) {
360      source.parsed = null;
361      return;
362    }
363
364    Double altitudeAsDouble = null;
365    try {
366      boolean containsNonnumeric = MEASURE_MARKER_PATTERN.matcher(source.verbatim).matches();
367
368      // if contains non numeric chars, check for range, remove chars and try to parse number
369      if (containsNonnumeric) {
370        source.addIssue(OccurrenceIssue.ALTITUDE_NON_NUMERIC);
371        boolean isInFeet = FEET_MARKER_PATTERN.matcher(source.verbatim).matches();
372        boolean isInInches = INCHES_MARKER_PATTERN.matcher(source.verbatim).matches();
373
374        // TODO: Log there is a problem with this value
375
376        // handle 6-7m (range) values
377        if (SEP_MARKER_PATTERN.matcher(source.verbatim).matches()) {
378          // we have been given a range
379          try {
380            String min = source.verbatim.substring(0, source.verbatim.indexOf('-')).trim();
381            min = removeMeasurementMarkers(min);
382            String max = source.verbatim.substring(source.verbatim.indexOf('-') + 1).trim();
383            max = removeMeasurementMarkers(max);
384
385            // TODO Log that field contains range in single field
386
387            Double minDouble = Double.parseDouble(min);
388            Double maxDouble = Double.parseDouble(max);
389
390            if (minDouble != 0 && maxDouble != 0 && maxDouble - minDouble != 0) {
391              altitudeAsDouble = (maxDouble + minDouble) / 2;
392            }
393          } catch (NumberFormatException ignored) {
394          }
395        } else {
396          source.verbatim = removeMeasurementMarkers(source.verbatim);
397          altitudeAsDouble = Double.parseDouble(source.verbatim);
398        }
399
400        if (altitudeAsDouble != null) {
401          // convert to metric
402          if (isInFeet || isInInches) {
403            source.addIssue(OccurrenceIssue.ALTITUDE_PRESUMED_IN_FEET);
404          }
405          if (isInFeet) {
406            altitudeAsDouble = convertFeetToMetres(altitudeAsDouble);
407          } else if (isInInches) {
408            altitudeAsDouble = convertInchesToMetres(altitudeAsDouble);
409          }
410        }
411      } else {
412        altitudeAsDouble = Double.parseDouble(source.verbatim);
413      }
414    } catch (NumberFormatException e) {
415      LOGGER.debug(e.getMessage(), e);
416    }
417    source.parsed = altitudeAsDouble;
418  }

488  /**
489   * Remove "m" etc.
490   *
491   * @param s to remove measurement markers from.
492   *
493   * @return a new string with all measurements removed (i.e. replaced by the empty string)
494   */
495  static String removeMeasurementMarkers(String s) {
496    if (s == null) return null;
497    return REMOVE_MEASURE_MARKER_PATTERN.matcher(s).replaceAll("");
498  }
499
500  static double convertInchesToMetres(double inches) {
501    return inches * INCHES_TO_METRES;
502  }
503
504  static double convertFeetToMetres(double feet) {
505    return feet * FEET_TO_METRES;
506  }

508  /**
509   * Utility container
510   */
511  static class ParseStringToDouble {
512
513    String verbatim;
514    Double parsed;
515    Set<OccurrenceIssue> issues = Sets.newHashSet();
516

518      this.verbatim = verbatim;
519    }
520
521    void addIssue(OccurrenceIssue issue) {
522      issues.add(issue);
523    }
