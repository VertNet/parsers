001package org.gbif.common.parsers.geospatial;
002
003import org.gbif.api.vocabulary.OccurrenceIssue;
004
005import com.google.common.base.Objects;
006
007/**
008 * Simple container class for an interpreted coordinate and a single interpretation issue.
009 * The issue will be null if non exists.
010 */
011public class LatLngIssue {
012
013  private final Double lat;
014  private final Double lng;
015  private final OccurrenceIssue issue;
016
017  public LatLngIssue(double lat, double lng, OccurrenceIssue issue) {
018    this.lat = lat;
019    this.lng = lng;
020    this.issue = issue;
021  }
022
023  public LatLngIssue(OccurrenceIssue issue) {
024    lat = null;
025    lng = null;
026    this.issue = issue;
027  }
028
029  public LatLngIssue(double lat, double lng) {
030    this.lat = lat;
031    this.lng = lng;
032    this.issue = null;
033  }
034
035  public Double getLat() {
036    return lat;
037  }
038
039  public Double getLng() {
040    return lng;
041  }
042
043  public OccurrenceIssue getIssue() {
044    return issue;
045  }
046
047
048  @Override
049  public String toString() {
050    return Objects.toStringHelper(this)
051      .add("lat", lat)
052      .add("lng", lng)
053      .add("issue", issue)
054      .toString();
055  }
056
057  @Override
058  public int hashCode() {
059    return Objects.hashCode(lat, lng, issue);
060  }
061
062  @Override
063  public boolean equals(Object object) {
064    if (this == object) {
065      return true;
066    }
067    if (!(object instanceof LatLngIssue)) {
068      return false;
069    }
070
071    LatLngIssue that = (LatLngIssue) object;
072    return Objects.equal(this.lat, that.lat)
073           && Objects.equal(this.lng, that.lng)
074           && Objects.equal(this.issue, that.issue);
075  }
076}
