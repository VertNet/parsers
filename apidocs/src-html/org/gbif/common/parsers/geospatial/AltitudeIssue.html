001package org.gbif.common.parsers.geospatial;
002
003public enum AltitudeIssue {
004
005  /**
006   * Set if altitude is >10000m or <-1000m
007   */
008  GEOSPATIAL_ALTITUDE_OUT_OF_RANGE(0x80),
009
010  /**
011   * Set if altitude is -9999 or some other bogus value
012   */
013  GEOSPATIAL_PRESUMED_ERRONOUS_ALTITUDE(0x100),
014
015  /**
016   * Set if supplied min > max altitude
017   */
018  GEOSPATIAL_PRESUMED_MIN_MAX_ALTITUDE_REVERSED(0x200),
019
020  /**
021   * Set if supplied altitude is in feet instead of metric
022   */
023  GEOSPATIAL_PRESUMED_ALTITUDE_IN_FEET(0x2000),
024
025  /**
026   * Set if altitude is a non numeric value
027   */
028  GEOSPATIAL_PRESUMED_ALTITUDE_NON_NUMERIC(0x4000);
029
030  private final int issue;
031
032  AltitudeIssue(int issue) {
033    this.issue = issue;
034  }
035
036  public int getIssueCode() {
037    return issue;
038  }
039}
