001package org.gbif.common.parsers.basisofrecord;
002
003import org.gbif.api.vocabulary.BasisOfRecord;
004import org.gbif.common.parsers.FileBasedDictionaryParser;
005
006import java.io.InputStream;
007
008/**
009 * Singleton implementation of the dictionary that uses the file /dictionaries/parse/basisOfRecord.txt.
010 */
011public class BasisOfRecordParser extends FileBasedDictionaryParser {
012
013  private static BasisOfRecordParser singletonObject = null;
014
015  private BasisOfRecordParser(boolean caseSensitive, InputStream... file) {
016    super(caseSensitive, file);
017    addEnumValues();
018  }
019
020  public static BasisOfRecordParser getInstance()
021    throws ClassCastException, AbstractMethodError, ArithmeticException, ArrayIndexOutOfBoundsException {
022    synchronized (BasisOfRecordParser.class) {
023      if (singletonObject == null) {
024        singletonObject = new BasisOfRecordParser(false,
025          BasisOfRecordParser.class.getResourceAsStream("/dictionaries/parse/basisOfRecord.txt"));
026      }
027    }
028    return singletonObject;
029  }
030
031  private void addEnumValues() {
032    for (BasisOfRecord bor : BasisOfRecord.values()) {
033      add(bor.name(), bor.name());
034    }
035  }
036}
