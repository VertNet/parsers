232  /**
233   * Carefully normalizes a scientific name trying to maintain the original as close as possible.
234   * In particular the string is normalized by:
235   * - adding commas in front of years
236   * - trims whitespace around hyphens
237   * - unescapes unicode chars \\uhhhh, \\nnn, \xhh
238   * - pads whitespace around &
239   * - adds whitespace after dots following a genus abbreviation or rank marker
240   * - keeps whitespace before opening and after closing brackets
241   * - removes whitespace inside brackets
242   * - removes whitespace before commas
243   * - normalized hybrid marker to be the ascii multiplication sign
244   * - removes whitespace between hybrid marker and following name part in case it is NOT a hybrid formula
245   * - trims the string and replaces multi whitespace with single space
246   * - capitalizes all only uppercase words (authors are often found in upper case only)
247   *
248   * @param name To normalize
249   *
250   * @return The normalized name
251   */
252  public static String normalize(String name) {
253    if (name == null) {
254      return null;
255    }
256    name = org.gbif.utils.text.StringUtils.unescapeUnicodeChars(name);
257
258    // normalise usage of dots after abbreviated genus and rank marker
259    Matcher m = NORM_DOTS.matcher(name);
260    if (m.find()) {
261      name = m.replaceAll("$1. ");
262    }
263    // use commas before years
264    // ICZN §22A.2 http://www.iczn.org/iczn/includes/page.jsp?article=22&nfv=
265    m = COMMA_BEFORE_YEAR.matcher(name);
266    if (m.find()) {
267      name = m.replaceAll("$1, $2");
268    }
269    // no whitespace around hyphens
270    name = NORM_HYPHENS.matcher(name).replaceAll("-");
271    // use whitespace with &
272    name = NORM_AMPERSAND_WS.matcher(name).replaceAll(" & ");
273
274    // whitespace before and after brackets, keeping the bracket style
275    m = NORM_BRACKETS_OPEN.matcher(name);
276    if (m.find()) {
277      name = m.replaceAll(" $1");
278    }
279    m = NORM_BRACKETS_CLOSE.matcher(name);
280    if (m.find()) {
281      name = m.replaceAll("$1 ");
282    }
283    // remove whitespace before commas and replace double commas with one
284    m = NORM_COMMAS.matcher(name);
285    if (m.find()) {
286      name = m.replaceAll(", ");
287    }
288    // normalize hybrid markers
289    m = NORM_HYBRIDS_GENUS.matcher(name);
290    if (m.find()) {
291      name = m.replaceFirst("×$1");
292    }
293    m = NORM_HYBRIDS_EPITH.matcher(name);
294    if (m.find()) {
295      name = m.replaceFirst("$1 ×$2");
296    }
297    m = NORM_HYBRIDS_FORM.matcher(name);
298    if (m.find()) {
299      name = m.replaceAll(" × ");
300    }
301    // capitalize all entire upper case words
302    m = NORM_UPPERCASE_WORDS.matcher(name);
303    while (m.find()) {
304      name = name.replaceFirst(m.group(0), m.group(1) + m.group(2).toLowerCase());
305    }
306
307    // finally whitespace and trimming
308    name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
309    return StringUtils.trimToEmpty(name);
310  }

312  /**
313   * Does the same as a normalize and additionally removes all ( ) and "und" etc
314   * Checks if a name starts with a blacklisted name part like "Undetermined" or "Uncertain" and only returns the
315   * blacklisted word in that case
316   * so its easy to catch names with blacklisted name parts.
317   *
318   * @param name To normalize
319   *
320   * @return The normalized name
321   */
322  protected static String normalizeStrong(String name) {
323    if (name == null) {
324      return null;
325    }
326    // normalize all quotes to single "
327    name = NORM_QUOTES.matcher(name).replaceAll("'");
328    // enclosing quotes
329    name = REPLACE_QUOTES.matcher(name).replaceAll("");
330    // no question marks after words (after years they should remain!)
331    Matcher m = NO_Q_MARKS.matcher(name);
332    if (m.find()) {
333      name = m.replaceAll("$1");
334    }
335
336    // remove prefixes
337    name = NORM_PREFIXES.matcher(name).replaceAll("");
338
339    // remove brackets inside the genus, the kind taxon finder produces
340    m = NORM_TF_GENUS.matcher(name);
341    if (m.find()) {
342      name = m.replaceAll("$1$2 ");
343    }
344    // replace imprint years. See ICZN §22A.2.3 http://www.iczn.org/iczn/index.jsp?nfv=&article=22
345    // Ctenotus alacer Storr, 1970 ["1969"] -> Ctenotus alacer Storr, 1970
346    // C. Flahault 1887 ("1886-1888") -> C. Flahault 1887
347    m = NORM_IMPRINT_YEAR.matcher(name);
348    if (m.find()) {
349      // System.out.println(m.group(0));
350      // System.out.println(m.group(1));
351      name = m.replaceAll("$1");
352    }
353
354    // replace bibliographic in authorship
355    name = NORM_IN_COMMA.matcher(name).replaceFirst(" in ");
356    /*
357     * m = NORM_IN_BIB.matcher(name);
358     * if (m.find()) {
359     * // keep year if it only exists in IN reference
360     * Matcher mIN = EXTRACT_YEAR.matcher(m.group(0));
361     * name = m.replaceAll("");
362     * Matcher mNAME = EXTRACT_YEAR.matcher(name);
363     * if (mIN.find() && !mNAME.find()) {
364     * name = name + ", " + mIN.group(1);
365     * }
366     * }
367     */
368
369    // This is redundant, as it is done in the regular normalize function already
370    // BUT somehow all upper case authors slow down parsing so much that it can even come to an hold in the next step
371    // so we pay the price and do it twice
372    // capitalize all entire upper case words
373    m = NORM_UPPERCASE_WORDS.matcher(name);
374    while (m.find()) {
375      name = name.replaceFirst(m.group(0), m.group(1) + m.group(2).toLowerCase());
376    }
377
378    /*
379     * ICBN §46.2, Note 1.
380     * When authorship of a name differs from authorship of the publication in which it was validly published, both are
381     * sometimes cited,
382     * connected by the word "in". In such a case, "in" and what follows are part of a bibliographic citation and are
383     * better omitted
384     * unless the place of publication is being cited.
385     */
386
387    // normalise original name authorship, putting author AND year in brackets
388    // with long authorships this gets slow. Test if year exists first:
389    m = EXTRACT_YEAR.matcher(name);
390    if (m.find() && name.length() < 80) {
391      m = NORM_ORIG_AUTH.matcher(name);
392      if (m.find()) {
393        name = m.replaceAll("($1 $2)");
394      }
395      m = NORM_ORIG_AUTH2.matcher(name);
396      if (m.find()) {
397        name = m.replaceAll("($1 $2)");
398      }
399    }
400
401    // replace square brackets, keeping content (or better remove all within?)
402    name = NORM_NO_SQUARE_BRACKETS.matcher(name).replaceAll(" $1 ");
403    // replace different kind of brackets with ()
404    name = NORM_BRACKETS_OPEN_STRONG.matcher(name).replaceAll(" (");
405    name = NORM_BRACKETS_CLOSE_STRONG.matcher(name).replaceAll(") ");
406    // normalise different usages of ampersand, and, et &amp;
407    name = NORM_AND.matcher(name).replaceAll(" & ");
408    // but keep "et al." instead of "& al."
409    name = NORM_ET_AL.matcher(name).replaceAll(" et al.");
410
411    // // add commas between authors in space delimited list
412    // m = NORM_AUTH_DELIMIT.matcher(name);
413    // if (m.find()){
414    // name = m.replaceAll("$1, $2");
415    // }
416    // Bryozoan indet. 1
417    // Bryozoa sp. 2
418    // Bryozoa sp. E
419
420    name = NORM_SUFFIXES.matcher(name).replaceAll("");
421
422    // add parenthesis around subgenus if missing
423    m = NORM_SUBGENUS.matcher(name);
424    if (m.find()) {
425      name = m.replaceAll("$1 ($2) $3");
426    }
427
428    // finally whitespace and trimming
429    name = normalize(name);
430    // name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
431    return StringUtils.trimToEmpty(name);
432  }

434  /**
435   * basic careful cleaning, trying to preserve all parsable name parts
436   */
437  protected static String preClean(String name) {
438    // unescape unicode
439    name = org.gbif.utils.text.StringUtils.unescapeUnicodeChars(name);
440    // remove bad whitespace in html entities
441    Matcher m = XML_ENTITY_STRIP.matcher(name);
442    if (m.find()) {
443      name = m.replaceAll("&$1;");
444    }
445    // unescape html entities
446    name = StringEscapeUtils.unescapeHtml4(name);
447    // finally remove still existing bad ampersands missing the closing ;
448    name = AMPERSAND_ENTITY.matcher(name).replaceAll("& ");
449    // replace xml tags
450    name = XML_TAGS.matcher(name).replaceAll("");
451    // trim
452    name = name.trim();
453    // remove quotes in beginning and matching ones at the end
454    for (char c : QUOTES) {
455      int idx = 0;
456      while (idx < name.length() && (c == name.charAt(idx) || Character.isWhitespace(name.charAt(idx)))) {
457        idx++;
458      }
459      if (idx > 0) {
460        // check if we also find this char at the end
461        int end = 0;
462        while (c == name.charAt(name.length() - 1 - end) && (name.length() - idx - end) > 0) {
463          end++;
464        }
465        name = name.substring(idx, name.length() - end);
466      }
467    }
468    name = NORM_WHITESPACE.matcher(name).replaceAll(" ");
469    return StringUtils.trimToEmpty(name);
470  }

472  /**
473   * Fully parse the supplied name also trying to extract authorships, a conceptual sec reference, remarks or notes
474   * on the nomenclatural status. In some cases the authorship parsing proves impossible and this nameparser will
475   * return null.
476   *
477   * For strings which are no scientific names and scientific names that cannot be expressed by the ParsedName class
478   * the parser will throw an UnparsableException with a given NameType and the original, unparsed name. This is the
479   * case for all virus names and proper hybrid formulas, so make sure you catch and process this exception.
480   *
481   * @param scientificName the full scientific name to parse
482   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
483   *
484   * @throws org.gbif.nameparser.UnparsableException
485   */
486  public ParsedName parse(final String scientificName, @Nullable Rank rank) throws UnparsableException {
487    if (Strings.isNullOrEmpty(scientificName)) {
488      throw new UnparsableException(NameType.NO_NAME, scientificName);
489    }
490    long start = 0;
491    if (LOG.isDebugEnabled()) {
492      start = System.currentTimeMillis();
493    }
494
495    ParsedName pn = new ParsedName();
496    pn.setScientificName(scientificName);
497
498    // clean name, removing seriously wrong things
499    String name = preClean(scientificName);
500
501    // remove extinct markers
502    name = EXTINCT_PATTERN.matcher(name).replaceFirst("");
503
504    // before any cleaning test for properly quoted candidate names
505    Matcher m = IS_CANDIDATUS_QUOTE_PATTERN.matcher(scientificName);
506    if (m.find()) {
507      pn.setType(NameType.CANDIDATUS);
508      name = m.replaceFirst(m.group(2));
509    }
510
511    // normalize bacterial rank markers
512    name = TYPE_TO_VAR.matcher(name).replaceAll("$1var");
513
514    // parse out species/strain names with numbers found in Genebank/EBI names, e.g. Advenella kashmirensis W13003
515    m = STRAIN.matcher(name);
516    if (m.find()) {
517      name = m.replaceFirst(m.group(1));
518      pn.setType(NameType.INFORMAL);
519      pn.setStrain(m.group(2));
520      LOG.debug("Strain: {}", m.group(2));
521    }
522
523    // detect unparsable names
524    if (PLACEHOLDER.matcher(name).find()) {
525      throw new UnparsableException(NameType.PLACEHOLDER, scientificName);
526    }
527
528    if (IS_VIRUS_PATTERN.matcher(name).find() || IS_VIRUS_PATTERN_CASE_SENSITIVE.matcher(name).find()) {
529      throw new UnparsableException(NameType.VIRUS, scientificName);
530    }
531
532    // detect RNA/DNA gene/strain names and flag as informal
533    if (IS_GENE.matcher(name).find()) {
534      pn.setType(NameType.INFORMAL);
535    }
536
537    // normalise name
538    name = normalize(name);
539    if (Strings.isNullOrEmpty(name)) {
540      throw new UnparsableException(NameType.NO_NAME, scientificName);
541    }
542
543    // parse cultivar names first BEFORE we strongly normalize
544    // this will potentially remove quotes needed to find cultivar names
545    m = CULTIVAR.matcher(name);
546    if (m.find()) {
547      pn.setCultivarEpithet(m.group(1));
548      name = m.replaceFirst(" ");
549      pn.setType(NameType.CULTIVAR);
550      pn.setRank(Rank.CULTIVAR);
551      LOG.debug("Cultivar: {}", pn.getCultivarEpithet());
552    }
553
554    // name without any latin char letter at all?
555    if (NO_LETTERS.matcher(name).find()) {
556      throw new UnparsableException(NameType.NO_NAME, scientificName);
557    }
558
559    if (HYBRID_FORMULA_PATTERN.matcher(name).find()) {
560      throw new UnparsableException(NameType.HYBRID, scientificName);
561    }
562
563    m = IS_CANDIDATUS_PATTERN.matcher(name);
564    if (m.find()) {
565      pn.setType(NameType.CANDIDATUS);
566      name = m.replaceFirst("");
567    }
568
569    // extract nom.illeg. and other nomen status notes
570    m = EXTRACT_NOMSTATUS.matcher(name);
571    if (m.find()) {
572      pn.setNomStatus(StringUtils.trimToNull(m.group(1)));
573      name = m.replaceFirst("");
574      // if there was a rank given in the nom status populate the rank marker field
575      if (pn.getNomStatus() != null) {
576        Matcher rm = NOV_RANK_MARKER.matcher(pn.getNomStatus());
577        if (rm.find()) {
578          pn.setRankMarker(rm.group(1).trim());
579        }
580      }
581    }
582
583    // extract sec reference
584    m = EXTRACT_SENSU.matcher(name);
585    if (m.find()) {
586      pn.setSensu(StringUtils.trimToNull(m.group(1)));
587      name = m.replaceFirst("");
588    }
589    // extract other remarks
590    m = EXTRACT_REMARKS.matcher(name);
591    if (m.find()) {
592      pn.setRemarks(StringUtils.trimToNull(m.group(1)));
593      name = m.replaceFirst("");
594    }
595
596    // check for indets
597    m = RANK_MARKER_AT_END.matcher(name);
598    // f. is a marker for forms, but more often also found in authorships as "filius" - son of.
599    // so ignore those
600    if (m.find() && !(name.endsWith(" f.") || name.endsWith(" f"))) {
601      pn.setType(NameType.INFORMAL);
602      pn.setRankMarker(m.group(2));
603      name = m.replaceAll(" ");
604    }
605    m = NORM_INDET.matcher(name);
606    if (m.find()) {
607      pn.setType(NameType.INFORMAL);
608      name = m.replaceAll(" ");
609    }
610
611    name = normalizeStrong(name);
612    if (Strings.isNullOrEmpty(name)) {
613      throw new UnparsableException(NameType.DOUBTFUL, scientificName);
614    }
615
616    // remember current rank for later reuse
617    Rank origRank = pn.getRank();
618
619    // try regular parsing
620    boolean parsed = nnParser.parseNormalisedName(pn, name, rank);
621    if (!parsed) {
622      // try again with stronger normalisation, maybe that helps...
623      LOG.debug("Can't parse, use dirty normalizer");
624      final String deDirtedName = cleanStrong(name);
625      parsed = nnParser.parseNormalisedName(pn, deDirtedName, rank);
626      if (!parsed) {
627        LOG.debug("Still can't parse, try to ignore authors");
628        // try to parse canonical alone ignoring authorship as last resort
629        parsed = nnParser.parseNormalisedNameIgnoreAuthors(pn, deDirtedName, rank);
630        pn.setAuthorsParsed(false);
631        if (!parsed) {
632          // we just cant parse this one
633          // try to spot a virus name once we know its not a scientific name
634          m = IS_VIRUS_PATTERN_POSTFAIL.matcher(name);
635          if (m.find()) {
636            throw new UnparsableException(NameType.VIRUS, scientificName);
637          }
638
639          throw new UnparsableException(NameType.DOUBTFUL, scientificName);
640        }
641      }
642    }
643    if (origRank != null) {
644      pn.setRank(origRank);
645    }
646
647    // primarily make sure again we haven't parsed a non name
648    postAssertParsing(pn, scientificName, name);
649
650    // determine name type if not yet assigned
651    if (pn.getType() == null) {
652      // a doubtful name?
653      m = DOUBTFUL.matcher(scientificName);
654      if (!m.find()) {
655        pn.setType(NameType.DOUBTFUL);
656      } else {
657        m = DOUBTFUL2.matcher(scientificName);
658        if (m.find()) {
659          pn.setType(NameType.DOUBTFUL);
660        } else {
661          pn.setType(NameType.SCIENTIFIC);
662        }
663      }
664    }
665
666    LOG.debug("Parsing time: {}", (System.currentTimeMillis() - start));
667    return pn;
668  }
669
670  private void postAssertParsing(ParsedName pn, final String rawName, final String normedName) throws UnparsableException {
671    // if we only match a monomial in the 3rd pass its suspicious
672    if (pn.getGenusOrAbove() != null && !pn.isBinomial() && Character.isLowerCase(normedName.charAt(0))) {
673      // a monomial match, but it was a lower case name - doubtful at least!
674      throw new UnparsableException(NameType.DOUBTFUL, rawName);
675
676    } else if (pn.getRank() != null) {
677      if (pn.getRank().isSpeciesOrBelow() && !pn.isBinomial() && !pn.getRank().equals(Rank.CULTIVAR)) {
678        pn.setType(NameType.INFORMAL);
679      } else if (pn.getRank().equals(Rank.CULTIVAR) && pn.getCultivarEpithet() == null) {
680        pn.setType(NameType.INFORMAL);
681      } else if (pn.getRank().isInfraspecific() && pn.getInfraSpecificEpithet() == null) {
682        pn.setType(NameType.INFORMAL);
683      } else if (!pn.getRank().isSpeciesOrBelow() && pn.isBinomial()) {
684        pn.setType(NameType.DOUBTFUL);
685      }
686    }
687  }

689  /**
690   * parses the name without authorship and returns the ParsedName.canonicalName() string
691   * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
692   */
693  public String parseToCanonical(String scientificName, @Nullable Rank rank) {
694    if (Strings.isNullOrEmpty(scientificName)) {
695      return null;
696    }
697    try {
698      ParsedName pn = parse(scientificName, rank);
699      if (pn != null) {
700        return pn.canonicalName();
701      }
702    } catch (UnparsableException e) {
703      LOG.warn("Unparsable name " + scientificName + " >>> " + e.getMessage());
704    }
705    return null;
706  }
